361 Val* BroFunc::Call(val_list* args, Frame* parent) const
362  {
363 #ifdef PROFILE_BRO_FUNCTIONS
364  DEBUG_MSG("Function: %s\n", Name());
365 #endif
366 
367  SegmentProfiler(segment_logger, location);
368 
369  if ( sample_logger )
370  sample_logger->FunctionSeen(this);
371 
372  std::pair<bool, Val*> plugin_result = PLUGIN_HOOK_WITH_RESULT(HOOK_CALL_FUNCTION, HookCallFunction(this, parent, args), empty_hook_result);
373 
374  plugin_result = HandlePluginResult(plugin_result, args, Flavor());
375 
376  if( plugin_result.first )
377  {
378  Val *result = plugin_result.second;
379  return result;
380  }
381 
382  if ( bodies.empty() )
383  {
384  // Can only happen for events and hooks.
385  assert(Flavor() == FUNC_FLAVOR_EVENT || Flavor() == FUNC_FLAVOR_HOOK);
386  loop_over_list(*args, i)
387  Unref((*args)[i]);
388 
389  return Flavor() == FUNC_FLAVOR_HOOK ? new Val(true, TYPE_BOOL) : 0;
390  }
391 
392  Frame* f = new Frame(frame_size, this, args);
393 
394  // Hand down any trigger.
395  if ( parent )
396  {
397  f->SetTrigger(parent->GetTrigger());
398  f->SetCall(parent->GetCall());
399  }
400 
401  g_frame_stack.push_back(f); // used for backtracing
402 
403  if ( g_trace_state.DoTrace() )
404  {
405  ODesc d;
406  DescribeDebug(&d, args);
407 
408  g_trace_state.LogTrace("%s called: %s\n",
409  FType()->FlavorString().c_str(), d.Description());
410  }
411 
412  stmt_flow_type flow = FLOW_NEXT;
413  Val* result = 0;
414 
415  for ( size_t i = 0; i < bodies.size(); ++i )
416  {
417  if ( sample_logger )
418  sample_logger->LocationSeen(
419  bodies[i].stmts->GetLocationInfo());
420 
421  Unref(result);
422 
423  loop_over_list(*args, j)
424  {
425  Val* arg = (*args)[j];
426 
427  if ( f->NthElement(j) != arg )
428  {
429  // Either not yet set, or somebody reassigned
430  // the frame slot.
431  Ref(arg);
432  f->SetElement(j, arg);
433  }
434  }
435 
436  f->Reset(args->length());
437 
438  try
439  {
440  result = bodies[i].stmts->Exec(f, flow);
441  }
442 
443  catch ( InterpreterException& e )
444  {
445  // Already reported, but we continue exec'ing remaining bodies.
446  continue;
447  }
448 
449  if ( f->HasDelayed() )
450  {
451  assert(! result);
452  assert(parent);
453  parent->SetDelayed();
454  break;
455  }
456 
457  if ( Flavor() == FUNC_FLAVOR_HOOK )
458  {
459  // Ignore any return values of hook bodies, final return value
460  // depends on whether a body returns as a result of break statement.
461  Unref(result);
462  result = 0;
463 
464  if ( flow == FLOW_BREAK )
465  {
466  // Short-circuit execution of remaining hook handler bodies.
467  result = new Val(false, TYPE_BOOL);
468  break;
469  }
470  }
471  }
472 
473  // We have an extra Ref for each argument (so that they don't get
474  // deleted between bodies), release that.
475  loop_over_list(*args, k)
476  Unref((*args)[k]);
477 
478  if ( Flavor() == FUNC_FLAVOR_HOOK )
479  {
480  if ( ! result )
481  result = new Val(true, TYPE_BOOL);
482  }
483 
484  // Warn if the function returns something, but we returned from
485  // the function without an explicit return, or without a value.
486  else if ( FType()->YieldType() && FType()->YieldType()->Tag() != TYPE_VOID &&
487  (flow != FLOW_RETURN /* we fell off the end */ ||
488  ! result /* explicit return with no result */) &&
489  ! f->HasDelayed() )
490  reporter->Warning("non-void function returns without a value: %s",
491  Name());
492 
493  if ( result && g_trace_state.DoTrace() )
494  {
495  ODesc d;
496  result->Describe(&d);
497 
498  g_trace_state.LogTrace("Function return: %s\n", d.Description());
499  }
500 
501  g_frame_stack.pop_back();
502  Unref(f);
503 
504  return result;
505  }

664 void builtin_error(const char* msg, BroObj* arg)
665  {
666  if ( call_stack.empty() )
667  {
668  reporter->Error(msg, arg);
669  return;
670  }
671 
672  auto last_call = call_stack.back();
673 
674  if ( call_stack.size() < 2 )
675  {
676  // Don't need to check for wrapper function like "<module>::__<func>"
677  last_call.call->Error(msg, arg);
678  return;
679  }
680 
681  auto starts_with_double_underscore = [](const std::string& name) -> bool
682  { return name.size() > 2 && name[0] == '_' && name[1] == '_'; };
683  auto last_loc = last_call.call->GetLocationInfo();
684  std::string last_func = last_call.func->Name();
685 
686  auto pos = last_func.find_first_of("::");
687  std::string wrapper_func;
688 
689  if ( pos == std::string::npos )
690  {
691  if ( ! starts_with_double_underscore(last_func) )
692  {
693  last_call.call->Error(msg, arg);
694  return;
695  }
696 
697  wrapper_func = last_func.substr(2);
698  }
699  else
700  {
701  auto module_name = last_func.substr(0, pos);
702  auto func_name = last_func.substr(pos + 2);
703 
704  if ( ! starts_with_double_underscore(func_name) )
705  {
706  last_call.call->Error(msg, arg);
707  return;
708  }
709 
710  wrapper_func = module_name + "::" + func_name.substr(2);
711  }
712 
713  auto parent_call = call_stack[call_stack.size() - 2];
714  auto parent_func = parent_call.func->Name();
715 
716  if ( wrapper_func == parent_func )
717  parent_call.call->Error(msg, arg);
718  else
719  last_call.call->Error(msg, arg);
720  }

795  int num_fmt = 0;
796  while ( *fmt_str )
797  {
798  if ( *(fmt_str++) != '%' )
799  continue;
800 
801  if ( ! *fmt_str )
802  {
803  call->Error("format string ends with bare '%'");
804  return false;
805  }
806 
807  if ( *(fmt_str++) != '%' )
808  // Not a "%%" escape.
809  ++num_fmt;
810  }
811 
812  if ( args.length() != num_fmt + 1 )
813  {
814  call->Error("mismatch between format string to fmt() and number of arguments passed");
