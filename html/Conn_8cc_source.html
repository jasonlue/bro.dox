515 Val* Connection::BuildVersionVal(const char* s, int len)
516  {
517  Val* name = 0;
518  Val* major = 0;
519  Val* minor = 0;
520  Val* minor2 = 0;
521  Val* addl = 0;
522 
523  const char* last = s + len;
524  const char* e = s;
525 
526  // This is all just a guess...
527 
528  // Eat non-alpha-numerical chars.
529  for ( ; s < last && ! isalnum(*s); ++s )
530  ;
531 
532  // Leading characters are the program name.
533  // (first character must not be a digit)
534  if ( isalpha(*s) )
535  {
536  for ( e = s; e < last && ! is_version_sep(e, last); ++e )
537  ;
538 
539  if ( s != e )
540  name = new StringVal(e - s, s);
541  }
542 
543  // Find first number - that's the major version.
544  for ( s = e; s < last && ! isdigit(*s); ++s )
545  ;
546  for ( e = s; e < last && isdigit(*e); ++e )
547  ;
548 
549  if ( s != e )
550  major = new Val(atoi(s), TYPE_INT);
551 
552  // Find second number seperated only by punctuation chars -
553  // that's the minor version.
554  for ( s = e; s < last && ispunct(*s); ++s )
555  ;
556  for ( e = s; e < last && isdigit(*e); ++e )
557  ;
558 
559  if ( s != e )
560  minor = new Val(atoi(s), TYPE_INT);
561 
562  // Find second number seperated only by punctuation chars; -
563  // that's the minor version.
564  for ( s = e; s < last && ispunct(*s); ++s )
565  ;
566  for ( e = s; e < last && isdigit(*e); ++e )
567  ;
568 
569  if ( s != e )
570  minor2 = new Val(atoi(s), TYPE_INT);
571 
572  // Anything after following punctuation and until next white space is
573  // an additional version string.
574  for ( s = e; s < last && ispunct(*s); ++s )
575  ;
576  for ( e = s; e < last && ! isspace(*e); ++e )
577  ;
578 
579  if ( s != e )
580  addl = new StringVal(e - s, s);
581 
582  // If we do not have a name yet, the next alphanumerical string is it.
583  if ( ! name )
584  { // eat non-alpha-numerical characters
585  for ( s = e; s < last && ! isalpha(*s); ++s )
586  ;
587 
588  // Get name.
589  for ( e = s; e < last && (isalnum(*e) || *e == '_'); ++e )
590  ;
591 
592  if ( s != e )
593  name = new StringVal(e - s, s);
594  }
595 
596  // We need at least a name.
597  if ( ! name )
598  {
599  Unref(major);
600  Unref(minor);
601  Unref(minor2);
602  Unref(addl);
603  return 0;
604  }
605 
606  RecordVal* version = new RecordVal(software_version);
607  version->Assign(0, major ? major : new Val(-1, TYPE_INT));
608  version->Assign(1, minor ? minor : new Val(-1, TYPE_INT));
609  version->Assign(2, minor2 ? minor2 : new Val(-1, TYPE_INT));
610  version->Assign(3, addl ? addl : new StringVal(""));
611 
612  RecordVal* sw = new RecordVal(software);
613  sw->Assign(0, name);
614  sw->Assign(1, version);
615 
616  return sw;
617  }
