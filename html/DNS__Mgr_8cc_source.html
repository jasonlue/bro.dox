582 void DNS_Mgr::Resolve()
583  {
584  if ( ! nb_dns )
585  return;
586 
587  int i;
588 
589  int first_req = 0;
590  int num_pending = min(requests.length(), MAX_PENDING_REQUESTS);
591  int last_req = num_pending - 1;
592 
593  // Prime with the initial requests.
594  for ( i = first_req; i <= last_req; ++i )
595  requests[i]->MakeRequest(nb_dns);
596 
597  // Start resolving. Each time an answer comes in, we can issue a
598  // new request, if we have more.
599  while ( num_pending > 0 )
600  {
601  int status = AnswerAvailable(DNS_TIMEOUT);
602 
603  if ( status <= 0 )
604  {
605  // Error or timeout. Process all pending requests as
606  // unanswered and reprime.
607  for ( i = first_req; i <= last_req; ++i )
608  {
609  DNS_Mgr_Request* dr = requests[i];
610  if ( dr->RequestPending() )
611  {
612  AddResult(dr, 0);
613  dr->RequestDone();
614  }
615  }
616 
617  first_req = last_req + 1;
618  num_pending = min(requests.length() - first_req,
619  MAX_PENDING_REQUESTS);
620  last_req = first_req + num_pending - 1;
621 
622  for ( i = first_req; i <= last_req; ++i )
623  requests[i]->MakeRequest(nb_dns);
624 
625  continue;
626  }
627 
628  char err[NB_DNS_ERRSIZE];
629  struct nb_dns_result r;
630  status = nb_dns_activity(nb_dns, &r, err);
631  if ( status < 0 )
632  reporter->Warning(
633  "NB-DNS error in DNS_Mgr::WaitForReplies (%s)",
634  err);
635  else if ( status > 0 )
636  {
637  DNS_Mgr_Request* dr = (DNS_Mgr_Request*) r.cookie;
638  if ( dr->RequestPending() )
639  {
640  AddResult(dr, &r);
641  dr->RequestDone();
642  }
643 
644  // Room for another, if we have it.
645  if ( last_req < requests.length() - 1 )
646  {
647  ++last_req;
648  requests[last_req]->MakeRequest(nb_dns);
649  }
650  else
651  --num_pending;
652  }
653  }
654 
655  // All done with the list of requests.
656  for ( i = requests.length() - 1; i >= 0; --i )
657  delete requests.remove_nth(i);
658  }
