92 void RandTest::end(double* r_ent, double* r_chisq,
93  double* r_mean, double* r_montepicalc, double* r_scc)
94  {
95  int i;
96  double ent, chisq, scc, datasum;
97  ent = 0.0; chisq = 0.0; scc = 0.0; datasum = 0.0;
98  double prob[256]; /* Probabilities per bin for entropy */
99 
100  /* Complete calculation of serial correlation coefficient */
101  scct1 = scct1 + scclast * sccu0;
102  scct2 = scct2 * scct2;
103  scc = totalc * scct3 - scct2;
104  if (scc == 0.0)
105  scc = -100000;
106  else
107  scc = (totalc * scct1 - scct2) / scc;
108 
109  /* Scan bins and calculate probability for each bin and
110  Chi-Square distribution. The probability will be reused
111  in the entropy calculation below. While we're at it,
112  we sum of all the data which will be used to compute the
113  mean. */
114  cexp = totalc / 256.0; /* Expected count per bin */
115  for (i = 0; i < 256; i++)
116  {
117  double a = ccount[i] - cexp;
118 
119  prob[i] = ((double) ccount[i]) / totalc;
120  chisq += (a * a) / cexp;
121  datasum += ((double) i) * ccount[i];
122  }
123 
124  /* Calculate entropy */
125  for (i = 0; i < 256; i++)
126  {
127  if (prob[i] > 0.0)
128  {
129  ent += prob[i] * rt_log2(1 / prob[i]);
130  }
131  }
132 
133  /* Calculate Monte Carlo value for PI from percentage of hits
134  within the circle */
135  montepi = 4.0 * (((double) inmont) / mcount);
136 
137  /* Return results through arguments */
138  *r_ent = ent;
139  *r_chisq = chisq;
140  *r_mean = datasum / totalc;
