18 // Returns the ones-complement checksum of a chunk of b short-aligned bytes.
19 int ones_complement_checksum(const void* p, int b, uint32 sum)
20  {
21  const unsigned char* sp = (unsigned char*) p;
22 
23  b /= 2; // convert to count of short's
24 
25  /* No need for endian conversions. */
26  while ( --b >= 0 )
27  {
28  sum += *sp + (*(sp+1) << 8);
29  sp += 2;
30  }
31 
32  while ( sum > 0xffff )
33  sum = (sum & 0xffff) + (sum >> 16);
34 
35  return sum;
36  }
