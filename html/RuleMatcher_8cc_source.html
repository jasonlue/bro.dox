740  hdr_test->psets[i][j];
741 
742  assert(set->re);
743 
744  RuleEndpointState::Matcher* m =
745  new RuleEndpointState::Matcher;
746  m->state = new RE_Match_State(set->re);
747  m->type = (Rule::PatternType) i;
748  state->matchers.append(m);
749  }
750  }
751  }
752 
753  if ( ip )
754  {
755  // Descend the RuleHdrTest tree further.
756  for ( RuleHdrTest* h = hdr_test->child; h;
757  h = h->sibling )
758  {
759  bool match = false;
760 
761  // Evaluate the header test.
762  switch ( h->prot ) {
763  case RuleHdrTest::NEXT:
764  match = compare(*h->vals, ip->NextProto(), h->comp);
765  break;
766 
767  case RuleHdrTest::IP:
768  if ( ! ip->IP4_Hdr() )
769  continue;
770 
771  match = compare(*h->vals, getval((const u_char*)ip->IP4_Hdr() + h->offset, h->size), h->comp);
772  break;
773 
774  case RuleHdrTest::IPv6:
775  if ( ! ip->IP6_Hdr() )
776  continue;
777 
778  match = compare(*h->vals, getval((const u_char*)ip->IP6_Hdr() + h->offset, h->size), h->comp);
779  break;
780 
781  case RuleHdrTest::ICMP:
782  case RuleHdrTest::ICMPv6:
783  case RuleHdrTest::TCP:
784  case RuleHdrTest::UDP:
785  match = compare(*h->vals, getval(ip->Payload() + h->offset, h->size), h->comp);
786  break;
787 
788  case RuleHdrTest::IPSrc:
789  match = compare(h->prefix_vals, ip->IPHeaderSrcAddr(), h->comp);
790  break;
791 
792  case RuleHdrTest::IPDst:
793  match = compare(h->prefix_vals, ip->IPHeaderDstAddr(), h->comp);
794  break;
795 
796  default:
797  reporter->InternalError("unknown RuleHdrTest protocol type");
798  break;
799  }
800 
801  if ( match )
802  tests.append(h);
803  }
804  }
805  }
806  // Save some memory.
807  state->hdr_tests.resize(0);
808  state->matchers.resize(0);
809 
810  // Send BOL to payload matchers.
811  Match(state, Rule::PAYLOAD, (const u_char *) "", 0, true, false, false);
812 
813  return state;
814  }

816 void RuleMatcher::Match(RuleEndpointState* state, Rule::PatternType type,
817  const u_char* data, int data_len,
818  bool bol, bool eol, bool clear)
819  {
820  if ( ! state )
821  {
822  reporter->Warning("RuleEndpointState not initialized yet.");
823  return;
824  }
825 
826  // FIXME: There is probably some room for performance improvements
827  // in this method. For example, it *may* help to use an IntSet
828  // for 'accepted' (that depends on the average number of matching
829  // patterns).
830 
831  bool newmatch = false;
832 
833 #ifdef DEBUG
834  if ( debug_logger.IsEnabled(DBG_RULES) )
835  {
836  const char* s =
837  fmt_bytes((const char *) data, min(40, data_len));
838 
839  DBG_LOG(DBG_RULES, "Matching %s rules [%d,%d] on |%s%s|",
840  Rule::TypeToString(type), bol, eol, s,
841  data_len > 40 ? "..." : "");
842  }
843 #endif
844 
845  // Remember size of first non-null data.
846  if ( type == Rule::PAYLOAD )
847  {
848  bol = state->payload_size < 0;
849 
850  if ( state->payload_size <= 0 && data_len )
851  state->payload_size = data_len;
852 
853  else if ( state->payload_size < 0 )
854  state->payload_size = 0;
855  }
856 
857  // Feed data into all relevant matchers.
858  loop_over_list(state->matchers, x)
859  {
860  RuleEndpointState::Matcher* m = state->matchers[x];
861  if ( m->type == type &&
862  m->state->Match((const u_char*) data, data_len,
863  bol, eol, clear) )
864  newmatch = true;
865  }
866 
867  // If no new match found, we're already done.
868  if ( ! newmatch )
869  return;
870 
871  DBG_LOG(DBG_RULES, "New pattern match found");
872 
873  AcceptingMatchSet accepted_matches;
874 
875  loop_over_list(state->matchers, y )
876  {
877  RuleEndpointState::Matcher* m = state->matchers[y];
878  const AcceptingMatchSet& ams = m->state->AcceptedMatches();
879  accepted_matches.insert(ams.begin(), ams.end());
880  }
881 
882  // Determine the rules for which all patterns have matched.
883  // This code should be fast enough as long as there are only very few
884  // matched patterns per connection (which is a plausible assumption).
885 
886  // Find rules for which patterns have matched.
887  set<Rule*> rule_matches;
888 
889  for ( AcceptingMatchSet::const_iterator it = accepted_matches.begin();
890  it != accepted_matches.end(); ++it )
891  {
892  AcceptIdx aidx = it->first;
893  MatchPos mpos = it->second;
894 
895  Rule* r = Rule::rule_table[aidx - 1];
896 
897  if ( AllRulePatternsMatched(r, mpos, accepted_matches) )
898  rule_matches.insert(r);
899  }
900 
901  // Check which of the matching rules really belong to any of our nodes.
902 
903  for ( set<Rule*>::const_iterator it = rule_matches.begin();
904  it != rule_matches.end(); ++it )
905  {
906  Rule* r = *it;
907 
908  DBG_LOG(DBG_RULES, "Accepted rule: %s", r->id);
909 
910  loop_over_list(state->hdr_tests, k)
911  {
912  RuleHdrTest* h = state->hdr_tests[k];
913 
914  DBG_LOG(DBG_RULES, "Checking for accepted rule on HdrTest %d", h->id);
915 
916  // Skip if rule does not belong to this node.
917  if ( ! h->ruleset->Contains(r->Index()) )
918  continue;
919 
920  DBG_LOG(DBG_RULES, "On current node");
921 
922  // Skip if rule already fired for this connection.
923  if ( state->matched_rules.is_member(r->Index()) )
924  continue;
925 
926  // Remember that all patterns have matched.
927  if ( ! state->matched_by_patterns.is_member(r) )
928  {
929  state->matched_by_patterns.append(r);
930  BroString* s = new BroString(data, data_len, 0);
931  state->matched_text.append(s);
932  }
933 
934  DBG_LOG(DBG_RULES, "And has not already fired");
935  // Eval additional conditions.
936  if ( ! EvalRuleConditions(r, state, data, data_len, 0) )
937  continue;
938 
939  // Found a match.
940  ExecRuleActions(r, state, data, data_len, 0);
941  }
942  }
943  }

994 bool RuleMatcher::EvalRuleConditions(Rule* r, RuleEndpointState* state,
995  const u_char* data, int len, bool eos)
996  {
997  DBG_LOG(DBG_RULES, "Evaluating conditions for rule %s", r->ID());
998 
999  // Check for other rules which have to match first.
1000  loop_over_list(r->preconds, i)
1001  {
1002  Rule::Precond* pc = r->preconds[i];
1003 
1004  RuleEndpointState* pc_state = state;
1005 
1006  if ( pc->opposite_dir )
1007  {
1008  if ( ! state->opposite )
1009  // No rule matching for other direction yet.
1010  return false;
1011 
1012  pc_state = state->opposite;
1013  }
1014 
1015  if ( ! pc->negate )
1016  {
1017  if ( ! pc_state->matched_rules.is_member(pc->rule->Index()) )
1018  // Precond rule has not matched yet.
1019  return false;
1020  }
1021  else
1022  {
1023  // Only at eos can we decide about negated conditions.
1024  if ( ! eos )
1025  return false;
1026 
1027  if ( pc_state->matched_rules.is_member(pc->rule->Index()) )
1028  return false;
1029  }
1030  }
1031 
1032  loop_over_list(r->conditions, l)
1033  if ( ! r->conditions[l]->DoMatch(r, state, data, len) )
1034  return false;
1035 
1036  DBG_LOG(DBG_RULES, "Conditions met: MATCH! %s", r->ID());
1037  return true;

1040 void RuleMatcher::ExecRuleActions(Rule* r, RuleEndpointState* state,
1041  const u_char* data, int len, bool eos)
1042  {
1043  if ( state->opposite &&
1044  state->opposite->matched_rules.is_member(r->Index()) )
1045  // We have already executed the actions.
1046  return;
1047 
1048  state->matched_rules.append(r->Index());
1049 
1050  loop_over_list(r->actions, i)
1051  r->actions[i]->DoAction(r, state, data, len);
1052 
1053  // This rule may trigger some other rules; check them.
1054  loop_over_list(r->dependents, j)
1055  {
1056  Rule* dep = (r->dependents)[j];
1057  ExecRule(dep, state, eos);
1058  if ( state->opposite )
1059  ExecRule(dep, state->opposite, eos);
1060  }

1063 void RuleMatcher::ExecRule(Rule* rule, RuleEndpointState* state, bool eos)
1064  {
1065  // Nothing to do if it has already matched.
1066  if ( state->matched_rules.is_member(rule->Index()) )
1067  return;
1068 
1069  loop_over_list(state->hdr_tests, i)
1070  {
1071  RuleHdrTest* h = state->hdr_tests[i];
1072 
1073  // Is it on this HdrTest at all?
1074  if ( ! h->ruleset->Contains(rule->Index()) )
1075  continue;
1076 
1077  // Is it a pure rule?
1078  for ( Rule* r = h->pure_rules; r; r = r->next )
1079  if ( r == rule )
1080  { // found, so let's evaluate it
1081  ExecRulePurely(rule, 0, state, eos);
1082  return;
1083  }
1084 
1085  // It must be a non-pure rule. It can only match right now if
1086  // all its patterns are satisfied already.
1087  int pos = state->matched_by_patterns.member_pos(rule);
1088  if ( pos >= 0 )
1089  { // they are, so let's evaluate it
1090  ExecRulePurely(rule, state->matched_text[pos], state, eos);
1091  return;
1092  }
1093  }
1094  }

1270 static bool val_to_maskedval(Val* v, maskedvalue_list* append_to,
1271  vector<IPPrefix>* prefix_vector)
1272  {

1274 
1275  switch ( v->Type()->Tag() ) {

1279  break;
1280 
1281  case TYPE_BOOL:
1282  case TYPE_COUNT:
1283  case TYPE_ENUM:

1287  break;

1308  m[1] == m[0] && m[2] == m[0];

1312  {

1319  mval->val = 0;
1320  mval->mask = 0;
1321  }
1322  }
1323  }
1324  break;
1325 
1326  default:
1327  rules_error("Wrong type of identifier");
1328  delete mval;
1329  return false;
1330  }
1331

1333

1335  }
