155 double PktSrc::CheckPseudoTime()
156  {
157  if ( ! IsOpen() )
158  return 0;
159 
160  if ( ! ExtractNextPacketInternal() )
161  return 0;
162 
163  if ( remote_trace_sync_interval )
164  {
165  if ( next_sync_point == 0 || current_packet.time >= next_sync_point )
166  {
167  int n = remote_serializer->SendSyncPoint();
168  next_sync_point = first_timestamp +
169  n * remote_trace_sync_interval;
170  remote_serializer->Log(RemoteSerializer::LogInfo,
171  fmt("stopping at packet %.6f, next sync-point at %.6f",
172  current_packet.time, next_sync_point));
173 
174  return 0;
175  }
176  }
177 
178  double pseudo_time = current_packet.time - first_timestamp;
179  double ct = (current_time(true) - first_wallclock) * pseudo_realtime;
180 
181  return pseudo_time <= ct ? bro_start_time + pseudo_time : 0;
182  }

275 bool PktSrc::ExtractNextPacketInternal()
276  {
277  if ( have_packet )
278  return true;
279 
280  have_packet = false;
281 
282  // Don't return any packets if processing is suspended (except for the
283  // very first packet which we need to set up times).
284  if ( net_is_processing_suspended() && first_timestamp )
285  {
286  SetIdle(true);
287  return 0;
288  }
289 
290  if ( pseudo_realtime )
291  current_wallclock = current_time(true);
292 
293  if ( ExtractNextPacket(&current_packet) )
294  {
295  if ( current_packet.time < 0 )
296  {
297  Weird("negative_packet_timestamp", &current_packet);
298  return 0;
299  }
300 
301  if ( ! first_timestamp )
302  first_timestamp = current_packet.time;
303 
304  SetIdle(false);
305  have_packet = true;
306  return 1;
307  }
308 
309  if ( pseudo_realtime && ! IsOpen() )
310  {
311  if ( using_communication )
312  {
313  // Source has gone dry, we're done.
314  if ( remote_trace_sync_interval )
315  remote_serializer->SendFinalSyncPoint();
316  else
317  remote_serializer->Terminate();
318  }
319 
320  if ( broker_mgr->Active() )
321  iosource_mgr->Terminate();
322  }
323 
324  SetIdle(true);
325  return 0;
326  }
