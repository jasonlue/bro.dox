389 bool RPC_Reasm_Buffer::ConsumeChunk(const u_char*& data, int& len)
390  {
391  // How many bytes do we want to process with this call? Either the
392  // all of the bytes available or the number of bytes that we are
393  // still missing.
394  int64_t to_process = min(int64_t(len), (expected-processed));
395 
396  if ( fill < maxsize )
397  {
398  // We haven't yet filled the buffer. How many bytes to copy
399  // into the buff. Either all of the bytes we want to process
400  // or the number of bytes until we reach maxsize.
401  int64_t to_copy = min( to_process, (maxsize-fill) );
402  if ( to_copy )
403  memcpy(buf+fill, data, to_copy);
404 
405  fill += to_copy;
406  }
407 
408  processed += to_process;
409  len -= to_process;
410  data += to_process;
411  return (expected == processed);
412  }

441 bool Contents_RPC::CheckResync(int& len, const u_char*& data, bool orig)
442  {
443  uint32 frame_len;
444  bool last_frag;
445  uint32 xid;
446  uint32 frame_type;
447 
448  bool discard_this_chunk = false;
449 
450  if ( resync_state == RESYNC_INIT )
451  {
452  // First time CheckResync is called. If the TCP endpoint
453  // is fully established we are in sync (since it's the first chunk
454  // of data after the SYN if its not established we need to
455  // resync.
456  tcp::TCP_Analyzer* tcp =
457  static_cast<tcp::TCP_ApplicationAnalyzer*>(Parent())->TCP();
458  assert(tcp);
459 
460  if ( (IsOrig() ? tcp->OrigState() : tcp->RespState()) !=
461  tcp::TCP_ENDPOINT_ESTABLISHED )
462  {
463  NeedResync();
464  }
465  else
466  resync_state = INSYNC;
467  }
468 
469  if ( resync_state == INSYNC )
470  return true;
471 
472  // This is an attempt to re-synchronize the stream with RPC frames
473  // after a content gap. Returns true if we are in sync. Returns
474  // false otherwise (we are in resync mode)
475  //
476  // We try to look for the beginning of a RPC frame, assuming RPC
477  // frames begin at packet boundaries (though they may span over
478  // multiple packets) (note that the data* of DeliverStream() usually
479  // starts at a packet boundrary).
480  //
481  // If we see a frame start that makes sense (direction and frame
482  // lenght seem ok), we try to read (skip over) the next RPC message.
483  // If this is successfull and we the place we are seems like a valid
484  // start of a RPC msg (direction and frame length seem ok). We assume
485  // that we have successfully resync'ed.
486 
487  // Assuming RPC frames align with packet boundaries ...
488 
489  while (len > 0)
490  {
491  if ( resync_toskip )
492  {
493  if ( DEBUG_rpc_resync )
494  DEBUG_MSG("RPC resync: skipping %d bytes.\n", len);
495 
496  // We have some bytes to skip over.
497  if ( resync_toskip < len )
498  {
499  len -= resync_toskip;
500  data += resync_toskip;
501  resync_toskip = 0;
502  }
503  else
504  {
505  resync_toskip -= len;
506  data += len;
507  len = 0;
508  return false;
509  }
510  }
511 
512  if ( resync_toskip != 0 )
513  {
514  // Should never happen.
515  reporter->AnalyzerError(this,
516  "RPC resync: skipping over data failed");
517  return false;
518  }
519 
520  // Now lets see whether data points to the beginning of a RPC
521  // frame. If the resync processs is successful, we should be
522  // at the beginning of a frame.
523 
524 
525  if ( len < 12 )
526  {
527  // Ignore small chunks.
528  if ( len != 1 && DEBUG_rpc_resync )
529  {
530  // One-byte fragments are likely caused by
531  // TCP keep-alive retransmissions.
532  DEBUG_MSG("%.6f RPC resync: "
533  "discard small pieces: %d\n",
534  network_time, len);
535  Conn()->Weird(
536  fmt("RPC resync: discard %d bytes\n",
537  len));
538  }
539 
540  NeedResync();
541  return false;
542  }
543 
544 
545  const u_char *xdata = data;
546  int xlen = len;
547  frame_len = extract_XDR_uint32(xdata, xlen);
548  last_frag = (frame_len & 0x80000000) != 0;
549  frame_len &= 0x7fffffff;
550  xid = extract_XDR_uint32(xdata, xlen);
551  frame_type = extract_XDR_uint32(xdata, xlen);
552 
553  // Check if the direction makes sense and the length of the
554  // frame to expect.
555  if ( (IsOrig() && frame_type != 0) ||
556  (! IsOrig() && frame_type != 1) ||
557  frame_len < 16 )
558 
559  discard_this_chunk = true;
560 
561  // Make sure the frame isn't too long.
562  // TODO: Could possible even reduce this number even further.
563  if ( frame_len > MAX_RPC_LEN )
564  discard_this_chunk = true;
565 
566  if ( discard_this_chunk )
567  {
568  // Skip this chunk
569  if ( DEBUG_rpc_resync )
570  DEBUG_MSG("RPC resync: Need to resync. dicarding %d bytes.\n", len);
571 
572  NeedResync(); // let's try the resync again from the beginning
573  return false;
574  }
575 
576  // Looks like we are at the start of a frame and have successfully
577  // extracted the frame length (marker).
578 
579  switch (resync_state) {
580  case NEED_RESYNC:
581  case RESYNC_WAIT_FOR_MSG_START:
582  // Initial phase of resyncing. Skip frames until we get a frame
583  // with the last_fragment bit set.
584  resync_toskip = frame_len + 4;
585 
586  if ( last_frag )
587  resync_state = RESYNC_WAIT_FOR_FULL_MSG;
588  else
589  resync_state = RESYNC_WAIT_FOR_MSG_START;
590  break;
591 
592  case RESYNC_WAIT_FOR_FULL_MSG:
593  // If the resync was successful so far, we should now be the start
594  // of a new RPC message. Try to skip over it.
595  resync_toskip = frame_len + 4;
596 
597  if ( last_frag )
598  resync_state = RESYNC_HAD_FULL_MSG;
599  break;
600 
601  case RESYNC_HAD_FULL_MSG:
602  // We have now successfully skipped over a full RPC message.
603  // If we got that far, we are in sync.
604  resync_state = INSYNC;
605 
606  if ( DEBUG_rpc_resync )
607  DEBUG_MSG("RPC resync: success.\n");
608  return true;
609 
610  default:
611  // Should never happen.
612  NeedResync();
613  return false;
614  } // end switch
615  } // end while (len>0)
616 
617  return false;
618  }

623 void Contents_RPC::DeliverStream(int len, const u_char* data, bool orig)
624  {
625  tcp::TCP_SupportAnalyzer::DeliverStream(len, data, orig);
626  uint32 marker;
627  bool last_frag;
628 
629  if ( ! CheckResync(len, data, orig) )
630  return; // Not in sync yet. Still resyncing.
631 
632  // Should be in sync now.
633 
634  while (len > 0)
635  {
636  last_time = network_time;
637 
638  switch (state) {
639  case WAIT_FOR_MESSAGE:
640  // A new RPC message is starting. Initialize state.
641 
642  // We expect and want 4 bytes of the frame markers.
643  marker_buf.Init(4,4);
644 
645  // We want at most 64KB of message data and we don't
646  // know yet how much we expect, so we set expected to
647  // 0.
648  msg_buf.Init(MAX_RPC_LEN, 0);
649  last_frag = 0;
650  state = WAIT_FOR_MARKER;
651  start_time = network_time;
652  // no break. fall through
653 
654  case WAIT_FOR_MARKER:
655  {
656  bool got_marker = marker_buf.ConsumeChunk(data,len);
657 
658  if ( got_marker )
659  {
660  const u_char *dummy_p = marker_buf.GetBuf();
661  int dummy_len = (int) marker_buf.GetFill();
662 
663  // have full marker
664  marker = extract_XDR_uint32(dummy_p, dummy_len);
665  marker_buf.Init(4,4);
666 
667  if ( ! dummy_p )
668  {
669  reporter->AnalyzerError(this,
670  "inconsistent RPC record marker extraction");
671  return;
672  }
673 
674  last_frag = (marker & 0x80000000) != 0;
675  marker &= 0x7fffffff;
676  //printf("%.6f %d marker= %u <> last_frag= %d <> expected=%llu <> processed= %llu <> len = %d\n",
677  // network_time, IsOrig(), marker, last_frag, msg_buf.GetExpected(), msg_buf.GetProcessed(), len);
678 
679  if ( ! msg_buf.AddToExpected(marker) )
680  Conn()->Weird(fmt("RPC_message_too_long (%" PRId64 ")" , msg_buf.GetExpected()));
681 
682  if ( last_frag )
683  state = WAIT_FOR_LAST_DATA;
684  else
685  state = WAIT_FOR_DATA;
686  }
687  }
688  // Else remain in state. Haven't got the full 4 bytes
689  // for the marker yet.
690  break;
691 
692  case WAIT_FOR_DATA:
693  case WAIT_FOR_LAST_DATA:
694  {
695  bool got_all_data = msg_buf.ConsumeChunk(data, len);
696 
697  if ( got_all_data )
698  {
699  // Got all the data we expected. Now let's
700  // see whether there is another fragment
701  // coming or whether we just finished the
702  // last fragment.
703  if ( state == WAIT_FOR_LAST_DATA )
704  {
705  const u_char *dummy_p = msg_buf.GetBuf();
706  int dummy_len = (int) msg_buf.GetFill();
707 
708  if ( ! interp->DeliverRPC(dummy_p, dummy_len, (int)msg_buf.GetExpected(), IsOrig(), start_time, last_time) )
709  Conn()->Weird("partial_RPC");
710 
711  state = WAIT_FOR_MESSAGE;
712  }
713  else
714  state = WAIT_FOR_MARKER;
715  }
716  // Else remain in state. Haven't read all the data
717  // yet.
718  }
719  break;
720  } // end switch
721  } // end while
722  }
