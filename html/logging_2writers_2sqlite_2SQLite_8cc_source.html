195  sqlite3_free(errorMsg);

197  }
198

200  string insert = "VALUES (";

202

204  {
205  bool ac = true;

208  ac = false;

222  }

229  names += ") ";
230

250 
251  case TYPE_COUNT:
252  case TYPE_COUNTER:

257

268  }
269 
270  case TYPE_TIME:
271  case TYPE_INTERVAL:
272  case TYPE_DOUBLE:

275  case TYPE_ENUM:
276  case TYPE_STRING:
277  case TYPE_FILE:
278  case TYPE_FUNC:
279  {
280  if ( ! val->val.string_val.length || val->val.string_val.length == 0 )
281  return sqlite3_bind_null(st, pos);
