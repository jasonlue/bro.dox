428 vector<string> dir_contents_recursive(string dir)
429  {
430  vector<string> rval;
431  struct stat st;
432 
433  if ( stat(dir.c_str(), &st) < 0 && errno == ENOENT )
434  return rval;
435 
436  while ( dir[dir.size() - 1] == '/' )
437  dir.erase(dir.size() - 1, 1);
438 
439  char* dir_copy = copy_string(dir.c_str());
440  char** scan_path = new char*[2];
441  scan_path[0] = dir_copy;
442  scan_path[1] = 0;
443 
444  FTS* fts = fts_open(scan_path, FTS_NOCHDIR, 0);
445 
446  if ( ! fts )
447  {
448  reporter->Error("fts_open failure: %s", strerror(errno));
449  delete [] scan_path;
450  delete [] dir_copy;
451  return rval;
452  }
453 
454  FTSENT* n;
455 
456  while ( (n = fts_read(fts)) )
457  {
458  if ( n->fts_info & FTS_F )
459  rval.push_back(n->fts_path);
460  }
461 
462  if ( errno )
463  reporter->Error("fts_read failure: %s", strerror(errno));
464 
465  if ( fts_close(fts) < 0 )
466  reporter->Error("fts_close failure: %s", strerror(errno));
467 
468  delete [] scan_path;
469  delete [] dir_copy;
470  return rval;
471  }
