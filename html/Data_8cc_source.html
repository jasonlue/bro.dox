848  case TYPE_FILE:

852  case TYPE_TABLE:

856  auto table_val = v->AsTableVal();

864  struct iter_guard {
865  iter_guard(HashKey* arg_k, ListVal* arg_lv)
866  : k(arg_k), lv(arg_lv)
867  {}
868 
869  ~iter_guard()
870  {
871  delete k;
872  Unref(lv);
873  }
874 
875  HashKey* k;
876  ListVal* lv;
877  };
878 
879  HashKey* k;
880  TableEntryVal* entry;
881  auto c = table->InitForIteration();
882 
883  while ( (entry = table->NextEntry(k, c)) )
884  {
885  auto vl = table_val->RecoverIndex(k);
886  iter_guard ig(k, vl);
887 
888  broker::vector composite_key;
889  composite_key.reserve(vl->Length());
890 
891  for ( auto k = 0; k < vl->Length(); ++k )
892  {
893  auto key_part = val_to_data((*vl->Vals())[k]);
894 
895  if ( ! key_part )
896  return broker::ec::invalid_data;
897 
898  composite_key.emplace_back(move(*key_part));
899  }
900 
901  broker::data key;
902 
903  if ( composite_key.size() == 1 )
904  key = move(composite_key[0]);
905  else
906  key = move(composite_key);
907 
908  if ( is_set )
909  caf::get<broker::set>(rval).emplace(move(key));
910  else
911  {
912  auto val = val_to_data(entry->Value());
913 
914  if ( ! val )
915  return broker::ec::invalid_data;
916 
917  caf::get<broker::table>(rval).emplace(move(key), move(*val));
918  }
919  }
920 
921  return {std::move(rval)};
922  }
923  case TYPE_VECTOR:
924  {
925  auto vec = v->AsVectorVal();
926  broker::vector rval;
927  rval.reserve(vec->Size());
928 
929  for ( auto i = 0u; i < vec->Size(); ++i )
930  {
931  auto item_val = vec->Lookup(i);
932 
933  if ( ! item_val )
934  continue;
935 
936  auto item = val_to_data(item_val);
937 
938  if ( ! item )
939  return broker::ec::invalid_data;
940 
941  rval.emplace_back(move(*item));
942  }
943 
944  return {std::move(rval)};
945  }
946  case TYPE_RECORD:
947  {
948  auto rec = v->AsRecordVal();
949  broker::vector rval;
950  size_t num_fields = v->Type()->AsRecordType()->NumFields();
951  rval.reserve(num_fields);
952 
953  for ( auto i = 0u; i < num_fields; ++i )
954  {
955  auto item_val = rec->LookupWithDefault(i);
956 
957  if ( ! item_val )
958  {
959  rval.emplace_back(broker::nil);
960  continue;
961  }
962 
963  auto item = val_to_data(item_val);
964  Unref(item_val);
965 
966  if ( ! item )
967  return broker::ec::invalid_data;
968 
969  rval.emplace_back(move(*item));
970  }
971 
972  return {std::move(rval)};
973  }
974  case TYPE_PATTERN:
975  {
976  RE_Matcher* p = v->AsPattern();
977  broker::vector rval = {p->PatternText(), p->AnywherePatternText()};
978  return {std::move(rval)};
979  }
980  case TYPE_OPAQUE:
981  {
982  SerializationFormat* form = new BinarySerializationFormat();
983  form->StartWrite();
984  CloneSerializer ss(form);
985  SerialInfo sinfo(&ss);
986  sinfo.cache = false;
987  sinfo.include_locations = false;
988 
989  if ( ! v->Serialize(&sinfo) )
990  return broker::ec::invalid_data;
991 
992  char* data;
993  uint32 len = form->EndWrite(&data);
994  string rval(data, len);
995  free(data);
996  return {std::move(rval)};
997  }
998  default:
999  reporter->Error("unsupported Broker::Data type: %s",
1000  type_name(v->Type()->Tag()));
1001  break;
1002  }
1003 
1004  return broker::ec::invalid_data;
1005  }
