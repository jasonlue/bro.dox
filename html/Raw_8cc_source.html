117 bool Raw::Execute()
118  {
119  // AFAICT, pipe/fork/exec should be thread-safe, but actually having
120  // multiple threads set up pipes and fork concurrently sometimes
121  // results in problems w/ a stdin pipe not ever getting an EOF even
122  // though both ends of it are closed. But if the same threads
123  // allocate pipes and fork individually or sequentially, that issue
124  // never crops up... ("never" meaning I haven't seen in it in
125  // hundreds of tests using 50+ threads where before I'd see the issue
126  // w/ just 2 threads ~33% of the time).
127  auto lock = AcquireForkMutex();
128 
129  if ( pipe(pipes) != 0 || pipe(pipes+2) || pipe(pipes+4) )
130  {
131  Error(Fmt("Could not open pipe: %d", errno));
132  return false;
133  }
134 
135  childpid = fork();
136  if ( childpid < 0 )
137  {
138  Error(Fmt("Could not create child process: %d", errno));
139  return false;
140  }
141 
142  else if ( childpid == 0 )
143  {
144  // we are the child.
145 
146  // Obtain a process group w/ child's PID.
147  if ( setpgid(0, 0) == -1 )
148  _exit(251);
149 
150  close(pipes[stdout_in]);
151  if ( dup2(pipes[stdout_out], stdout_fileno) == -1 )
152  _exit(252);
153 
154  close(pipes[stdout_out]);
155 
156  close(pipes[stdin_out]);
157  if ( stdin_towrite && dup2(pipes[stdin_in], stdin_fileno) == -1 )
158  _exit(253);
159 
160  close(pipes[stdin_in]);
161 
162  close(pipes[stderr_in]);
163  if ( use_stderr && dup2(pipes[stderr_out], stderr_fileno) == -1 )
164  _exit(254);
165 
166  close(pipes[stderr_out]);
167 
168  // Signal mask is inherited over fork-exec, so reset any ignored
169  // signals to default behavior and unblock any blocked signals.
170  setsignal(SIGPIPE, SIG_DFL); // May be ignored when debugging scripts.
171  sigset_t mask;
172  sigfillset(&mask);
173  // Assuming the fork-one model of pthreads' fork(), using sigprocmask()
174  // makes sense over pthread_sigmask() here.
175  sigprocmask(SIG_UNBLOCK, &mask, 0);
176 
177  execl("/bin/sh", "sh", "-c", fname.c_str(), (char*) NULL);
178  fprintf(stderr, "Exec failed :(......\n");
179  _exit(255);
180  }
181  else
182  {
183  // we are the parent
184 
185  // Parent also sets child process group immediately to avoid a race.
186  if ( setpgid(childpid, childpid) == -1 )
187  {
188  if ( errno == EACCES )
189  // Child already did exec. That's fine since then it must have
190  // already done the setpgid() itself.
191  ;
192 
193  else if ( errno == ESRCH && kill(childpid, 0) == 0 )
194  // Sometimes (e.g. FreeBSD) this error is reported even though
195  // child exists, so do extra sanity check of whether it exists.
196  ;
197 
198  else
199  {
200  char buf[256];
201  bro_strerror_r(errno, buf, sizeof(buf));
202  Warning(Fmt("Could not set child process group: %s", buf));
203  }
204  }
205 
206  lock.unlock();
207 
208  ClosePipeEnd(stdout_out);
209 
210  if ( Info().mode == MODE_STREAM )
211  {
212  if ( ! SetFDFlags(pipes[stdout_in], F_SETFL, O_NONBLOCK) )
213  return false;
214  }
215 
216  ClosePipeEnd(stdin_in);
217 
218  if ( stdin_towrite )
219  {
220  // Ya, just always set this to nonblocking. we do not
221  // want to block on a program receiving data. Note
222  // that there is a small gotcha with it. More data is
223  // queued when more data is read from the program
224  // output. Hence, when having a program in
225  // mode_manual where the first write cannot write
226  // everything, the rest will be stuck in a queue that
227  // is never emptied.
228  if ( ! SetFDFlags(pipes[stdin_out], F_SETFL, O_NONBLOCK) )
229  return false;
230  }
231  else
232  ClosePipeEnd(stdin_out);
233 
234  ClosePipeEnd(stderr_out);
235 
236  if ( use_stderr )
237  {
238  if ( ! SetFDFlags(pipes[stderr_in], F_SETFL, O_NONBLOCK) )
239  return false;
240  }
241  else
242  ClosePipeEnd(stderr_in);
243 
244  file = std::unique_ptr<FILE, int(*)(FILE*)>(fdopen(pipes[stdout_in], "r"), fclose);
245 
246  if ( ! file )
247  {
248  Error("Could not convert stdout_in fileno to file");
249  return false;
250  }
251 
252  pipes[stdout_in] = -1; // will be closed by fclose
253 
254  if ( use_stderr )
255  {
256  stderrfile = std::unique_ptr<FILE, int(*)(FILE*)>(fdopen(pipes[stderr_in], "r"), fclose);
257 
258  if ( ! stderrfile )
259  {
260  Error("Could not convert stderr_in fileno to file");
261  return false;
262  }
263 
264  pipes[stderr_in] = -1; // will be closed by fclose
265  }
266 
267  return true;
268  }

436 int64_t Raw::GetLine(FILE* arg_file)
437  {
438  errno = 0;
439  int pos = 0; // strstr_n only works on ints - so no use to use something different here
440  int offset = 0;
441 
442  if ( ! buf )
443  buf = std::unique_ptr<char[]>(new char[block_size]);
444 
445  int repeats = 1;
446 
447  for ( ;; )
448  {
449  size_t readbytes = fread(buf.get()+bufpos+offset, 1, block_size-bufpos, arg_file);
450  pos += bufpos + readbytes;
451  //printf("Pos: %d\n", pos);
452  bufpos = offset = 0; // read full block size in next read...
453 
454  if ( pos == 0 && errno != 0 )
455  break;
456 
457  // researching everything each time is a bit... cpu-intensive. But otherwhise we have
458  // to deal with situations where the separator is multi-character and split over multiple
459  // reads...
460  int found = strstr_n(pos, (unsigned char*) buf.get(), separator.size(), (unsigned char*) separator.c_str());
461 
462  if ( found == -1 )
463  {
464  // we did not find it and have to search again in the next try. resize buffer....
465  // but first check if we encountered the file end - because if we did this was it.
466  if ( feof(arg_file) != 0 )
467  {
468  if ( pos == 0 )
469  return -1; // signal EOF - and that we had no more data.
470  else
471  {
472  outbuf = std::move(buf); // buf is null after this
473  return pos;
474  }
475  }
476 
477  repeats++;
478  // bah, we cannot use realloc because we would have to change the delete in the manager to a free.
479  std::unique_ptr<char[]> newbuf = std::unique_ptr<char[]>(new char[block_size*repeats]);
480  memcpy(newbuf.get(), buf.get(), block_size*(repeats-1));
481  buf = std::move(newbuf);
482  offset = block_size*(repeats-1);
483  }
484  else
485  {
486  outbuf = std::move(buf);
487 
488  if ( found < pos )
489  {
490  // we have leftovers. copy them into the buffer for the next line
491  buf = std::unique_ptr<char[]>(new char[block_size]);
492  memcpy(buf.get(), outbuf.get() + found + sep_length, pos - found - sep_length);
493  bufpos = pos - found - sep_length;
494  }
495 
496  return found;
497  }
498 
499  }
500 
501  if ( errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR )
502  return -2;
503 
504  else
505  {
506  // an error code we did no expect. This probably is bad.
507  Error(Fmt("Reader encountered unexpected error code %d", errno));
508  return -3;
509  }
