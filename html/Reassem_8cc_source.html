56 void Reassembler::CheckOverlap(DataBlock *head, DataBlock *tail,
57  uint64 seq, uint64 len, const u_char* data)
58  {
59  if ( ! head || ! tail )
60  return;
61 
62  if ( seq == tail->upper )
63  // Special case check for common case of appending to the end.
64  return;
65 
66  uint64 upper = (seq + len);
67 
68  for ( DataBlock* b = head; b; b = b->next )
69  {
70  uint64 nseq = seq;
71  uint64 nupper = upper;
72  const u_char* ndata = data;
73 
74  if ( nupper <= b->seq )
75  continue;
76 
77  if ( nseq >= b->upper )
78  continue;
79 
80  if ( nseq < b->seq )
81  {
82  ndata += (b->seq - seq);
83  nseq = b->seq;
84  }
85 
86  if ( nupper > b->upper )
87  nupper = b->upper;
88 
89  uint64 overlap_offset = (nseq - b->seq);
90  uint64 overlap_len = (nupper - nseq);
91

132 uint64 Reassembler::TrimToSeq(uint64 seq)
133  {
134  uint64 num_missing = 0;
135 
136  // Do this accounting before looking for Undelivered data,
137  // since that will alter last_reassem_seq.
138 
139  if ( blocks )
140  {
141  if ( blocks->seq > last_reassem_seq )
142  // An initial hole.
143  num_missing += blocks->seq - last_reassem_seq;
144  }
145 
146  else if ( seq > last_reassem_seq )
147  { // Trimming data we never delivered.
148  if ( ! blocks )
149  // We won't have any accounting based on blocks
150  // for this hole.
151  num_missing += seq - last_reassem_seq;
152  }
153 
154  if ( seq > last_reassem_seq )
155  {
156  // We're trimming data we never delivered.
157  Undelivered(seq);
158  }
159 
160  while ( blocks && blocks->upper <= seq )
161  {
162  DataBlock* b = blocks->next;
163 
164  if ( b && b->seq <= seq )
165  {
166  if ( blocks->upper != b->seq )
167  num_missing += b->seq - blocks->upper;
168  }
169  else
170  {
171  // No more blocks - did this one make it to seq?
172  // Second half of test is for acks of FINs, which
173  // don't get entered into the sequence space.
174  if ( blocks->upper != seq && blocks->upper != seq - 1 )
175  num_missing += seq - blocks->upper;
176  }
177 
178  if ( max_old_blocks )
179  {
180  // Move block over to old_blocks queue.
181  blocks->next = 0;
182 
183  if ( last_old_block )
184  {
185  blocks->prev = last_old_block;
186  last_old_block->next = blocks;
187  }
188  else
189  {
190  blocks->prev = 0;
191  old_blocks = blocks;
192  }
193 
194  last_old_block = blocks;
195  total_old_blocks++;
196 
197  while ( old_blocks && total_old_blocks > max_old_blocks )
198  {
199  DataBlock* next = old_blocks->next;
200  delete old_blocks;
201  old_blocks = next;
202  total_old_blocks--;
203  }
204  }
205 
206  else
207  delete blocks;
208 
209  blocks = b;
210  }
211 
212  if ( blocks )
213  {
214  blocks->prev = 0;
215 
216  // If we skipped over some undeliverable data, then
217  // it's possible that this block is now deliverable.
218  // Give it a try.
219  if ( blocks->seq == last_reassem_seq )
220  BlockInserted(blocks);
221  }
222  else
223  last_block = 0;
224 
225  if ( seq > trim_seq )
226  // seq is further ahead in the sequence space.
227  trim_seq = seq;
228 
229  return num_missing;
230  }
