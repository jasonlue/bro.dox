138  draining = true;

140  // Past Bro versions drained as long as there events, including when
141  // a handler queued new events during its execution. This could lead
142  // to endless loops in case a handler kept triggering its own event.
143  // We now limit this to just a couple of rounds. We do more than
144  // just one round to make it less likley to break existing scripts

146
