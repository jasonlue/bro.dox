178 void Contents_NCP_Analyzer::DeliverStream(int len, const u_char* data, bool orig)
179  {
180  tcp::TCP_SupportAnalyzer::DeliverStream(len, data, orig);
181 
182  auto tcp = static_cast<NCP_Analyzer*>(Parent())->TCP();
183 
184  if ( ! resync_set )
185  {
186  resync_set = true;
187  resync = (IsOrig() ? tcp->OrigState() : tcp->RespState()) !=
188  tcp::TCP_ENDPOINT_ESTABLISHED;
189  }
190 
191  if ( tcp && tcp->HadGap(orig) )
192  return;
193 
194  if ( buffer.empty() && resync )
195  {
196  // Assume NCP frames align with packet boundary.
197  if ( (IsOrig() && len < 22) || (! IsOrig() && len < 16) )
198  { // ignore small fragmeents
199  return;
200  }
201 
202  int frame_type_index = IsOrig() ? 16 : 8;
203  int frame_type = ((int) data[frame_type_index]) << 8 |
204  data[frame_type_index + 1];
205 
206  if ( frame_type != 0x1111 && frame_type != 0x2222 &&
207  frame_type != 0x3333 && frame_type != 0x5555 &&
208  frame_type != 0x7777 && frame_type != 0x9999 )
209  // Skip this packet
210  return;
211 
212  resync = false;
213  }
214 
215  for ( ; ; )
216  {
217  auto result = buffer.Deliver(len, data);
218 
219  if ( result < 0 )
220  break;
221 
222  if ( result == 0 )
223  {
224  session->Deliver(IsOrig(), buffer.Len(), buffer.Data());
225  buffer.Reset();
226  }
227  else
228  {
229  // The rest of the data available in this delivery will
230  // be discarded and will need to resync to a new frame header.
231  Weird("ncp_large_frame");
232  buffer.Reset();
233  resync = true;
234  break;
235  }
236  }
237  }
