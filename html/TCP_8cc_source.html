579 static void update_window(TCP_Endpoint* endpoint, unsigned int window,
580  uint32 base_seq, uint32 ack_seq, TCP_Flags flags)
581  {
582  // Note, applying scaling here would be incorrect for an initial SYN,
583  // whose window value is always unscaled. However, we don't
584  // check the window's value for recision in that case anyway, so
585  // no-harm-no-foul.
586  int scale = endpoint->window_scale;
587  window = window << scale;
588 
589  // Zero windows are boring if either (1) they come with a RST packet
590  // or after a RST packet, or (2) they come after the peer has sent
591  // a FIN (because there's no relevant window at that point anyway).
592  // (They're also boring if they come after the peer has sent a RST,
593  // but *nothing* should be sent in response to a RST, so we ignore
594  // that case.)
595  //
596  // However, they *are* potentially interesting if sent by an
597  // endpoint that's already sent a FIN, since that FIN meant "I'm
598  // not going to send any more", but doesn't mean "I won't receive
599  // any more".
600  if ( window == 0 && ! flags.RST() &&
601  endpoint->peer->state != TCP_ENDPOINT_CLOSED &&
602  endpoint->state != TCP_ENDPOINT_RESET )
603  endpoint->ZeroWindow();
604 
605  // Don't analyze window values off of SYNs, they're sometimes
606  // immediately rescinded. Also don't do so for FINs or RSTs,
607  // or if the connection has already been partially closed, since
608  // such recisions occur frequently in practice, probably as the
609  // receiver loses buffer memory due to its process going away.
610 
611  if ( ! flags.SYN() && ! flags.FIN() && ! flags.RST() &&
612  endpoint->state != TCP_ENDPOINT_CLOSED &&
613  endpoint->state != TCP_ENDPOINT_RESET )
614  {
615  // ### Decide whether to accept new window based on Active
616  // Mapping policy.
617  if ( seq_delta(base_seq, endpoint->window_seq) >= 0 &&
618  seq_delta(ack_seq, endpoint->window_ack_seq) >= 0 )
619  {
620  uint32 new_edge = ack_seq + window;
621  uint32 old_edge = endpoint->window_ack_seq + endpoint->window;
622  int32 advance = seq_delta(new_edge, old_edge);
623 
624  if ( advance < 0 )
625  {
626  // An apparent window recision. Allow a
627  // bit of slop for window scaling. This is
628  // because sometimes there will be an
629  // apparent recision due to the granularity
630  // of the scaling.
631  if ( (-advance) >= (1 << scale) )
632  endpoint->Conn()->Weird("window_recision");
633  }
634 
635  endpoint->window = window;
636  endpoint->window_ack_seq = ack_seq;
637  endpoint->window_seq = base_seq;
638  }
639  }

738  endpoint->SetState(TCP_ENDPOINT_RESET);
739 
740  int is_reject = 0;
741 
742  if ( is_orig )
743  {
744  // If our peer is established then we saw
745  // a SYN-ack but not SYN - so a reverse
746  // scan, and we should treat this as a
747  // reject.
748  if ( peer->state == TCP_ENDPOINT_ESTABLISHED )
749  is_reject = 1;
750  }
751 
752  else if ( peer->state == TCP_ENDPOINT_SYN_SENT ||
753  peer->state == TCP_ENDPOINT_SYN_ACK_SENT )
754  // We're rejecting an initial SYN.
755  is_reject = 1;
756 
757  do_close = 1;
758  gen_event = ! is_reject;
759 
760  if ( is_reject )
761  Event(connection_rejected);
762 
763  else if ( peer->state == TCP_ENDPOINT_INACTIVE )
764  Weird("spontaneous_RST");
765  }
766 
767  if ( endpoint->state == TCP_ENDPOINT_INACTIVE )
768  { // No control flags to change the state.
769  if ( ! is_orig && len == 0 &&
770  orig->state == TCP_ENDPOINT_SYN_SENT )
771  // Some eccentric TCP's will ack an initial
772  // SYN prior to sending a SYN reply (hello,
773  // ftp.microsoft.com). For those, don't
774  // consider the ack as forming a partial
775  // connection.
776  ;
777 
778  else if ( flags.ACK() && peer->state == TCP_ENDPOINT_ESTABLISHED )
779  {
780  // No SYN packet from originator but SYN/ACK from
781  // responder, and now a pure ACK. Problably means we
782  // just missed that initial SYN. Let's not treat it
783  // as partial and instead establish the connection.
784  endpoint->SetState(TCP_ENDPOINT_ESTABLISHED);
785  is_partial = 0;
786  }
787 
788  else
789  {
790  endpoint->SetState(TCP_ENDPOINT_PARTIAL);
791  Conn()->EnableStatusUpdateTimer();
792 
793  if ( peer->state == TCP_ENDPOINT_PARTIAL )
794  // We've seen both sides of a partial
795  // connection, report it.
796  Event(partial_connection);
797  }
798  }

1076 static void init_endpoint(TCP_Endpoint* endpoint, TCP_Flags flags,
1077  uint32 first_seg_seq, uint32 last_seq, double t)
1078  {
1079  switch ( endpoint->state ) {
1080  case TCP_ENDPOINT_INACTIVE:
1081  if ( flags.SYN() )
1082  {
1083  endpoint->InitAckSeq(first_seg_seq);
1084  endpoint->InitStartSeq(first_seg_seq);
1085  }
1086  else
1087  {
1088  // This is a partial connection - set up the initial sequence
1089  // numbers as though we saw a SYN, to keep the relative byte
1090  // numbering consistent.
1091  endpoint->InitAckSeq(first_seg_seq - 1);
1092  endpoint->InitStartSeq(first_seg_seq - 1);
1093  }
1094 
1095  endpoint->InitLastSeq(last_seq);
1096  endpoint->start_time = t;
1097  break;
1098 
1099  case TCP_ENDPOINT_SYN_SENT:
1100  case TCP_ENDPOINT_SYN_ACK_SENT:
1101  if ( flags.SYN() && first_seg_seq != endpoint->StartSeq() )
1102  {
1103  endpoint->Conn()->Weird("SYN_seq_jump");
1104  endpoint->InitStartSeq(first_seg_seq);
1105  endpoint->InitAckSeq(first_seg_seq);
1106  endpoint->InitLastSeq(last_seq);
1107  }
1108  break;
1109 
1110  case TCP_ENDPOINT_ESTABLISHED:
1111  case TCP_ENDPOINT_PARTIAL:
1112  if ( flags.SYN() )
1113  {
1114  if ( endpoint->Size() > 0 )
1115  endpoint->Conn()->Weird("SYN_inside_connection");
1116 
1117  if ( first_seg_seq != endpoint->StartSeq() )
1118  endpoint->Conn()->Weird("SYN_seq_jump");
1119 
1120  // Make a guess that somehow the connection didn't get established,
1121  // and this SYN will be the one that actually sets it up.
1122  endpoint->InitStartSeq(first_seg_seq);
1123  endpoint->InitAckSeq(first_seg_seq);
1124  endpoint->InitLastSeq(last_seq);
1125  }
1126  break;
1127 
1128  case TCP_ENDPOINT_RESET:
1129  if ( flags.SYN() )
1130  {
1131  if ( endpoint->prev_state == TCP_ENDPOINT_INACTIVE )
1132  {
1133  // Seq. numbers were initialized by a RST packet from this
1134  // endpoint, but now that a SYN is seen from it, that could mean
1135  // the earlier RST was spoofed/injected, so re-initialize. This
1136  // mostly just helps prevent misrepresentations of payload sizes
1137  // that are based on bad initial sequence values.
1138  endpoint->InitStartSeq(first_seg_seq);
1139  endpoint->InitAckSeq(first_seg_seq);
1140  endpoint->InitLastSeq(last_seq);
1141  }
1142  }
1143  break;
1144 
1145  default:
1146  break;
1147  }
1148  }

1150 static void init_peer(TCP_Endpoint* peer, TCP_Endpoint* endpoint,
1151  TCP_Flags flags, uint32 ack_seq)
1152  {
1153  if ( ! flags.SYN() && ! flags.FIN() && ! flags.RST() )
1154  {
1155  if ( endpoint->state == TCP_ENDPOINT_SYN_SENT ||
1156  endpoint->state == TCP_ENDPOINT_SYN_ACK_SENT ||
1157  endpoint->state == TCP_ENDPOINT_ESTABLISHED )

1160  // hasn't roused the other end, yet we're
1161  // ack'ing their data.

1164  endpoint->Conn()->Weird("possible_split_routing");

1169  // SYN, so the relative numbering of subsequent data packets
1170  // stays consistent.

1189 // Returns the difference between last_seq and the last sequence
1190 // seen by the endpoint (may be negative).
1191 static int32 update_last_seq(TCP_Endpoint* endpoint, uint32 last_seq,
1192  TCP_Flags flags, int len)
1193  {
1194  int32 delta_last = seq_delta(last_seq, endpoint->LastSeq());
1195 
1196  if ( (flags.SYN() || flags.RST()) &&
1197  (delta_last > TOO_LARGE_SEQ_DELTA ||
1198  delta_last < -TOO_LARGE_SEQ_DELTA) )
1199  // ### perhaps trust RST seq #'s if initial and not too
1200  // outlandish, but not if they're coming after the other
1201  // side has sent a FIN - trust the FIN ack instead
1202  ;
1203 
1204  else if ( flags.FIN() &&
1205  endpoint->LastSeq() == endpoint->StartSeq() + 1 )
1206  // Update last_seq based on the FIN even if delta_last < 0.
1207  // This is to accommodate > 2 GB connections for which
1208  // we've only seen the SYN and the FIN (hence the check
1209  // for last_seq == start_seq + 1).
1210  endpoint->UpdateLastSeq(last_seq);
1211 
1212  else if ( endpoint->state == TCP_ENDPOINT_RESET )
1213  // don't trust any subsequent sequence numbers
1214  ;
1215 
1216  else if ( delta_last > 0 )
1217  // ### check for large jumps here.
1218  // ## endpoint->last_seq = last_seq;
1219  endpoint->UpdateLastSeq(last_seq);
1220 
1221  else if ( delta_last < 0 && len > 0 )
1222  endpoint->DidRxmit();
1223 
1224  return delta_last;
1225  }

1676 void TCP_Analyzer::ConnectionClosed(TCP_Endpoint* endpoint, TCP_Endpoint* peer,
1677  int gen_event)
1678  {

1680  LOOP_OVER_CONST_CHILDREN(i)

1684  static_cast<TCP_ApplicationAnalyzer*>
1685  (*i)->ConnectionClosed(endpoint, peer, gen_event);

1690  // deliver.
1691  close_deferred = 1;

1693  deferred_gen_event = gen_event;
1694  return;
1695  }
1696 
1697  close_deferred = 0;
1698 
1699  if ( endpoint->did_close )
1700  return; // nothing new to report
1701 
1702  endpoint->did_close = 1;
1703

1705  endpoint->state == TCP_ENDPOINT_RESET ||
1706  peer->did_close ||
1707  peer->state == TCP_ENDPOINT_INACTIVE;
1708 
1709  if ( DEBUG_tcp_connection_close )
1710  {
1711  DEBUG_MSG("%.6f close_complete=%d tcp_close_delay=%f\n",
1712  network_time, close_complete, tcp_close_delay);
1713  }
1714 
1715  if ( close_complete )
1716  {
1717  if ( endpoint->prev_state != TCP_ENDPOINT_INACTIVE ||
1718  peer->state != TCP_ENDPOINT_INACTIVE )
1719  {
1720  if ( deferred_gen_event )
1721  {
1722  gen_event = 1;
1723  deferred_gen_event = 0; // clear flag
1724  }
1725 
1726  // We have something interesting to report.
1727  if ( gen_event )
1728  {
1729  if ( peer->state == TCP_ENDPOINT_INACTIVE )
1730  ConnectionFinished(1);
1731  else
1732  ConnectionFinished(0);
1733  }
1734  }

1739  // simply do:
1740  //
1741  // sessions->Remove(this);
1742  //
1743  // here, because that would cause the object to be
1744  // deleted out from under us.
1745  if ( tcp_close_delay != 0.0 )
1746  ADD_ANALYZER_TIMER(&TCP_Analyzer::ConnDeleteTimer,

1752  }

1762  }

1767  // too.

1863 bool TCP_Analyzer::IsReuse(double t, const u_char* pkt)
1864  {
1865  const struct tcphdr* tp = (const struct tcphdr*) pkt;
1866 
1867  if ( unsigned(tp->th_off) < sizeof(struct tcphdr) / 4 )
1868  // Bogus header, don't interpret further.
1869  return false;
1870 
1871  TCP_Endpoint* conn_orig = orig;
1872 
1873  // Reuse only occurs on initial SYN's, except for half connections
1874  // it can occur on SYN-acks.
1875  if ( ! (tp->th_flags & TH_SYN) )
1876  return false;
1877 
1878  if ( (tp->th_flags & TH_ACK) )
1879  {
1880  if ( orig->state != TCP_ENDPOINT_INACTIVE )
1881  // Not a half connection.
1882  return false;
1883 
1884  conn_orig = resp;
1885  }
1886 
1887  if ( ! IsClosed() )
1888  {
1889  uint32 base_seq = ntohl(tp->th_seq);
1890  if ( base_seq == conn_orig->StartSeq() )
1891  return false;
1892 
1893  if ( (tp->th_flags & TH_ACK) == 0 &&
1894  conn_orig->state == TCP_ENDPOINT_SYN_ACK_SENT &&
1895  resp->state == TCP_ENDPOINT_INACTIVE &&
1896  base_seq == resp->StartSeq() )
1897  {
1898  // This is an initial SYN with the right sequence
1899  // number, and the state is consistent with the
1900  // SYN & the SYN-ACK being flipped (e.g., due to
1901  // reading from two interfaces w/ interrupt
1902  // coalescence). Don't treat this as a reuse.
1903  // NextPacket() will flip set the connection
1904  // state correctly
1905  return false;
1906  }
1907 
1908  if ( conn_orig->state == TCP_ENDPOINT_SYN_SENT )
1909  Weird("SYN_seq_jump");
1910  else
1911  Weird("active_connection_reuse");
1912  }
1913 
1914  else if ( (orig->IsActive() || resp->IsActive()) &&
1915  orig->state != TCP_ENDPOINT_RESET &&
1916  resp->state != TCP_ENDPOINT_RESET )
1917  Weird("active_connection_reuse");
1918 
1919  else if ( t - Conn()->LastTime() < tcp_connection_linger &&
1920  orig->state != TCP_ENDPOINT_RESET &&
1921  resp->state != TCP_ENDPOINT_RESET )
1922  Weird("premature_connection_reuse");
1923 
1924  return true;
1925  }
