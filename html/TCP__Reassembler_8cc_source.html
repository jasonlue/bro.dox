363 void TCP_Reassembler::BlockInserted(DataBlock* start_block)
364  {
365  if ( start_block->seq > last_reassem_seq ||
366  start_block->upper <= last_reassem_seq )
367  return;
368 
369  // We've filled a leading hole. Deliver as much as possible.
370  // Note that the new block may include both some old stuff
371  // and some new stuff. AddAndCheck() will have split the
372  // new stuff off into its own block(s), but in the following
373  // loop we have to take care not to deliver already-delivered
374  // data.
375  for ( DataBlock* b = start_block;
376  b && b->seq <= last_reassem_seq; b = b->next )
377  {
378  if ( b->seq == last_reassem_seq )
379  { // New stuff.
380  uint64 len = b->Size();
381  uint64 seq = last_reassem_seq;
382  last_reassem_seq += len;
383 
384  if ( record_contents_file )
385  RecordBlock(b, record_contents_file);
386 
387  DeliverBlock(seq, len, b->block);
388  }
389  }
390 
391  TCP_Endpoint* e = endp;
392 
393  if ( ! e->peer->HasContents() )
394  // Our endpoint's peer doesn't do reassembly and so
395  // (presumably) isn't processing acks. So don't hold
396  // the now-delivered data.
397  TrimToSeq(last_reassem_seq);
398 
399  else if ( e->NoDataAcked() && tcp_max_initial_window &&
400  e->Size() > static_cast<uint64>(tcp_max_initial_window) )
401  // We've sent quite a bit of data, yet none of it has
402  // been acked. Presume that we're not seeing the peer's
403  // acks (perhaps due to filtering or split routing) and
404  // don't hang onto the data further, as we may wind up
405  // carrying it all the way until this connection ends.
406  TrimToSeq(last_reassem_seq);
