365 int DNS_Interpreter::ExtractLabel(const u_char*& data, int& len,
366  u_char*& name, int& name_len,
367  const u_char* msg_start)
368  {
369  if ( len <= 0 )
370  return 0;
371 
372  const u_char* orig_data = data;
373  int label_len = data[0];
374 
375  ++data;
376  --len;
377 
378  if ( len <= 0 )
379  return 0;
380 
381  if ( label_len == 0 )
382  // Found terminating label.
383  return 0;
384 
385  if ( (label_len & 0xc0) == 0xc0 )
386  {
387  unsigned short offset = (label_len & ~0xc0) << 8;
388 
389  offset |= *data;
390 
391  ++data;
392  --len;
393 
394  if ( offset >= orig_data - msg_start )
395  {
396  // (You'd think that actually the offset should be
397  // at least 6 bytes below our current position:
398  // 2 bytes for a non-trivial label, plus 4 bytes for
399  // its class and type, which presumably are between
400  // our current location and the instance of the label.
401  // But actually this turns out not to be the case -
402  // sometimes compression points to compression.)
403 
404  analyzer->Weird("DNS_label_forward_compress_offset");
405  return 0;
406  }
407 
408  // Recursively resolve name.
409  const u_char* recurse_data = msg_start + offset;
410  int recurse_max_len = orig_data - recurse_data;
411 
412  u_char* name_end = ExtractName(recurse_data, recurse_max_len,
413  name, name_len, msg_start);
414 
415  name_len -= name_end - name;
416  name = name_end;
417 
418  return 0;
419  }
420 
421  if ( label_len > len )
422  {
423  analyzer->Weird("DNS_label_len_gt_pkt");
424  data += len; // consume the rest of the packet
425  len = 0;
426  return 0;
427  }
428 
429  if ( label_len > 63 &&
430  // NetBIOS name service look ups can use longer labels.
431  ntohs(analyzer->Conn()->RespPort()) != 137 )
432  {
433  analyzer->Weird("DNS_label_too_long");
434  return 0;
435  }
436 
437  if ( label_len >= name_len )
438  {
439  analyzer->Weird("DNS_label_len_gt_name_len");
440  return 0;
441  }
442 
443  memcpy(name, data, label_len);
444  name[label_len] = '.';
445 
446  name += label_len + 1;
447  name_len -= label_len + 1;
448 
449  data += label_len;
450  len -= label_len;
451 
452  return 1;
453  }

1133 void Contents_DNS::DeliverStream(int len, const u_char* data, bool orig)
1134  {
1135  if ( state == DNS_LEN_HI )
1136  {
1137  msg_size = (*data) << 8;
1138  state = DNS_LEN_LO;
1139 
1140  ++data;
1141  --len;
1142 
1143  if ( len == 0 )
1144  return;
1145  }
1146 
1147  if ( state == DNS_LEN_LO )
1148  {
1149  msg_size += *data;
1150  state = DNS_MESSAGE_BUFFER;
1151 
1152  buf_n = 0;
1153 
1154  if ( msg_buf )
1155  {
1156  if ( buf_len < msg_size )
1157  {
1158  buf_len = msg_size;
1159  msg_buf = (u_char*) safe_realloc((void*) msg_buf, buf_len);
1160  }
1161  }
1162  else
1163  {
1164  buf_len = msg_size;
1165  msg_buf = (u_char*) safe_malloc(buf_len);
1166  }
1167 
1168  ++data;
1169  --len;
1170 
1171  if ( len == 0 )
1172  return;
1173  }
1174 
1175  if ( state != DNS_MESSAGE_BUFFER )
1176  Conn()->Internal("state inconsistency in Contents_DNS::DeliverStream");
1177 
1178  int n;
1179  for ( n = 0; buf_n < msg_size && n < len; ++n )
1180  msg_buf[buf_n++] = data[n];
1181 
1182  if ( buf_n < msg_size )
1183  // Haven't filled up the message buffer yet, no more to do.
1184  return;
1185 
1186  ForwardPacket(msg_size, msg_buf, orig, -1, 0, 0);
1187 
1188  buf_n = 0;
1189  state = DNS_LEN_HI;
1190 
1191  if ( n < len )
1192  // More data to munch on.
1193  DeliverStream(len - n, data + n, orig);
1194  }
