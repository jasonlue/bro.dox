90  {
91  // The first thing we see and want to know is the length of
92  // SSH_MSG_SERVICE_REQUEST, which has a fixed (decrypted) size
93  // of 24 bytes (17 for content pad-aligned to 8-byte
94  // boundaries)
95  if ( ! service_accept_size )
96  {
97  service_accept_size = len;
98  return;
99  }
100 
101  // If our user can authenticate via the "none" method, this
102  // packet will be a SSH_MSG_USERAUTH_SUCCESS, which has a
103  // fixed (decrypted) size of 8 bytes (1 for content
104  // pad-aligned to 8-byte boundaries). relative_len would be
105  // -16.
106  if ( ! userauth_failure_size && (len + 16 == service_accept_size) )
107  {
108  auth_decision_made = true;
109  if ( ssh_auth_attempted )
110  BifEvent::generate_ssh_auth_attempted(interp->bro_analyzer(), interp->bro_analyzer()->Conn(), true);
111  if ( ssh_auth_successful )
112  BifEvent::generate_ssh_auth_successful(interp->bro_analyzer(), interp->bro_analyzer()->Conn(), true);
113  return;
114  }
115 
116  // Normally, this packet would be a SSH_MSG_USERAUTH_FAILURE
117  // message, with a variable length, depending on the
118  // authentication methods the server supports. If it's too
119  // big, it might contain a pre-auth MOTD/banner, so we'll just
120  // skip it.
121  if ( ! userauth_failure_size )
122  {
123  if ( ! skipped_banner && (len - service_accept_size) > 256 )
124  {

126  return;
127  }
128  userauth_failure_size = len;
129  return;
130  }
131

133  // another packet of the same size.
134  if ( len == userauth_failure_size )
135  {
136  if ( ssh_auth_attempted )
137  BifEvent::generate_ssh_auth_attempted(interp->bro_analyzer(), interp->bro_analyzer()->Conn(), false);
