1711 void HTTP_Analyzer::ParseVersion(data_chunk_t ver, const IPAddr& host,
1712  bool user_agent)
1713  {
1714  int len = ver.length;
1715  const char* data = ver.data;
1716 
1717  if ( software_unparsed_version_found )
1718  Conn()->UnparsedVersionFoundEvent(host, data, len, this);
1719 
1720  // The RFC defines:
1721  //
1722  // product = token ["/" product-version]
1723  // product-version = token
1724  // Server = "Server" ":" 1*( product | comment )
1725 
1726  int offset;
1727  data_chunk_t product, product_version;
1728  int num_version = 0;
1729 
1730  while ( len > 0 )
1731  {
1732  // Skip white space.
1733  while ( len && mime::is_lws(*data) )
1734  {
1735  ++data;
1736  --len;
1737  }
1738 
1739  // See if a comment is coming next. For User-Agent,
1740  // we parse it, too.
1741  if ( user_agent && len && *data == '(' )
1742  {
1743  // Find end of comment.
1744  const char* data_start = data;
1745  const char* eoc =
1746  data + mime::MIME_skip_lws_comments(len, data);
1747 
1748  // Split into parts.
1749  // (This may get confused by nested comments,
1750  // but we ignore this for now.)
1751  const char* eot;
1752  ++data;
1753  while ( 1 )
1754  {
1755  // Eat spaces.
1756  while ( data < eoc && mime::is_lws(*data) )
1757  ++data;
1758 
1759  // Find end of token.
1760  for ( eot = data;
1761  eot < eoc && *eot != ';' && *eot != ')';
1762  ++eot )
1763  ;
1764 
1765  if ( eot == eoc )
1766  break;
1767 
1768  // Delete spaces at end of token.
1769  for ( ; eot > data && mime::is_lws(*(eot-1)); --eot )
1770  ;
1771 
1772  if ( data != eot && software_version_found )
1773  Conn()->VersionFoundEvent(host, data, eot - data, this);
1774  data = eot + 1;
1775  }
1776 
1777  len -= eoc - data_start;
1778  data = eoc;
1779  continue;
1780  }
1781 
1782  offset = mime::MIME_get_slash_token_pair(len, data,
1783  &product, &product_version);
1784  if ( offset < 0 )
1785  {
1786  // I guess version detection is best-effort,
1787  // so we do not complain in the final version
1788  if ( num_version == 0 )
1789  HTTP_Event("bad_HTTP_version",
1790  mime::new_string_val(len, data));
1791 
1792  // Try to simply skip next token.
1793  offset = mime::MIME_get_token(len, data, &product);
1794  if ( offset < 0 )
1795  break;
1796 
1797  len -= offset;
1798  data += offset;
1799  }
1800 
1801  else
1802  {
1803  len -= offset;
1804  data += offset;
1805

1807  product.length + 1 + product_version.length;
1808 
1809  char* version_str = new char[version_len+1];
1810  char* s = version_str;
1811 
1812  memcpy(s, product.data, product.length);
1813 
1814  s += product.length;
1815  *(s++) = '/';
1816 
1817  memcpy(s, product_version.data, product_version.length);
1818 
1819  s += product_version.length;
1820  *s = 0;
1821

1823  Conn()->VersionFoundEvent(host, version_str,
