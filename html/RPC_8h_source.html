133 /* A simple buffer for reassembling the fragments that RPC-over-TCP
134  * uses. Only needed by RPC_Contents.
135 
136  * However, RPC messages can be quite large. As a first step, we only
137  * extract and analyzer the first part of an RPC message and skip
138  * over the rest.
139  *
140  * We specify:
141  * maxsize: the number of bytes we want to copy into the buffer to analyze.
142  * expected: the total number of bytes in the RPC message. Can be
143  * quite large. We will be "skipping over" expected-maxsize bytes.
144  *
145  * We can extend "expected" (by calling AddToExpected()), but maxsize is
146  * fixed.
147  *
148  * TODO: grow buffer dynamically
149  */
150 class RPC_Reasm_Buffer {
151 public:
152  RPC_Reasm_Buffer() {
153  maxsize = expected = 0;
154  fill = processed = 0;
155  buf = 0;
156  };
157 
158  ~RPC_Reasm_Buffer() { if (buf) delete [] buf; }
159 
160  void Init(int64_t arg_maxsize, int64_t arg_expected);
161 
162  const u_char *GetBuf() { return buf; } // Pointer to the buffer
163  int64_t GetFill() { return fill; } // Number of bytes in buf
164  int64_t GetSkipped() { return processed-fill; } // How many bytes did we skipped?
165  int64_t GetExpected() { return expected; } // How many bytes are we expecting?
166  int64_t GetProcessed() { return processed; } // How many bytes are we expecting?
167 
168  // Expand expected by delta bytes. Returns false if the number of
169  // expected bytes exceeds maxsize (which means that we will truncate
170  // the message).
171  bool AddToExpected(int64_t delta)
172  { expected += delta; return ! (expected > maxsize); }
173 
174  // Consume a chunk of input data (pointed to by data, up len in
175  // size). data and len will be adjusted accordingly. Returns true if
176  // "expected" bytes have been processed, i.e., returns true when we
177  // don't expect any more data.
178  bool ConsumeChunk(const u_char*& data, int& len);
179 
180 protected:
181  int64_t fill; // how many bytes we currently have in the buffer
182  int64_t maxsize; // maximum buffer size we want to allocate
183  int64_t processed; // number of bytes we have processed so far
184  int64_t expected; // number of input bytes we expect
185  u_char *buf;
186 
187 };
