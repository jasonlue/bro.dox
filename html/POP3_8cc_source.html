93 static string trim_whitespace(const char* in)
94  {
95  int n = strlen(in);
96  char* out = new char[n + 1];
97  char* out_p = out;
98 
99  in = skip_whitespace(in);
100 
101  while ( *in )
102  {
103  // It might be better to use isspace() here, but the
104  // original code just compared with ' '.
105  if ( *in == ' ' )
106  {
107  // See if there is any following character.
108  ++in;
109  while ( *in && *in == ' ' )
110  ++in;
111 
112  if ( ! *in )
113  break;
114 
115  // There's a following character, so put in a
116  // single blank to represent the ones we
117  // compressed out.
118  *(out_p++) = ' ';
119  }
120 
121  // If we get this far, then we have a non-blank
122  // character to copy.
123  *(out_p++) = *(in++);
124  }
125 
126  *out_p = 0;
127 
128  string rval(out);
129  delete [] out;
130  return rval;

133 void POP3_Analyzer::ProcessRequest(int length, const char* line)
134  {
135  if ( length == 0 )
136  return;
137 
138  if ( waitingForAuthentication )
139  {
140  ++authLines;
141 
142  BroString encoded(line);
143  BroString* decoded = decode_base64(&encoded, 0, Conn());
144 
145  if ( ! decoded )
146  {
147  Weird("pop3_bad_base64_encoding");
148  return;
149  }
150 
151  switch ( state ) {
152  case AUTH_LOGIN:
153  // Format: Line 1 - User
154  // Line 2 - Password
155  if ( authLines == 1 )
156  user = decoded->CheckString();
157 
158  else if ( authLines == 2 )
159  password = decoded->CheckString();
160 
161  break;
162 
163  case AUTH_PLAIN:
164  {
165  // Format: "authorization identity<NUL>authentication
166  // identity<NUL>password"
167  char* str = (char*) decoded->Bytes();
168  int len = decoded->Len();
169  char* end = str + len;
170  char* s;
171  char* e;
172 
173  for ( s = str; s < end && *s; ++s )
174  ;
175  ++s;
176 
177  for ( e = s; e < end && *e; ++e )
178  ;
179 
180  if ( e >= end )
181  {
182  Weird("pop3_malformed_auth_plain");
183  delete decoded;
184  return;
185  }
186 
187  user = s;
188  s = e + 1;
189 
190  if ( s >= end )
191  {
192  Weird("pop3_malformed_auth_plain");
193  delete decoded;
194  return;
195  }
196 
197  char tmp[len]; // more than enough
198  int n = len - (s - str);
199  memcpy(tmp, s, n);
200  tmp[n] = '\0';
201  password = tmp;
202 
203  break;
204  }
205 
206  case AUTH_CRAM_MD5:
207  { // Format: "user<space>password-hash"
208  const char* s;
209  const char* str = (char*) decoded->CheckString();
210 
211  for ( s = str; *s && *s != '\t' && *s != ' '; ++s )
212  ;
213 
214  user = std::string(str, s);
215  password = "";
216 
217  break;
218  }
219 
220  case AUTH:
221  break;
222 
223  default:
224  reporter->AnalyzerError(this,
225  "unexpected POP3 authorization state");
226  delete decoded;
227  return;
228  }
229 
230  delete decoded;
231  }
232 
233  else
234  {
235  // Some clients pipeline their commands (i.e., keep sending
236  // without waiting for a server's responses). Therefore we
237  // keep a list of pending commands.
238  cmds.push_back(string(line));
239 
240  if ( cmds.size() == 1 )
241  // Not waiting for another server response,
242  // so we can process it immediately.
243  ProcessClientCmd();
244  }

597 void POP3_Analyzer::ProcessReply(int length, const char* line)
598  {
599  const char* end_of_line = line + length;
600  string str = trim_whitespace(line);
601 
602  if ( multiLine == true )
603  {
604  bool terminator =
605  line[0] == '.' &&
606  (length == 1 ||
607  (length > 1 &&
608  (line[1] == '\n' ||
609  (length > 2 && line[1] == '\r' && line[2] == '\n'))));
610 
611  if ( terminator )
612  {
613  requestForMultiLine = false;
614  multiLine = false;
615  if ( mail )
616  EndData();
617  FinishClientCmd();
618  }
619  else
620  {
621  if ( state == RETR || state == TOP )
622  {
623  int data_len = end_of_line - line;
624  ProcessData(data_len, line);
625  }
626 
627  // ### It can be quite costly doing this per-line
628  // as opposed to amortized over large packets that
629  // contain many lines.
630  POP3Event(pop3_data, false, str.c_str());
631  }
632  return;
633  }
634 
635  int cmd_code = -1;
636  const char* cmd = "";
637 
638  vector<string> tokens = TokenizeLine(str, ' ');
639  if ( tokens.size() > 0 )
640  cmd_code = ParseCmd(tokens[0]);
641 
642  if ( cmd_code == -1 )
643  {
644  if ( ! waitingForAuthentication )
645  {
646  ProtocolViolation(fmt("unknown server command (%s)",
647  (tokens.size() > 0 ?
648  tokens[0].c_str() :
649  "???")),
650  line, length);
651 
652  Weird("pop3_server_command_unknown");
653  if ( subState == POP3_WOK )
654  subState = POP3_OK;
655  }
656  return;
657  }
658 
659  cmd = commands[cmd_code].c_str();
660 
661  const char* message = tokens.size() > 1 ? tokens[1].c_str() : "";
662 
663  switch ( cmd_code ) {
664  case POP3_CMD_OK:
665  if ( subState == POP3_WOK )
666  subState = POP3_OK;
667 
668  if ( guessing )
669  {
670  masterState = POP3_TRANSACTION;
671  guessing = false;
672  state = lastState;
673  POP3Event(pop3_unexpected, false, cmd,
674  "no auth required -> state changed to 'transaction'");
675  }
676 
677  switch ( state ) {
678  case START:
679  masterState = POP3_AUTHORIZATION;
680  break;
681 
682  case USER:
683  state = USER;
684  masterState = POP3_AUTHORIZATION;
685  ProtocolConfirmation();
686  break;
687 
688  case PASS:
689  case APOP:
690  case NOOP:
691  case LAST:
692  case STAT:
693  case RSET:
694  case DELE:
695  case XSENDER:
696  if ( masterState == POP3_AUTHORIZATION )
697  AuthSuccessfull();
698  masterState = POP3_TRANSACTION;
699  break;
700 
701  case AUTH:
702  case AUTH_PLAIN:
703  case AUTH_CRAM_MD5:
704  case AUTH_LOGIN:
705  if ( requestForMultiLine == true )
706  multiLine = true;
707  if ( waitingForAuthentication )
708  masterState = POP3_TRANSACTION;
709  waitingForAuthentication = false;
710  AuthSuccessfull();
711  break;
712 
713  case TOP:
714  case RETR:
715  {
716  int data_len = end_of_line - line;
717  if ( ! mail )
718  // ProcessReply is only called if orig == false
719  BeginData(false);
720  ProcessData(data_len, line);
721  if ( requestForMultiLine == true )
722  multiLine = true;
723  break;
724  }
725 
726  case CAPA:
727  ProtocolConfirmation();
728  // Fall-through.
729 
730  case UIDL:
731  case LIST:
732  if (requestForMultiLine == true)
733  multiLine = true;
734  break;
735 
736  case STLS:
737  ProtocolConfirmation();
738  tls = true;
739  StartTLS();
740  return;
741 
742  case QUIT:
743  if ( masterState == POP3_AUTHORIZATION ||
744  masterState == POP3_START )
745  masterState = POP3_FINISHED;
746 
747  else if ( masterState == POP3_TRANSACTION )
748  masterState = POP3_UPDATE;
749 
750  break;
751  }
752 
753  POP3Event(pop3_reply, false, cmd, message);
754  // no else part, ignoring multiple OKs
755 
756  if ( ! multiLine )
757  FinishClientCmd();
758  break;
759 
760  case POP3_CMD_ERR:
761  if ( subState == POP3_WOK )
762  subState = POP3_OK;
763 
764  multiLine = false;
765  requestForMultiLine = false;
766  guessing = false;
767  waitingForAuthentication = false;
768 
769  switch ( state ) {
770  case START:
771  break;
772 
773  case USER:
774  case PASS:
775  case APOP:
776  case AUTH:
777  case AUTH_LOGIN:
778  case AUTH_PLAIN:
779  case AUTH_CRAM_MD5:
780  masterState = POP3_AUTHORIZATION;
781  state = START;
782  waitingForAuthentication = false;
783 
784  if ( user.size() )
785  POP3Event(pop3_login_failure, false,
786  user.c_str(), password.c_str());
787  break;
788 
789  case NOOP:
790  case LAST:
791  case STAT:
792  case RSET:
793  case DELE:
794  case LIST:
795  case RETR:
796  case UIDL:
797  case TOP:
798  case XSENDER:
799  masterState = POP3_TRANSACTION;
800  break;
801 
802  case CAPA:
803  break;
804 
805  case QUIT:
806  if ( masterState == POP3_AUTHORIZATION ||
807  masterState == POP3_TRANSACTION ||
808  masterState == POP3_START )
809  masterState = POP3_FINISHED;
810  break;
811  }
812 
813  POP3Event(pop3_reply, false, cmd, message);
814 
815  if ( ! multiLine )
816  FinishClientCmd();
817  break;
818 
819  default:
820  Weird("pop3_server_sending_client_commands");
821  break;
822  }
