429 cq_resize(register struct cq_handle *hp, register int grow)
430 {
431  register int n, nbuckets, oldnbuckets;
432  register size_t size;
433  register struct cq_bucket *bp, *bp2, *buckets, *oldbuckets;
434  struct cq_handle savedhandle;
435 
436  if (hp->noresize)
437  return (0);
438 #ifdef DEBUG
439  if (debug)
440  cq_debug(hp, 0);
441 #endif
442 
443  /* Save old bucket array */
444  oldnbuckets = hp->nbuckets;
445  oldbuckets = hp->buckets;
446 
447  /* If no old buckets, we're initializing */
448  if (oldbuckets == NULL)
449  nbuckets = 2;
450  else if (grow)
451  nbuckets = oldnbuckets * 2;
452  else
453  nbuckets = oldnbuckets / 2;
454 
455  /* XXX could check that nbuckets is a power of 2 */
456 
457  size = sizeof(*buckets) * nbuckets;
458  buckets = (struct cq_bucket *)malloc(size);
459  memory_allocation += size;
460 
461  if (buckets == NULL)
462  return (-1);
463  memset(buckets, 0, size);
464 
465  /* Save a copy of the handle in case we have dynamic memory problems */
466  savedhandle = *hp;
467 
468  /* Install new bucket array */
469  hp->nbuckets = nbuckets;
470  hp->buckets = buckets;
471 
472  /* Initialize other parameters */
473  hp->himark = hp->nbuckets * 1.5;
474  hp->lowmark = (hp->nbuckets / 2) - 2;
475  hp->bwidth = hp->ysize / (double)hp->nbuckets;
476 
477  /* Tell cq_enqueue() to update nextbucket and buckettop */
478  hp->lastpri = 0.0;
479 
480 #ifdef DEBUG
481  if (debug > 1)
482  fprintf(stderr,
483  "buckets 0x%lx, nbuckets %d, bwidth %f, buckettop %f\n",
484  (u_long)hp->buckets,
485  hp->nbuckets,
486  hp->bwidth, hp->buckettop);
487 #endif
488 
489  /* We're done if we were just initializing */
490  if (oldbuckets == NULL)
491  return (0);
492 
493  /* Insert entries from old bucket array into new one */
494  ++hp->noresize;
495  hp->qlen = 0;
496  for (bp = oldbuckets, n = oldnbuckets; n > 0; --n, ++bp)
497  if (BUCKETINUSE(bp))
498  for (bp2 = bp; bp2 != NULL; bp2 = bp2->next) {
499  if (cq_enqueue(hp, bp2->pri, bp2->cookie) < 0) {
500  /* Bummer! */
501  *hp = savedhandle;
502  /* hp->resize restored */
503  cq_destroybuckets(buckets, nbuckets);
504  free(buckets);
505  memory_allocation -= size;
506  return (-1);
507  }
508  }
509  --hp->noresize;
510 
511  cq_destroybuckets(oldbuckets, oldnbuckets);
512  free(oldbuckets);
513  memory_allocation -= sizeof(*buckets) * oldnbuckets;
514 #ifdef DEBUG
515  if (debug)
516  cq_debug(hp, 0);
517 #endif
518  return (0);
519 }
