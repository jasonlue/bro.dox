260 bool BitTorrentTracker_Analyzer::ParseRequest(char* line)
261  {
262  static bool initialized = false;
263  static regex_t r_get, r_get_end, r_hdr;
264 
265  if ( ! initialized )
266  {
267  regcomp(&r_get, "^GET[ \t]+", REG_EXTENDED | REG_ICASE);
268  regcomp(&r_get_end, "[ \t]+HTTP/[0123456789.]+$",
269  REG_EXTENDED | REG_ICASE);
270  regcomp(&r_hdr, "^[^: \t]+:[ ]*", REG_EXTENDED | REG_ICASE);
271  initialized = true;
272  }
273 
274  switch ( req_state ) {
275  case BTT_REQ_GET:
276  {
277  regmatch_t match[1];
278  if ( regexec(&r_get, line, 1, match, 0) )
279  {
280  ProtocolViolation("BitTorrentTracker: invalid HTTP GET");
281  stop_orig = true;
282  return false;
283  }
284 
285  regmatch_t match_end[1];
286  if ( ! regexec(&r_get_end, line, 1, match_end, 0) )
287  {
288  if ( match_end[0].rm_so <= match[0].rm_eo )
289  {
290  ProtocolViolation("BitTorrentTracker: invalid HTTP GET");
291  stop_orig = true;
292  return false;
293  }
294 
295  keep_alive = (line[match_end[0].rm_eo - 1] == '1');
296  line[match_end[0].rm_so] = 0;
297  }
298 
299  RequestGet(&line[match[0].rm_eo]);
300 
301  req_state = BTT_REQ_HEADER;
302  }
303  break;
304 
305  case BTT_REQ_HEADER:
306  {
307  if ( ! *line )
308  {
309  EmitRequest();
310  req_state = BTT_REQ_DONE;
311  break;
312  }
313 
314  regmatch_t match[1];
315  if ( regexec(&r_hdr, line, 1, match, 0) )
316  {
317  ProtocolViolation("BitTorrentTracker: invalid HTTP request header");
318  stop_orig = true;
319  return false;
320  }
321 
322  *strchr(line, ':') = 0; // this cannot fail - see regex_hdr
323  RequestHeader(line, &line[match[0].rm_eo]);
324  }
325  break;
326 
327  case BTT_REQ_DONE:
328  if ( *line )
329  DeliverWeird(fmt("Got post request data: %s\n", line),
330  true);
331  break;
332 
333  default:
334  // Make the compiler happy.
335  break;
336  }
337

363 bool BitTorrentTracker_Analyzer::ParseResponse(char* line)
364  {
365  static bool initialized = false;
366  static regex_t r_stat, r_hdr;
367 
368  if ( ! initialized )
369  {
370  regcomp(&r_stat, "^HTTP/[0123456789.]* ",
371  REG_EXTENDED | REG_ICASE);
372  regcomp(&r_hdr, "^[^: \t]+:[ ]*", REG_EXTENDED | REG_ICASE);
373  initialized = true;
374  }
375 
376  switch ( res_state ) {
377  case BTT_RES_STATUS:
378  {
379  if ( res_allow_blank_line && ! *line )
380  {
381  // There may be an empty line after the bencoded
382  // directory, if this is a keep-alive connection.
383  // Ignore it.
384  res_allow_blank_line = false;
385  break;
386  }
387 
388  regmatch_t match[1];
389  if ( regexec(&r_stat, line, 1, match, 0) )
390  {
391  ProtocolViolation("BitTorrentTracker: invalid HTTP status");
392  stop_resp = true;
393  return false;
394  }
395 
396  ResponseStatus(&line[match[0].rm_eo]);
397  res_state = BTT_RES_HEADER;
398  }
399  break;
400 
401  case BTT_RES_HEADER:
402  if ( ! *line )
403  {
404  if ( res_status != 200 )
405  {
406  val_list* vl = new val_list;
407  vl->append(BuildConnVal());
408  vl->append(new Val(res_status, TYPE_COUNT));
409  vl->append(res_val_headers);
410  ConnectionEvent(bt_tracker_response_not_ok, vl);
411  res_val_headers = 0;
412  res_buf_pos = res_buf + res_buf_len;
413  res_state = BTT_RES_DONE;
414  }
415  else
416  res_state = BTT_RES_BODY;
417 
418  break;
419  }
420 
421  {
422  regmatch_t match[1];
423  if ( regexec(&r_hdr, line, 1, match, 0) )
424  {
425  ProtocolViolation("BitTorrentTracker: invalid HTTP response header");
426  stop_resp = true;
427  return false;
428  }
429 
430  *strchr(line, ':') = 0; // this cannot fail - see regex_hdr
431  ResponseHeader(line, &line[match[0].rm_eo]);
432  }
433  break;
434 
435  default:
436  // Make the compiler happy.
437  break;
438  }

469 void BitTorrentTracker_Analyzer::ResponseBenc(int name_len, char* name,
470  enum btt_benc_types type, int value_len, char* value)
471  {
472  if ( name_len == 5 && ! strncmp(name, "peers", 5) )
473  {
474  for ( char* end = value + value_len; value < end; value += 6 )
475  {
476  // Note, weirdly/unfortunately AddrVal's take
477  // addresses in network order but PortVal's
478  // take ports in host order. BitTorrent specifies
479  // that both are in network order here.
480  uint32 ad = extract_uint32((u_char*) value);
481  uint16 pt = ntohs((value[4] << 8) | value[5]);
482 
483  RecordVal* peer = new RecordVal(bittorrent_peer);
484  peer->Assign(0, new AddrVal(ad));
485  peer->Assign(1, port_mgr->Get(pt, TRANSPORT_TCP));
486  res_val_peers->Assign(peer, 0);
487 
488  Unref(peer);
489  }
490  }
491  else
492  {
493  StringVal* name_ = new StringVal(name_len, name);
494  RecordVal* benc_value = new RecordVal(bittorrent_benc_value);
495  benc_value->Assign(type, new StringVal(value_len, value));
496  res_val_benc->Assign(name_, benc_value);

528 int BitTorrentTracker_Analyzer::ResponseParseBenc(void)
529  {
530 #define VIOLATION_IF(expr, msg) \
531  { \
532  if ( expr ) \
533  { \
534  ProtocolViolation(msg); \
535  stop_resp = true; \
536  return -1; \
537  } \
538  }
539 
540 #define INC_COUNT \
541  { \
542  unsigned int count = benc_count.back(); \
543  benc_count.pop_back(); \
544  benc_count.push_back(count + 1); \
545  }
546 
547  for ( unsigned int len = res_buf_len - (res_buf_pos - res_buf); len;
548  --len, ++res_buf_pos )
549  {
550  switch ( benc_state ) {
551  case BENC_STATE_EMPTY:
552  {
553  switch ( res_buf_pos[0] ) {
554  case 'd':
555  switch ( benc_stack.size() ) {
556  case 0: break;
557  case 1:
558  benc_raw = res_buf_pos;
559  benc_raw_type = BENC_TYPE_DIR;
560  /* fall through */
561  default:
562  VIOLATION_IF(benc_stack.back() == 'd' &&
563  ! (benc_count.back() % 2),
564  "BitTorrentTracker: directory key is not a string but a directory")
565  ++benc_raw_len;
566  }
567 
568  benc_stack.push_back('d');
569  benc_count.push_back(0);
570  break;
571 
572  case 'l':
573  switch ( benc_stack.size() ) {
574  case 0:
575  VIOLATION_IF(1, "BitTorrentTracker: not a bencoded directory (first char: l)")
576  /* fall through */
577 
578  case 1:
579  benc_raw = res_buf_pos;
580  benc_raw_type = BENC_TYPE_LIST;
581  /* fall through */
582 
583  default:
584  VIOLATION_IF(benc_stack.back() == 'd' &&
585  ! (benc_count.back() % 2),
586  "BitTorrentTracker: directory key is not a string but a list")
587  ++benc_raw_len;
588  }
589 
590  benc_stack.push_back('l');
591  benc_count.push_back(0);
592  break;
593 
594  case 'i':
595  VIOLATION_IF(! benc_stack.size(),
596  "BitTorrentTracker: not a bencoded directory (first char: i)")
597  VIOLATION_IF(benc_stack.back() == 'd' &&
598  ! (benc_count.back() % 2),
599  "BitTorrentTracker: directory key is not a string but an int")
600 
601  if ( benc_raw_type != BENC_TYPE_NONE )
602  ++benc_raw_len;
603 
604  benc_state = BENC_STATE_INT1;
605  break;
606 
607  case 'e':
608  VIOLATION_IF(! benc_stack.size(),
609  "BitTorrentTracker: not a bencoded directory (first char: e)")
610  VIOLATION_IF(benc_stack.back() == 'd' &&
611  benc_count.back() % 2,
612  "BitTorrentTracker: directory has an odd count of members")
613 
614  if ( benc_raw_type != BENC_TYPE_NONE )
615  ++benc_raw_len;
616 
617  if ( benc_stack.size() == 2 )
618  { // coming back to level 1
619  ResponseBenc(benc_key_len, benc_key,
620  benc_raw_type,
621  benc_raw_len, benc_raw);
622  benc_key = 0;
623  benc_key_len = 0;
624  benc_raw = 0;
625  benc_raw_len = 0;
626  benc_raw_type = BENC_TYPE_NONE;
627  }
628 
629  benc_stack.pop_back();
630  benc_count.pop_back();
631 
632  if ( benc_stack.size() )
633  INC_COUNT
634  else
635  { // benc parsing successful
636  ++res_buf_pos;
637  return 0;
638  }
639  break;
640 
641  case '0': case '1': case '2': case '3': case '4':
642  case '5': case '6': case '7': case '8': case '9':
643  VIOLATION_IF(! benc_stack.size(),
644  "BitTorrentTracker: not a bencoded directory (first char: [0-9])")
645 
646  if ( benc_raw_type != BENC_TYPE_NONE )
647  ++benc_raw_len;
648 
649  benc_strlen = res_buf_pos;
650  benc_state = BENC_STATE_STR1;
651  break;
652 
653  default:
654  VIOLATION_IF(1, "BitTorrentTracker: no valid bencoding")
655  }
656  }
657  break;
658 
659  case BENC_STATE_INT1:
660  benc_int = res_buf_pos;
661  if ( res_buf_pos[0] == '-' )
662  {
663  if ( benc_raw_type != BENC_TYPE_NONE )
664  ++benc_raw_len;
665  benc_state = BENC_STATE_INT2;
666  break;
667  }
668 
669  case BENC_STATE_INT2:
670  VIOLATION_IF(res_buf_pos[0] < '0' ||
671  res_buf_pos[0] > '9',
672  "BitTorrentTracker: no valid bencoding")
673 
674  if ( benc_raw_type != BENC_TYPE_NONE )
675  ++benc_raw_len;
676 
677  benc_state = BENC_STATE_INT3;
678  break;
679 
680  case BENC_STATE_INT3:
681  if ( res_buf_pos[0] == 'e' )
682  {
683  if ( sscanf(benc_int, FMT_INT,
684  &benc_int_val) == 1 )
685  {
686  if ( benc_stack.size() == 1 )
687  {
688  ResponseBenc(benc_key_len,
689  benc_key, BENC_TYPE_INT,
690  benc_int_val);
691  benc_key = 0;
692  benc_key_len = 0;
693  }
694  }
695  else
696  VIOLATION_IF(1, "BitTorrentTracker: no valid bencoding")
697 
698  INC_COUNT
699  benc_state = BENC_STATE_EMPTY;
700  }
701 
702  else
703  VIOLATION_IF(res_buf_pos[0] < '0' ||
704  res_buf_pos[0] > '9',
705  "BitTorrentTracker: no valid bencoding");
706 
707  if ( benc_raw_type != BENC_TYPE_NONE )
708  ++benc_raw_len;
709 
710  break;
711 
712  case BENC_STATE_STR1:
713  switch ( res_buf_pos[0] ) {
714  case '0': case '1': case '2': case '3': case '4':
715  case '5': case '6': case '7': case '8': case '9':
716  if ( benc_raw_type != BENC_TYPE_NONE )
717  ++benc_raw_len;
718  break;
719 
720  case ':':
721  VIOLATION_IF(sscanf(benc_strlen, "%u",
722  &benc_str_len) != 1,
723  "BitTorrentTracker: no valid bencoding")
724 
725  benc_str_have = 0;
726  benc_str = res_buf_pos + 1;
727 
728  if ( benc_stack.size() == 1 &&
729  ! (benc_count.front() % 2) )
