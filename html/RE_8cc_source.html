234  {

264

306 bool RE_Match_State::Match(const u_char* bv, int n,
307  bool bol, bool eol, bool clear)
308  {
309  if ( current_pos == -1 )
310  {
311  // First call to Match().
312  if ( ! dfa )
313  return false;
314 
315  // Initialize state and copy the accepting states of the start
316  // state into the acceptance set.
317  current_state = dfa->StartState();
318 
319  const AcceptingSet* ac = current_state->Accept();
320 
321  if ( ac )
322  AddMatches(*ac, 0);
323  }
324 
325  else if ( clear )
326  current_state = dfa->StartState();
327 
328  if ( ! current_state )
329  return false;
330 
331  current_pos = 0;
332 
333  size_t old_matches = accepted_matches.size();
334 
335  int ec;
336  int m = bol ? n + 1 : n;
337  int e = eol ? -1 : 0;
338 
339  while ( --m >= e )
340  {
341  if ( m == n )
342  ec = ecs[SYM_BOL];
343  else if ( m == -1 )
344  ec = ecs[SYM_EOL];
345  else
346  ec = ecs[*(bv++)];
347 
348  DFA_State* next_state = current_state->Xtion(ec,dfa);
349 
350  if ( ! next_state )
351  {
352  current_state = 0;
353  break;
354  }
355 
356  const AcceptingSet* ac = next_state->Accept();
357 
358  if ( ac )
359  AddMatches(*ac, current_pos);
360 
361  ++current_pos;
362 
363  current_state = next_state;
364  }
365 
366  return accepted_matches.size() != old_matches;
367  }
