11  {

21  // character which has its own transition gets its own equivalence
22  // class. Thus only characters which are only in character classes

29  ec->UniqueChar(sym);

85 NFA_state_list* NFA_State::EpsilonClosure()
86  {
87  if ( epsclosure )
88  return epsclosure;
89 
90  epsclosure = new NFA_state_list;
91 
92  NFA_state_list states;
93  states.append(this);
94  SetMark(this);
95 
96  int i;
97  for ( i = 0; i < states.length(); ++i )
98  {
99  NFA_State* ns = states[i];
100  if ( ns->TransSym() == SYM_EPSILON )
101  {
102  NFA_state_list* x = ns->Transitions();
103  for ( int j = 0; j < x->length(); ++j )
104  {
105  NFA_State* nxt = (*x)[j];
106  if ( ! nxt->Mark() )
107  {
108  states.append(nxt);
109  nxt->SetMark(nxt);
110  }
111  }
112 
113  if ( ns->Accept() != NO_ACCEPT )
114  epsclosure->append(ns);
115  }
116 
117  else
118  // Non-epsilon transition - keep it.
119  epsclosure->append(ns);
120  }
121 
122  // Clear out markers.
123  for ( i = 0; i < states.length(); ++i )
124  states[i]->SetMark(0);
125 
126  // Make it fit.
127  epsclosure->resize(0);
128 
129  return epsclosure;
130  }

259 void NFA_Machine::MakeRepl(int lower, int upper)
260  {
261  NFA_Machine* dup = 0;
262  if ( upper > lower || upper == NO_UPPER_BOUND )
263  dup = DuplicateMachine();
264 
265  LinkCopies(lower - 1);
266 
267  if ( upper == NO_UPPER_BOUND )
268  {

273
