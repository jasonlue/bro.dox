129 // A connection between two peers has four phases:
130 //
131 // Setup:
132 // Initial phase.
133 // VERSION messages must be exchanged.
134 // Ends when both peers have sent VERSION.
135 // Handshake:
136 // REQUEST_EVENTS/REQUEST_SYNC/CAPTURE_FILTER/CAPS/selected SERIALs
137 // may be exchanged.
138 // Phase ends when both peers have sent PHASE_DONE.
139 // State synchronization:
140 // Entered iff at least one of the peers has sent REQUEST_SYNC.
141 // The peer with the smallest runtime (incl. in VERSION msg) sends
142 // SERIAL messages compromising all of its state.
143 // Phase ends when peer sends another PHASE_DONE.
144 // Running:
145 // Peers exchange SERIAL (and PING/PONG) messages.
146 // Phase ends with connection tear-down by one of the peers.

611 void RemoteSerializer::Fork()
612  {
613  if ( child_pid )
614  return;
615 
616  // Register as a "does-count" source now.
617  iosource_mgr->Register(this, false);
618 
619  // If we are re-forking, remove old entries
620  loop_over_list(peers, i)
621  RemovePeer(peers[i]);
622 
623  // Create pipe for communication between parent and child.
624  int pipe[2];
625 
626  if ( socketpair(AF_UNIX, SOCK_STREAM, 0, pipe) < 0 )
627  {
628  Error(fmt("can't create pipe: %s", strerror(errno)));
629  return;
630  }
631 
632  // Try to increase the size of the socket send and receive buffers.
633  SetSocketBufferSize(pipe[0], SO_SNDBUF, "SO_SNDBUF", SOCKBUF_SIZE, 1);
634  SetSocketBufferSize(pipe[0], SO_RCVBUF, "SO_RCVBUF", SOCKBUF_SIZE, 0);
635  SetSocketBufferSize(pipe[1], SO_SNDBUF, "SO_SNDBUF", SOCKBUF_SIZE, 0);
636  SetSocketBufferSize(pipe[1], SO_RCVBUF, "SO_RCVBUF", SOCKBUF_SIZE, 0);
637 
638  child_pid = 0;
639 
640  int pid = fork();
641 
642  if ( pid < 0 )
643  {
644  Error(fmt("can't fork: %s", strerror(errno)));
645  return;
646  }
647 
648  if ( pid > 0 )
649  {
650  // Parent
651  child_pid = pid;
652 
653  io = new ChunkedIOFd(pipe[0], "parent->child", child_pid);
654  if ( ! io->Init() )
655  {
656  Error(fmt("can't init child io: %s", io->Error()));
657  exit(1); // FIXME: Better way to handle this?
658  }
659 
660  safe_close(pipe[1]);
661 
662  return;
663  }
664  else
665  { // child
666  SocketComm child;
667 
668  ChunkedIOFd* io =
669  new ChunkedIOFd(pipe[1], "child->parent", getppid());
670  if ( ! io->Init() )
671  {
672  Error(fmt("can't init parent io: %s", io->Error()));
673  exit(1);
674  }
675 
676  child.SetParentIO(io);
677  safe_close(pipe[0]);
678 
679  // Close file descriptors.
680  safe_close(0);
681  safe_close(1);
682  safe_close(2);
683 
684  // Be nice.
685  setpriority(PRIO_PROCESS, 0, 5);
686 
687  child.Run();
688  reporter->InternalError("cannot be reached");
689  }
