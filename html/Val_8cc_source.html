1050  {
1051  // We need to special-case a mask width of zero, since
1052  // the compiler doesn't guarantee that 1 << 32 yields 0.
1053  uint32 m[4];
1054  for ( unsigned int i = 0; i < 4; ++i )
1055  m[i] = 0;

1065  *(mp++) = 0xffffffff;

1067  *mp = ~((1 << (32 - w)) - 1);
1068
