1 // Infrastructure for serializable objects.
2 //
3 // How to make objects of class Foo serializable:
4 //
5 // 1. Derive Foo (directly or indirectly) from SerialObj.
6 // 2. Add a SER_FOO constant to SerialTypes in SerialTypes.h.
7 // 3. Add DECLARE_SERIAL(Foo) into class definition.
8 // 4. Add a (preferably protected) default ctor if it doesn't already exist.
9 // 5. For non-abstract classes, add IMPLEMENT_SERIAL(Foo, SER_FOO) to *.cc
10 // 6. Add two methods like this to *.cc (keep names of arguments!)
11 //
12 // bool Foo::DoSerialize(SerialInfo* info) const
13 // {
14 // DO_SERIALIZE(SER_FOO, ParentClassOfFoo);
15 // <... serialize class members via methods in Serializer ...>
16 // return true if everything ok;
17 // }
18 //
19 // bool Foo::DoUnserialize(UnserialInfo* info)
20 // {
21 // DO_UNSERIALIZE(ParentClassOfFoo);
22 // <... unserialize class members via methods in Serializer ...>
23 // return true if everything ok;
24 // }
25 //
26 // (7. If no parent class of Foo already contains Serialize()/Unserialize()
27 // methods, these need to be added somewhere too. But most of the various
28 // parts of the class hierarchy already have them.)
