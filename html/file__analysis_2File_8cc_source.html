382 void File::DeliverStream(const u_char* data, uint64 len)
383  {
384  bool bof_was_full = bof_buffer.full;
385  // Buffer enough data for the BOF buffer
386  BufferBOF(data, len);
387 
388  if ( ! did_metadata_inference && bof_buffer.full &&
389  LookupFieldDefaultCount(missing_bytes_idx) == 0 )
390  InferMetadata();
391 
392  DBG_LOG(DBG_FILE_ANALYSIS,
393  "[%s] %" PRIu64 " stream bytes in at offset %" PRIu64 "; %s [%s%s]",
394  id.c_str(), len, stream_offset,
395  IsComplete() ? "complete" : "incomplete",
396  fmt_bytes((const char*) data, min((uint64)40, len)),
397  len > 40 ? "..." : "");
398 
399  file_analysis::Analyzer* a = 0;
400  IterCookie* c = analyzers.InitForIteration();
401 
402  while ( (a = analyzers.NextEntry(c)) )
403  {
404  DBG_LOG(DBG_FILE_ANALYSIS, "stream delivery to analyzer %s", file_mgr->GetComponentName(a->Tag()).c_str());
405  if ( ! a->GotStreamDelivery() )
406  {
407  DBG_LOG(DBG_FILE_ANALYSIS, "skipping stream delivery to analyzer %s", file_mgr->GetComponentName(a->Tag()).c_str());
408  int num_bof_chunks_behind = bof_buffer.chunks.size();
409 
410  if ( ! bof_was_full )
411  // We just added a chunk to the BOF buffer, don't count it
412  // as it will get delivered on its own.
413  num_bof_chunks_behind -= 1;
414 
415  uint64 bytes_delivered = 0;
416 
417  // Catch this analyzer up with the BOF buffer.
418  for ( int i = 0; i < num_bof_chunks_behind; ++i )
419  {
420  if ( ! a->Skipping() )
421  {
422  if ( ! a->DeliverStream(bof_buffer.chunks[i]->Bytes(),
423  bof_buffer.chunks[i]->Len()) )
424  {
425  a->SetSkip(true);
426  analyzers.QueueRemove(a->Tag(), a->Args());
427  }
428  }
429 
430  bytes_delivered += bof_buffer.chunks[i]->Len();
431  }
432 
433  a->SetGotStreamDelivery();
434  // May need to catch analyzer up on missed gap?
435  // Analyzer should be fully caught up to stream_offset now.
436  }
437 
438  if ( ! a->Skipping() )
439  {
440  if ( ! a->DeliverStream(data, len) )
441  {
442  a->SetSkip(true);
443  analyzers.QueueRemove(a->Tag(), a->Args());
444  }
445  }
446  }
447 
448  stream_offset += len;
449  IncrementByteCount(len, seen_bytes_idx);
450  }

452 void File::DeliverChunk(const u_char* data, uint64 len, uint64 offset)
453  {
454  // Potentially handle reassembly and deliver to the stream analyzers.
455  if ( file_reassembler )
456  {
457  if ( reassembly_max_buffer > 0 &&
458  reassembly_max_buffer < file_reassembler->TotalSize() )
459  {
460  uint64 current_offset = stream_offset;
461  uint64 gap_bytes = file_reassembler->Flush();
462  IncrementByteCount(gap_bytes, overflow_bytes_idx);
463 
464  if ( FileEventAvailable(file_reassembly_overflow) )
465  {
466  val_list* vl = new val_list();
467  vl->append(val->Ref());
468  vl->append(new Val(current_offset, TYPE_COUNT));
469  vl->append(new Val(gap_bytes, TYPE_COUNT));
470  FileEvent(file_reassembly_overflow, vl);
471  }
472  }
473 
474  // Forward data to the reassembler.
475  file_reassembler->NewBlock(network_time, offset, len, data);
476  }
477  else if ( stream_offset == offset )
478  {
479  // This is the normal case where a file is transferred linearly.
480  // Nothing special should be done here.
481  DeliverStream(data, len);
482  }
483  else if ( reassembly_enabled )
484  {
485  // This is data that doesn't match the offset and the reassembler
486  // needs to be enabled.
487  file_reassembler = new FileReassembler(this, stream_offset);
488  file_reassembler->NewBlock(network_time, offset, len, data);
489  }
490  else
491  {
492  // We can't reassemble so we throw out the data for streaming.
493  IncrementByteCount(len, overflow_bytes_idx);
494  }
495 
496  DBG_LOG(DBG_FILE_ANALYSIS,
497  "[%s] %" PRIu64 " chunk bytes in at offset %" PRIu64 "; %s [%s%s]",
498  id.c_str(), len, offset,
499  IsComplete() ? "complete" : "incomplete",
500  fmt_bytes((const char*) data, min((uint64)40, len)),
501  len > 40 ? "..." : "");
502 
503  file_analysis::Analyzer* a = 0;
504  IterCookie* c = analyzers.InitForIteration();
505 
506  while ( (a = analyzers.NextEntry(c)) )
507  {
508  DBG_LOG(DBG_FILE_ANALYSIS, "chunk delivery to analyzer %s", file_mgr->GetComponentName(a->Tag()).c_str());
509  if ( ! a->Skipping() )
510  {
511  if ( ! a->DeliverChunk(data, len, offset) )
512  {
513  a->SetSkip(true);
514  analyzers.QueueRemove(a->Tag(), a->Args());
515  }
516  }
517  }
518 
519  if ( IsComplete() )
520  EndOfFile();
521  }
