367  {

370  if ( state == NETBIOS_SSN_TYPE )
371  {
372  type = *data;
373  state = NETBIOS_SSN_FLAGS;
374 
375  ++data;
376  --len;
377 
378  if ( len == 0 )
379  return;
380  }
381 
382  if ( state == NETBIOS_SSN_FLAGS )

386 
387  ++data;
388  --len;
389

392  }
393

397  state = NETBIOS_SSN_LEN_LO;
398 
399  ++data;
400  --len;
401 
402  if ( len == 0 )
403  return;
404  }
405 
406  if ( state == NETBIOS_SSN_LEN_LO )
407  {
408  msg_size += *data;
409  state = NETBIOS_SSN_BUF;
410 
411  buf_n = 0;
412 
413  if ( msg_buf )
414  {
415  if ( buf_len < msg_size )
416  {
417  delete [] msg_buf;
418  buf_len = msg_size;
419  msg_buf = new u_char[buf_len];
420  }
421  }
422  else
423  {
424  buf_len = msg_size;
425  if ( buf_len > 0 )
426  msg_buf = new u_char[buf_len];
427  }
428 
429  ++data;
430  --len;
431 
432  if ( len == 0 && msg_size != 0 )
433  return;
434  }
435 
436  if ( state != NETBIOS_SSN_BUF )
437  Conn()->Internal("state inconsistency in Contents_NetbiosSSN::Deliver");
438 
439  int n;
440  for ( n = 0; buf_n < msg_size && n < len; ++n )
441  msg_buf[buf_n++] = data[n];
442 
443  if ( buf_n < msg_size )
444  // Haven't filled up the message buffer yet, no more to do.
445  return;
446 
447  (void) interp->ParseMessage(type, flags, msg_buf, msg_size, IsOrig());
448  buf_n = 0;
449 
450  state = NETBIOS_SSN_TYPE;
