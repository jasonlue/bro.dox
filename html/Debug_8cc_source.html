474 void tokenize(const char* cstr, string& operation, vector<string>& arguments)
475  {
476  int num_tokens = 0;
477  char delim = '\0';
478  const string str(cstr);
479 
480  for ( int i = 0; i < (signed int) str.length(); ++i )
481  {
482  while ( isspace((unsigned char) str[i]) )
483  ++i;
484 
485  int start = i;
486 
487  for ( ; str[i]; ++i )
488  {
489  if ( str[i] == '\\' )
490  {
491  if ( i < (signed int) str.length() )
492  ++i;
493  }
494 
495  else if ( ! delim && str[i] == '(' )
496  delim = ')';
497 
498  else if ( ! delim && (str[i] == '\'' || str[i] == '"') )
499  delim = str[i];
500 
501  else if ( delim && str[i] == delim )
502  {
503  delim = '\0';
504  ++i;
505  break;
506  }
507 
508  else if ( ! delim && isspace(str[i]) )
509  break;
510  }
511 
512  size_t len = i - start;
513 
514  if ( ! num_tokens )
515  operation = string(str, start, len);
516  else
517  arguments.push_back(string(str, start, len));
518 
519  ++num_tokens;
520  }
