6 // BaseQueue.h --
7 // Interface for class BaseQueue, current implementation is as an
8 // array of ent's. This implementation was chosen to optimize
9 // getting to the ent's rather than inserting and deleting.
10 // Also push's and pop's from the front or the end of the queue
11 // are very efficient. The only really expensive operation
12 // is resizing the list, which involves getting new space
13 // and moving the data. Resizing occurs automatically when inserting
14 // more elements than the list can currently hold. Automatic
15 // resizing is done one "chunk_size" of elements at a time and
16 // always increases the size of the list. Resizing to zero
17 // (or to less than the current value of num_entries)
18 // will decrease the size of the list to the current number of
19 // elements. Resize returns the new max_entries.
20 //
21 // Entries must be either a pointer to the data or nonzero data with
22 // sizeof(data) <= sizeof(void*).
