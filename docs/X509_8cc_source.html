89 RecordVal* file_analysis::X509::ParseCertificate(X509Val* cert_val, const char* fid)
90  {
91  ::X509* ssl_cert = cert_val->GetCertificate();
92 
93  char buf[2048]; // we need a buffer for some of the openssl functions
94  memset(buf, 0, sizeof(buf));
95 
96  RecordVal* pX509Cert = new RecordVal(BifType::Record::X509::Certificate);
97  BIO *bio = BIO_new(BIO_s_mem());
98 
99  pX509Cert->Assign(0, new Val((uint64) X509_get_version(ssl_cert) + 1, TYPE_COUNT));
100  i2a_ASN1_INTEGER(bio, X509_get_serialNumber(ssl_cert));
101  int len = BIO_read(bio, buf, sizeof(buf));
102  pX509Cert->Assign(1, new StringVal(len, buf));
103  BIO_reset(bio);
104 
105  X509_NAME_print_ex(bio, X509_get_subject_name(ssl_cert), 0, XN_FLAG_RFC2253);
106  len = BIO_gets(bio, buf, sizeof(buf));
107  pX509Cert->Assign(2, new StringVal(len, buf));
108  BIO_reset(bio);
109 
110  X509_NAME *subject_name = X509_get_subject_name(ssl_cert);
111  // extract the most specific (last) common name from the subject
112  int namepos = -1;
113  for ( ;; )
114  {
115  int j = X509_NAME_get_index_by_NID(subject_name, NID_commonName, namepos);
116  if ( j == -1 )
117  break;
118 
119  namepos = j;
120  }
121 
122  if ( namepos != -1 )
123  {
124  // we found a common name
125  ASN1_STRING_print(bio, X509_NAME_ENTRY_get_data(X509_NAME_get_entry(subject_name, namepos)));
126  len = BIO_gets(bio, buf, sizeof(buf));
127  pX509Cert->Assign(4, new StringVal(len, buf));
128  BIO_reset(bio);
129  }
130 
131  X509_NAME_print_ex(bio, X509_get_issuer_name(ssl_cert), 0, XN_FLAG_RFC2253);
132  len = BIO_gets(bio, buf, sizeof(buf));
133  pX509Cert->Assign(3, new StringVal(len, buf));
134  BIO_free(bio);
135 
136  pX509Cert->Assign(5, new Val(GetTimeFromAsn1(X509_get_notBefore(ssl_cert), fid, reporter), TYPE_TIME));
137  pX509Cert->Assign(6, new Val(GetTimeFromAsn1(X509_get_notAfter(ssl_cert), fid, reporter), TYPE_TIME));
138 
139  // we only read 255 bytes because byte 256 is always 0.
140  // if the string is longer than 255, that will be our null-termination,
141  // otherwhise i2t does null-terminate.
142  ASN1_OBJECT *algorithm;
143  X509_PUBKEY_get0_param(&algorithm, NULL, NULL, NULL, X509_get_X509_PUBKEY(ssl_cert));
144  if ( ! i2t_ASN1_OBJECT(buf, 255, algorithm) )
145  buf[0] = 0;
146 
147  pX509Cert->Assign(7, new StringVal(buf));
148 
149  // Special case for RDP server certificates. For some reason some (all?) RDP server
150  // certificates like to specify their key algorithm as md5WithRSAEncryption, which
151  // is wrong on so many levels. We catch this special case here and set it to what is
152  // actually should be (namely - rsaEncryption), so that OpenSSL will parse out the
153  // key later. Otherwise it will just fail to parse the certificate key.
154 
155  if ( OBJ_obj2nid(algorithm) == NID_md5WithRSAEncryption )
156  {
157  ASN1_OBJECT *copy = OBJ_dup(algorithm); // the next line will destroy the original algorithm.
158  X509_PUBKEY_set0_param(X509_get_X509_PUBKEY(ssl_cert), OBJ_nid2obj(NID_rsaEncryption), 0, NULL, NULL, 0);
159  algorithm = copy;
160  // we do not have to worry about freeing algorithm in that case - since it will be re-assigned using
161  // set0_param and the cert will take ownership.
162  }
163  else
164  algorithm = 0;
165 
166  if ( ! i2t_ASN1_OBJECT(buf, 255, OBJ_nid2obj(X509_get_signature_nid(ssl_cert))) )
167  buf[0] = 0;
168 
169  pX509Cert->Assign(8, new StringVal(buf));
170 
171  // Things we can do when we have the key...
172  EVP_PKEY *pkey = X509_extract_key(ssl_cert);
173  if ( pkey != NULL )
174  {
175  if ( EVP_PKEY_base_id(pkey) == EVP_PKEY_DSA )
176  pX509Cert->Assign(9, new StringVal("dsa"));
177 
178  else if ( EVP_PKEY_base_id(pkey) == EVP_PKEY_RSA )
179  {
180  pX509Cert->Assign(9, new StringVal("rsa"));
181 
182  const BIGNUM *e;
183  RSA_get0_key(EVP_PKEY_get0_RSA(pkey), NULL, &e, NULL);
184  char *exponent = BN_bn2dec(e);
185  if ( exponent != NULL )
186  {
187  pX509Cert->Assign(11, new StringVal(exponent));
188  OPENSSL_free(exponent);
189  exponent = NULL;
190  }
191  }
192 #ifndef OPENSSL_NO_EC
193  else if ( EVP_PKEY_base_id(pkey) == EVP_PKEY_EC )
194  {
195  pX509Cert->Assign(9, new StringVal("ecdsa"));
196  pX509Cert->Assign(12, KeyCurve(pkey));
197  }
198 #endif
199 
200  // set key algorithm back. We do not have to free the value that we created because (I think) it
201  // comes out of a static array from OpenSSL memory.
202  if ( algorithm )
203  X509_PUBKEY_set0_param(X509_get_X509_PUBKEY(ssl_cert), algorithm, 0, NULL, NULL, 0);
204 
205  unsigned int length = KeyLength(pkey);
206  if ( length > 0 )
207  pX509Cert->Assign(10, new Val(length, TYPE_COUNT));
208 
209  EVP_PKEY_free(pkey);
210  }
211 
212 
213  return pX509Cert;
214  }

267  if ( ! altname )

269  reporter->Weird(fmt("Could not parse subject alternative names. fuid %s", GetFile()->GetID().c_str()));
270  return;
271  }
272 
273  VectorVal* names = 0;
274  VectorVal* emails = 0;
275  VectorVal* uris = 0;
276  VectorVal* ips = 0;
277 
278  unsigned int otherfields = 0;
279 
280  for ( int i = 0; i < sk_GENERAL_NAME_num(altname); i++ )
281  {
282  GENERAL_NAME *gen = sk_GENERAL_NAME_value(altname, i);
283  assert(gen);
284 
285  if ( gen->type == GEN_DNS || gen->type == GEN_URI || gen->type == GEN_EMAIL )
286  {
287  if ( ASN1_STRING_type(gen->d.ia5) != V_ASN1_IA5STRING )
288  {
289  reporter->Weird(fmt("DNS-field does not contain an IA5String. fuid %s", GetFile()->GetID().c_str()));
290  continue;
291  }
292 
293 #if ( OPENSSL_VERSION_NUMBER < 0x10100000L )
294  const char* name = (const char*) ASN1_STRING_data(gen->d.ia5);
295 #else
296  const char* name = (const char*) ASN1_STRING_get0_data(gen->d.ia5);
297 #endif
298  StringVal* bs = new StringVal(name);
299 
300  switch ( gen->type )
301  {
302  case GEN_DNS:
303  if ( names == 0 )
304  names = new VectorVal(internal_type("string_vec")->AsVectorType());
305 
306  names->Assign(names->Size(), bs);
307  break;
308 
309  case GEN_URI:
310  if ( uris == 0 )
311  uris = new VectorVal(internal_type("string_vec")->AsVectorType());
312 
313  uris->Assign(uris->Size(), bs);
314  break;
315 
316  case GEN_EMAIL:
317  if ( emails == 0 )
318  emails = new VectorVal(internal_type("string_vec")->AsVectorType());
319 
320  emails->Assign(emails->Size(), bs);
321  break;
322  }
323  }
324 
325  else if ( gen->type == GEN_IPADD )
326  {
327  if ( ips == 0 )
328  ips = new VectorVal(internal_type("addr_vec")->AsVectorType());
329 
330  uint32* addr = (uint32*) gen->d.ip->data;
331 
332  if( gen->d.ip->length == 4 )
333  ips->Assign(ips->Size(), new AddrVal(*addr));
334 
335  else if ( gen->d.ip->length == 16 )
336  ips->Assign(ips->Size(), new AddrVal(addr));
337 
338  else

341  continue;
342  }
343  }
344 
345  else
346  {
347  // reporter->Error("Subject alternative name contained unsupported fields. fuid %s", GetFile()->GetID().c_str());
348  // This happens quite often - just mark it

352  }

355 
356  if ( names != 0 )

358 
359  if ( uris != 0 )
360  sanExt->Assign(1, uris);
361 
362  if ( emails != 0 )
363  sanExt->Assign(2, emails);
364 
365  if ( ips != 0 )

367 
368  sanExt->Assign(4, new Val(otherfields, TYPE_BOOL));
369 
370  val_list* vl = new val_list();
371  vl->append(GetFile()->GetVal()->Ref());
372  vl->append(sanExt);
373  mgr.QueueEvent(x509_ext_subject_alternative_name, vl);
374  GENERAL_NAMES_free(altname);
375  }

410 unsigned int file_analysis::X509::KeyLength(EVP_PKEY *key)
411  {
412  assert(key != NULL);
413 
414  switch(EVP_PKEY_base_id(key)) {
415  case EVP_PKEY_RSA:
416  const BIGNUM *n;
417  RSA_get0_key(EVP_PKEY_get0_RSA(key), &n, NULL, NULL);
418  return BN_num_bits(n);
419 
420  case EVP_PKEY_DSA:
421  const BIGNUM *p;
422  DSA_get0_pqg(EVP_PKEY_get0_DSA(key), &p, NULL, NULL);
423  return BN_num_bits(p);
424 
425 #ifndef OPENSSL_NO_EC
426  case EVP_PKEY_EC:
427  {
428  BIGNUM* ec_order = BN_new();
429  if ( ! ec_order )
430  // could not malloc bignum?
431  return 0;
432 
433  const EC_GROUP *group = EC_KEY_get0_group(EVP_PKEY_get0_EC_KEY(key));
434 
435  if ( ! group )
436  {
437  // unknown ex-group
438  BN_free(ec_order);
439  return 0;
440  }
441 
442  if ( ! EC_GROUP_get_order(group, ec_order, NULL) )
443  {
444  // could not get ec-group-order
445  BN_free(ec_order);
446  return 0;
447  }
448 
449  unsigned int length = BN_num_bits(ec_order);
450  BN_free(ec_order);
451  return length;
452  }
453 #endif
454  default:
455  return 0; // unknown public key type
456  }
457 
458  reporter->InternalError("cannot be reached");
459  }
