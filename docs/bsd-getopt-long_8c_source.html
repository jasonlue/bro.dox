111 /*
112  * Compute the greatest common divisor of a and b.
113  */
114 static int pure_gcd(int a, int b)
115 {
116  int c;
117 
118  c = a % b;
119  while (c != 0) {
120  a = b;
121  b = c;
122  c = a % b;
123  }
124  return b;
125 }

127 /*
128  * Exchange the block from nonopt_start to nonopt_end with the block
129  * from nonopt_end to opt_end (keeping the same order of arguments
130  * in each block).
131  */
132 static void pure_permute_args(int panonopt_start, int panonopt_end,
133  int opt_end, char * const *nargv)
134 {
135  int cstart, cyclelen, i, j, ncycle, nnonopts, nopts, pos;
136  char *swap;
137 
138  /*
139  * compute lengths of blocks and number and size of cycles
140  */
141  nnonopts = panonopt_end - panonopt_start;
142  nopts = opt_end - panonopt_end;
143  ncycle = pure_gcd(nnonopts, nopts);
144  cyclelen = (opt_end - panonopt_start) / ncycle;
145 
146  for (i = 0; i < ncycle; i++) {
147  cstart = panonopt_end+i;
148  pos = cstart;
149  for (j = 0; j < cyclelen; j++) {
150  if (pos >= panonopt_end)
151  pos -= nnonopts;
152  else
153  pos += nopts;
154  swap = nargv[pos];
155  /* LINTED const cast */
156  ((char **) nargv)[pos] = nargv[cstart];
157  /* LINTED const cast */
158  ((char **)nargv)[cstart] = swap;
159  }
160  }
161 }

163 /*
164  * pure_parse_long_options --
165  * Parse long options in argc/argv argument vector.
166  * Returns -1 if short_too is set and the option does not match long_options.
167  */
168 static int pure_parse_long_options(char * const *nargv, const char *options,
169  const struct pure_option *long_options,
170  int *idx, int short_too)
171 {
172  const char *current_argv, *has_equal;
173  size_t current_argv_len;
174  int i, match;
175 
176  current_argv = pure_place;
177  match = -1;
178 
179  pure_optind++;
180 
181  if ((has_equal = strchr(current_argv, '=')) != NULL) {
182  /* argument found (--option=arg) */
183  current_argv_len = has_equal - current_argv;
184  has_equal++;
185  } else
186  current_argv_len = strlen(current_argv);
187 
188  for (i = 0; long_options[i].name; i++) {
189  /* find matching long option */
190  if (strncmp(current_argv, long_options[i].name,
191  current_argv_len))
192  continue;
193 
194  if (strlen(long_options[i].name) == current_argv_len) {
195  /* exact match */
196  match = i;
197  break;
198  }
199  /*
200  * If this is a known short option, don't allow
201  * a partial match of a single character.
202  */
203  if (short_too && current_argv_len == 1)
204  continue;
205 
206  if (match == -1) /* partial match */
207  match = i;
208  else {
209  /* ambiguous abbreviation */
210  if (PRINT_ERROR)
211  fprintf(stderr, ambig, (int)current_argv_len,
212  current_argv);
213  pure_optopt = 0;
214  return BADCH;
215  }
216  }
217  if (match != -1) { /* option found */
218  if (long_options[match].has_arg == no_argument
219  && has_equal) {
220  if (PRINT_ERROR)
221  fprintf(stderr, noarg, (int)current_argv_len,
222  current_argv);
223  /*
224  * XXX: GNU sets pure_optopt to val regardless of flag
225  */
226  if (long_options[match].flag == NULL)
227  pure_optopt = long_options[match].val;
228  else
229  pure_optopt = 0;
230  return BADARG;
231  }
232  if (long_options[match].has_arg == required_argument ||
233  long_options[match].has_arg == optional_argument) {
234  if (has_equal)
235  pure_optarg = has_equal;
236  else if (long_options[match].has_arg ==
237  required_argument) {
238  /*
239  * optional argument doesn't use next nargv
240  */
241  pure_optarg = nargv[pure_optind++];
242  }
243  }
244  if ((long_options[match].has_arg == required_argument)
245  && (pure_optarg == NULL)) {
246  /*
247  * Missing argument; leading ':' indicates no error
248  * should be generated.
249  */
250  if (PRINT_ERROR)
251  fprintf(stderr, recargstring,
252  current_argv);
253  /*
254  * XXX: GNU sets pure_optopt to val regardless of flag
255  */
256  if (long_options[match].flag == NULL)
257  pure_optopt = long_options[match].val;
258  else
259  pure_optopt = 0;
260  --pure_optind;
261  return BADARG;
262  }
263  } else { /* unknown option */
264  if (short_too) {
265  --pure_optind;
266  return -1;
267  }
268  if (PRINT_ERROR)
269  fprintf(stderr, illoptstring, current_argv);
270  pure_optopt = 0;
271  return BADCH;
272  }
273  if (idx)
274  *idx = match;
275  if (long_options[match].flag) {
276  *long_options[match].flag = long_options[match].val;
277  return 0;
278  } else
279  return long_options[match].val;
280 }

283  * getopt_internal --
284  * Parse argc/argv argument vector. Called by user level routines.
285  */
286 static int pure_getopt_internal(int nargc, char * const *nargv,
287  const char *options,
288  const struct pure_option *long_options,
289  int *idx, int flags)
290 {
291  char *oli; /* option letter list index */
292  int optchar, short_too;
293  static int posixly_correct = -1;
294 
295  if (options == NULL)
296  return -1;
297 
298  /*
299  * Disable GNU extensions if POSIXLY_CORRECT is set or options
300  * string begins with a '+'.
301  */
302  if (posixly_correct == -1)
303  posixly_correct = (getenv("POSIXLY_CORRECT") != NULL);
304  if (posixly_correct || *options == '+')
305  flags &= ~FLAG_PERMUTE;
306  else if (*options == '-')
307  flags |= FLAG_ALLARGS;
308  if (*options == '+' || *options == '-')
309  options++;
310 
311  /*
312  * XXX Some GNU programs (like cvs) set pure_optind to 0 instead of
313  * XXX using pure_optreset. Work around this braindamage.
314  */
315  if (pure_optind == 0)
316  pure_optind = pure_optreset = 1;
317 
318  pure_optarg = NULL;
319  if (pure_optreset)
320  nonopt_start = nonopt_end = -1;
321  start:
322  if (pure_optreset || !*pure_place) { /* update scanning pointer */
323  pure_optreset = 0;
324  if (pure_optind >= nargc) { /* end of argument vector */
325  pure_place = EMSG;
326  if (nonopt_end != -1) {
327  /* do permutation, if we have to */
328  pure_permute_args(nonopt_start, nonopt_end,
329  pure_optind, nargv);
330  pure_optind -= nonopt_end - nonopt_start;
331  }
332  else if (nonopt_start != -1) {
333  /*
334  * If we skipped non-options, set pure_optind
335  * to the first of them.
336  */
337  pure_optind = nonopt_start;
338  }
339  nonopt_start = nonopt_end = -1;
340  return -1;
341  }
342  if (*(pure_place = nargv[pure_optind]) != '-' ||
343  (pure_place[1] == '\0' && strchr(options, '-') == NULL)) {
344  pure_place = EMSG; /* found non-option */
345  if (flags & FLAG_ALLARGS) {
346  /*
347  * GNU extension:
348  * return non-option as argument to option 1
349  */
350  pure_optarg = nargv[pure_optind++];
351  return INORDER;
352  }
353  if (!(flags & FLAG_PERMUTE)) {
354  /*
355  * If no permutation wanted, stop parsing
356  * at first non-option.
357  */
358  return -1;
359  }
360  /* do permutation */
361  if (nonopt_start == -1)
362  nonopt_start = pure_optind;
363  else if (nonopt_end != -1) {
364  pure_permute_args(nonopt_start, nonopt_end,
365  pure_optind, nargv);
366  nonopt_start = pure_optind -
367  (nonopt_end - nonopt_start);
368  nonopt_end = -1;
369  }
370  pure_optind++;
371  /* process next argument */
372  goto start;
373  }
374  if (nonopt_start != -1 && nonopt_end == -1)
375  nonopt_end = pure_optind;
376 
377  /*
378  * Check for "--" or "--foo" with no long options
379  * but if pure_place is simply "-" leave it unmolested.
380  */
381 
382  if (pure_place[1] != '\0' && *++pure_place == '-' &&
383  (pure_place[1] == '\0' || long_options == NULL)) {
384  pure_optind++;
385  pure_place = EMSG;
386  /*
387  * We found an option (--), so if we skipped
388  * non-options, we have to permute.
389  */
390  if (nonopt_end != -1) {
391  pure_permute_args(nonopt_start, nonopt_end,
392  pure_optind, nargv);
393  pure_optind -= nonopt_end - nonopt_start;
394  }
395  nonopt_start = nonopt_end = -1;
396  return -1;
397  }
398  }
399 
400  /*
401  * Check long options if:
402  * 1) we were passed some
403  * 2) the arg is not just "-"
404  * 3) either the arg starts with -- we are pure_getopt_long_only()
405  */
406  if (long_options != NULL && pure_place != nargv[pure_optind] &&
407  (*pure_place == '-' || (flags & FLAG_LONGONLY))) {
408  short_too = 0;
409  if (*pure_place == '-')
410  pure_place++; /* --foo long option */
411  else if (*pure_place != ':' && strchr(options, *pure_place) != NULL)
412  short_too = 1; /* could be short option too */
413 
414  optchar = pure_parse_long_options(nargv, options, long_options,
415  idx, short_too);
416  if (optchar != -1) {
417  pure_place = EMSG;
418  return optchar;
419  }
420  }
421 
422  if ((optchar = (int) *pure_place++) == ':' ||
423  (optchar == '-' && *pure_place != '\0') ||
424  (oli = strchr(options, optchar)) == NULL) {
425  /*
426  * If the user specified "-" and '-' isn't listed in
427  * options, return -1 (non-option) as per POSIX.
428  * Otherwise, it is an unknown option character (or :').
429  */
430  if (optchar == '-' && *pure_place == '\0')
431  return -1;
432  if (!*pure_place)
433  ++pure_optind;
434  if (PRINT_ERROR)
435  fprintf(stderr, illoptchar, optchar);
436  pure_optopt = optchar;
437  return BADCH;
438  }
439  if (long_options != NULL && optchar == 'W' && oli[1] == ';') {
440  /* -W long-option */
441  if (*pure_place) /* no space */
442  /* NOTHING */;
443  else if (++pure_optind >= nargc) { /* no arg */
444  pure_place = EMSG;
445  if (PRINT_ERROR)
446  fprintf(stderr, recargchar, optchar);
447  pure_optopt = optchar;
448  return BADARG;
449  } else /* white space */
450  pure_place = nargv[pure_optind];
451  optchar = pure_parse_long_options(nargv, options, long_options,
452  idx, 0);
453  pure_place = EMSG;
454  return optchar;
455  }
456  if (*++oli != ':') { /* doesn't take argument */
457  if (!*pure_place)
458  ++pure_optind;
459  } else { /* takes (optional) argument */
460  pure_optarg = NULL;
461  if (*pure_place) /* no white space */
462  pure_optarg = pure_place;
463  /* XXX: disable test for :: if PC? (GNU doesn't) */
464  else if (oli[1] != ':') { /* arg not optional */
465  if (++pure_optind >= nargc) { /* no arg */
466  pure_place = EMSG;
467  if (PRINT_ERROR)
468  fprintf(stderr, recargchar, optchar);
469  pure_optopt = optchar;
470  return BADARG;
471  } else {
472  pure_optarg = nargv[pure_optind];
473  }
474  }
475  pure_place = EMSG;
476  ++pure_optind;
477  }
478  /* dump back option letter */
479  return optchar;
480 }
