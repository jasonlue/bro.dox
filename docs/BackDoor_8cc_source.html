127 void BackDoorEndpoint::CheckForRlogin(uint64 seq, int len, const u_char* data)
128  {
129  if ( rlogin_checking_done )
130  return;
131 
132  // Looking for pattern:
133  // <null>string<null>string<null>string/string<null>
134  // where all string's are non-empty 7-bit-ascii string
135  //
136  // To avoid having to reassemble, we keep testing each byte until
137  // one of the following happens:
138  //
139  // - A gap in sequence number occurs
140  // - Four null's have been found
141  // - The number of bytes we examined reaches RLOGIN_MAX_SIGNATURE_LENGTH
142  // - An empty or non-7-bit-ascii string is found
143  //
144  if ( seq == 1 )
145  { // Check if first byte is a NUL.
146  if ( data[0] == 0 )
147  {
148  rlogin_num_null = 1;
149 
150  if ( ! endp->IsOrig() )
151  {
152  RloginSignatureFound(len);
153  return;
154  }
155 
156  rlogin_string_separator_pos = 1;
157 
158  ++seq; // move past the byte
159  ++data;
160  --len;
161  }
162  else
163  {
164  rlogin_checking_done = 1;
165  return;
166  }
167  }
168 
169  if ( seq > max_top_seq && max_top_seq != 0 )
170  { // A gap! Since we don't reassemble things, stop now.
171  RloginSignatureFound(0);
172  return;
173  }
174 
175  if ( seq + len <= max_top_seq )
176  return; // nothing new
177 
178  if ( seq < max_top_seq )
179  { // trim to just the new data
180  int64 delta = max_top_seq - seq;
181  seq += delta;
182  data += delta;
183  len -= delta;
184  }
185 
186  // Search for rlogin signature.
187  for ( int i = 0; i < len && rlogin_num_null < 4; ++i )
188  {
189  if ( data[i] == 0 )
190  {
191  if ( i + seq == rlogin_string_separator_pos + 1 )
192  { // Empty string found.
193  rlogin_checking_done = 1;
194  return;
195  }
196  else
197  {
198  rlogin_string_separator_pos = i + seq;
199  ++rlogin_num_null;
200  }
201  }
202 
203  else if ( data[i] == '/' )
204  {
205  if ( rlogin_num_null == 3 )
206  {
207  if ( i + seq == rlogin_string_separator_pos + 1 )
208  { // Empty terminal type.
209  rlogin_checking_done = 1;
210  return;
211  }
212 
213  rlogin_string_separator_pos = i + seq;
214  rlogin_slash_seen = 1;
215  }
216  }
217 
218  else if ( data[i] >= 128 )
219  { // Non-7-bit-ascii
220  rlogin_checking_done = 1;
221  return;
222  }
223  }
224 
225  if ( rlogin_num_null == 4 )
226  {
227  if ( rlogin_slash_seen )
228  RloginSignatureFound(0);
229  else
230  rlogin_checking_done = 1;
231 
232  return;
233  }
234 
235  if ( seq + len > RLOGIN_MAX_SIGNATURE_LENGTH )
236  { // We've waited for too long
237  RloginSignatureFound(0);
238  return;
239  }
240  }

258 void BackDoorEndpoint::CheckForTelnet(uint64 /* seq */, int len, const u_char* data)
259  {
260  if ( len >= 3 &&
261  data[0] == TELNET_IAC && IS_TELNET_NEGOTIATION_CMD(data[1]) )
262  {
263  TelnetSignatureFound(len);
264  return;
265  }
266 
267  // Note, we do the analysis per-packet rather than on the reassembled
268  // stream. This is a lot more efficient as then we don't need to
269  // do stream reassembly; but it's potentially less accurate, and
270  // subject to evasion. *But*: backdoor detection is inherently
271  // subject to a wide variety of evasion, so allowing this form
272  // (which is a pain to exploit) costs little.
273 
274  num_bytes += len;
275 
276  int last_char = 0;
277  int offset = 0; // where we consider the latest line to have begun
278  int option_length = 0; // length of options in a line
279 
280  for ( int i = 0; i < len; ++i )
281  {
282  unsigned int c = data[i];
283 
284  if ( c == '\n' && last_char == '\r' )
285  {
286  // Compress CRLF to just one line termination.
287  last_char = c;
288  continue;
289  }
290 
291  if ( c == '\n' || c == '\r' )
292  {
293  ++num_lines;
294 
295  if ( i - offset - option_length <= NORMAL_LINE_LENGTH )
296  ++num_normal_lines;
297 
298  option_length = 0;
299  offset = i;
300  }
301 
302  else if ( c == TELNET_IAC )
303  {
304  ++option_length;
305  --num_bytes;
306 
307  if ( ++i < len )
308  {
309  unsigned int code = data[i];
310  if ( code == TELNET_IAC )
311  // Escaped IAC.
312  last_char = code;
313 
314  else if ( code >= 251 && code <= 254 )
315  { // 3-byte option: ignore next byte
316  ++i;
317  option_length += 2;
318  num_bytes -= 2;
319  }
320 
321  else
322  // XXX: We don't deal with sub option for simplicity
323  // although we SHOULD!
324  {
325  ++option_length;
326  --num_bytes;
327  }
328  }
329  continue;
330  }
331 
332  else if ( c != 0 && c < 128 )
333  ++num_7bit_ascii;
334 
335  last_char = c;
336  }
337  }

349 void BackDoorEndpoint::CheckForSSH(uint64 seq, int len, const u_char* data)
350  {
351  if ( seq == 1 && CheckForString("SSH-", data, len) && len > 4 &&
352  (data[4] == '1' || data[4] == '2') )
353  {
354  SignatureFound(ssh_signature_found, 1);
355  return;
356  }
357 
358  // Check for length pattern.
359 
360  if ( seq < max_top_seq || max_top_seq == 0 )
361  // Retransmission involved, or first pkt => size info useless.
362  return;
363 
364  if ( seq > max_top_seq )
365  { // Estimate number of packets in the sequence gap
366  int64 gap = seq - max_top_seq;
367  if ( gap > 0 )
368  num_pkts += uint64((gap + DEFAULT_MTU - 1) / DEFAULT_MTU);
369  }
370 
371  ++num_pkts;
372 
373  // According to the spec:
374  // SSH 1.x pkts have size 8k+4
375  // SSH 2.x pkts have size 8k >= 16 (most cipher blocks are 8n)
376  if ( len <= 127 )
377  switch ( len & 7 ) {
378  case 0:
379  if ( len >= 16 )
380  ++num_8k0_pkts;
381  break;
382 
383  case 4:
384  ++num_8k4_pkts;
385  break;
386  }
387  else
388  { // len is likely to be some MTU.
389  }

569 void BackDoorEndpoint::CheckForHTTP(uint64 seq, int len, const u_char* data)
570  {
571  // According to the RFC, we should look for
572  // '<method> SP <url> SP HTTP/<version> CR LF'
573  // where:
574  //
575  // <method> = GET | HEAD | POST
576  //
577  // (i.e., HTTP 1.1 methods are ignored for now)
578  // <version> = 1.0 | 1.1.
579  //
580  // However, this is probably too restrictive to catch 'non-standard'
581  // requests. Instead, we look for certain methods only in the first
582  // line of the first packet only.
583  //
584  // "The method is case-sensitive." -- RFC 2616
585 
586  const char* http_method[] = { "GET", "HEAD", "POST", 0 };
587 
588  if ( seq != 1 )
589  return; // first packet only
590 
591  // Pick up the method.
592  int pos = skip_http_whitespace (data, len, 0);
593  if ( pos < 0 )
594  return;
595 
596  int method;
597  for ( method = 0; http_method[method]; ++method )
598  {
599  const char* s = http_method[method];
600  int i;
601  for ( i = pos; i < len; ++i, ++s )
602  if ( data[i] != *s )
603  break;
604 
605  if ( *s == '\0' )
606  {
607  pos = i;
608  break;
609  }
610  }
611 
612  if ( ! http_method[method] )
613  return;
614 
615  if ( pos >= len || ! is_http_whitespace(data[pos]) )
616  return;
617 
618  if ( http_signature_found )
619  SignatureFound(http_signature_found);
620 
621  if ( http_proxy_signature_found )
622  {
623  const u_char* rest = data + pos;
624  int rest_len = len - pos;
625 
626  pos = skip_http_whitespace(rest, rest_len, rest_len);
627 
628  if ( pos >= 0 )
629  CheckForHTTPProxy(seq, rest_len - pos, rest + pos);
630  }
631  }
