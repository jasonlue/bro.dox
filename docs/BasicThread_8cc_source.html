188 void* BasicThread::launcher(void *arg)
189  {
190  static_assert(std::is_same<std::thread::native_handle_type, pthread_t>::value, "libstdc++ doesn't use pthread_t");
191  BasicThread* thread = (BasicThread *)arg;
192 
193  // Block signals in thread. We handle signals only in the main
194  // process.
195  sigset_t mask_set;
196  sigfillset(&mask_set);
197 
198  // Unblock the signals where according to POSIX the result is undefined if they are blocked
199  // in a thread and received by that thread. If those are not unblocked, threads will just
200  // hang when they crash without the user being notified.
201  sigdelset(&mask_set, SIGFPE);
202  sigdelset(&mask_set, SIGILL);
203  sigdelset(&mask_set, SIGSEGV);
204  sigdelset(&mask_set, SIGBUS);
205  int res = pthread_sigmask(SIG_BLOCK, &mask_set, 0);
206  assert(res == 0);
207 
208  // Run thread's main function.
209  thread->Run();
210 
211  thread->Done();
212 
213  return 0;
214  }
