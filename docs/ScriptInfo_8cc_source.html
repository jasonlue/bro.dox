173 void ScriptInfo::DoInitPostScript()
174  {
175  for ( id_info_map::const_iterator it = id_info.begin();
176  it != id_info.end(); ++it )
177  {
178  IdentifierInfo* info = it->second;
179  ID* id = info->GetID();
180 
181  if ( ! broxygen::is_public_api(id) )
182  continue;
183 
184  if ( id->AsType() )
185  {
186  types.push_back(info);
187  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a type",
188  id->Name(), name.c_str());
189  continue;
190  }
191 
192  if ( IsFunc(id->Type()->Tag()) )
193  {
194  switch ( id->Type()->AsFuncType()->Flavor() ) {
195  case FUNC_FLAVOR_HOOK:
196  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a hook",
197  id->Name(), name.c_str());
198  hooks.push_back(info);
199  break;
200  case FUNC_FLAVOR_EVENT:
201  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a event",
202  id->Name(), name.c_str());
203  events.push_back(info);
204  break;
205  case FUNC_FLAVOR_FUNCTION:
206  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a function",
207  id->Name(), name.c_str());
208  functions.push_back(info);
209  break;
210  default:
211  reporter->InternalError("Invalid function flavor");
212  break;
213  }
214 
215  continue;
216  }
217 
218  if ( id->IsConst() )
219  {
220  if ( id->FindAttr(ATTR_REDEF) )
221  {
222  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a redef_option",
223  id->Name(), name.c_str());
224  redef_options.push_back(info);
225  }
226  else
227  {
228  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a constant",
229  id->Name(), name.c_str());
230  constants.push_back(info);
231  }
232 
233  continue;
234  }
235  else if ( id->IsOption() )
236  {
237  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as an runtime option",
238  id->Name(), name.c_str());
239  options.push_back(info);
240 
241  continue;
242  }
243 
244  if ( id->Type()->Tag() == TYPE_ENUM )
245  // Enums are always referenced/documented from the type's
246  // documentation.
247  continue;
248 
249  DBG_LOG(DBG_BROXYGEN, "Filter id '%s' in '%s' as a state variable",
250  id->Name(), name.c_str());
251  state_vars.push_back(info);
252  }
253 
254  // The following enum types are automatically created internally in Bro,
255  // so just manually associating them with scripts for now.
256  if ( name == "base/frameworks/input/main.bro" )
257  {
258  auto id = global_scope()->Lookup("Input::Reader");
259  types.push_back(new IdentifierInfo(id, this));
260  }
261  else if ( name == "base/frameworks/logging/main.bro" )
262  {
263  auto id = global_scope()->Lookup("Log::Writer");
264  types.push_back(new IdentifierInfo(id, this));
265  }
266  }
