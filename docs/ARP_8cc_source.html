78 void ARP_Analyzer::NextPacket(double t, const Packet* pkt)
79  {
80  const u_char *data = pkt->data;
81  // Check whether the packet is OK ("inspired" in tcpdump's print-arp.c).
82  const struct arp_pkthdr* ah =
83  (const struct arp_pkthdr*) (data + pkt->hdr_size);
84 
85  // Check the size.
86  int min_length = (ar_tpa(ah) - (char*) (data + pkt->hdr_size)) + ah->ar_pln;
87  int real_length = pkt->cap_len - pkt->hdr_size;
88  if ( min_length > real_length )
89  {
90  Corrupted("truncated_ARP");
91  return;
92  }
93 
94  char errbuf[1024];
95 
96  // Check the address description fields.
97  switch ( ntohs(ah->ar_hrd) ) {
98  case ARPHRD_ETHER:
99  if ( ah->ar_hln != 6 )
100  { // don't know how to handle the opcode
101  snprintf(errbuf, sizeof(errbuf),
102  "corrupt-arp-header (hrd=%i, hln=%i)",
103  ntohs(ah->ar_hrd), ah->ar_hln);
104  BadARP(ah, errbuf);
105  return;
106  }
107  break;
108 
109  default:
110  { // don't know how to proceed
111  snprintf(errbuf, sizeof(errbuf),
112  "unknown-arp-hw-address (hrd=%i)", ntohs(ah->ar_hrd));
113  BadARP(ah, errbuf);
114  return;
115  }
116  }
117 
118  // ### Note, we don't support IPv6 addresses yet.
119  switch ( ntohs(ah->ar_pro) ) {
120  case ETHERTYPE_IP:
121  if ( ah->ar_pln != 4 )
122  { // don't know how to handle the opcode
123  snprintf(errbuf, sizeof(errbuf),
124  "corrupt-arp-header (pro=%i, pln=%i)",
125  ntohs(ah->ar_pro), ah->ar_pln);
126  BadARP(ah, errbuf);
127  return;
128  }
129  break;
130 
131  default:
132  { // don't know how to proceed
133  snprintf(errbuf, sizeof(errbuf),
134  "unknown-arp-proto-address (pro=%i)",
135  ntohs(ah->ar_pro));
136  BadARP(ah, errbuf);
137  return;
138  }
139  }
140 
141 
142  // Check MAC src address = ARP sender MAC address.
143  if ( memcmp(pkt->l2_src, ar_sha(ah), ah->ar_hln) )
144  {
145  BadARP(ah, "weird-arp-sha");
146  return;
147  }
148 
149  // Check the code is supported.
150  switch ( ntohs(ah->ar_op) ) {
151  case ARPOP_REQUEST:
152  RREvent(arp_request, pkt->l2_src, pkt->l2_dst,
153  ar_spa(ah), ar_sha(ah), ar_tpa(ah), ar_tha(ah));
154  break;
155 
156  case ARPOP_REPLY:
157  RREvent(arp_reply, pkt->l2_src, pkt->l2_dst,
158  ar_spa(ah), ar_sha(ah), ar_tpa(ah), ar_tha(ah));
159  break;
160 
161  case ARPOP_REVREQUEST:
162  case ARPOP_REVREPLY:
163  case ARPOP_INVREQUEST:
164  case ARPOP_INVREPLY:
165  { // don't know how to handle the opcode
166  snprintf(errbuf, sizeof(errbuf),
167  "unimplemented-arp-opcode (%i)", ntohs(ah->ar_op));
168  BadARP(ah, errbuf);
169  break;
170  }
171 
172  default:
173  { // invalid opcode
174  snprintf(errbuf, sizeof(errbuf),
175  "invalid-arp-opcode (opcode=%i)", ntohs(ah->ar_op));
176  BadARP(ah, errbuf);
177  return;
178  }
179  }
