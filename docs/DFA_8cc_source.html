44 void DFA_State::SymPartition(const EquivClass* ec)
45  {
46  // Partitioning is done by creating equivalence classes for those
47  // characters which have out-transitions from the given state. Thus
48  // we are really creating equivalence classes of equivalence classes.
49  meta_ec = new EquivClass(ec->NumClasses());
50 
51  assert(nfa_states);
52  for ( int i = 0; i < nfa_states->length(); ++i )
53  {
54  NFA_State* n = (*nfa_states)[i];
55  int sym = n->TransSym();
56 
57  if ( sym == SYM_EPSILON )
58  continue;
59 
60  if ( sym != SYM_CCL )
61  { // character transition
62  if ( ec->IsRep(sym) )
63  {
64  sym = ec->SymEquivClass(sym);
65  meta_ec->UniqueChar(sym);
66  }
67  continue;
68  }
69 
70  // Character class.
71  meta_ec->CCL_Use(n->TransCCL());
72  }
73 
74  meta_ec->BuildECs();
75  }

119 NFA_state_list* DFA_State::SymFollowSet(int ec_sym, const EquivClass* ec)
120  {
121  NFA_state_list* ns = new NFA_state_list;
122 
123  assert(nfa_states);
124 
125  for ( int i = 0; i < nfa_states->length(); ++i )
126  {
127  NFA_State* n = (*nfa_states)[i];
128 
129  if ( n->TransSym() == SYM_CCL )
130  { // it's a character class
131  CCL* ccl = n->TransCCL();
132  int_list* syms = ccl->Syms();
133 
134  if ( ccl->IsNegated() )
135  {
136  int j;
137  for ( j = 0; j < syms->length(); ++j )
138  {
139  // Loop through (sorted) negated
140  // character class, which has
141  // presumably already been converted
142  // over to equivalence classes.
143  if ( (*syms)[j] >= ec_sym )
144  break;
145  }
146 
147  if ( j >= syms->length() || (*syms)[j] > ec_sym )
148  // Didn't find ec_sym in ccl.
149  n->AddXtionsTo(ns);
150 
151  continue;
152  }
153 
154  for ( int j = 0; j < syms->length(); ++j )
155  {
156  if ( (*syms)[j] > ec_sym )
157  break;
158 
159  if ( (*syms)[j] == ec_sym )
160  {
161  n->AddXtionsTo(ns);
162  break;
163  }
164  }
165  }
166 
167  else if ( n->TransSym() == SYM_EPSILON )
168  { // do nothing
169  }
170 
171  else if ( ec->IsRep(n->TransSym()) )
172  {
173  if ( ec_sym == ec->SymEquivClass(n->TransSym()) )
174  n->AddXtionsTo(ns);
175  }
176  }
177 
178  ns->resize(0);
179  return ns;
180  }

203 void DFA_State::Dump(FILE* f, DFA_Machine* m)
204  {
205  if ( mark )
206  return;
207 
208  fprintf(f, "\nDFA state %d:", StateNum());
209 
210  if ( accept )
211  {
212  AcceptingSet::const_iterator it;
213 
214  for ( it = accept->begin(); it != accept->end(); ++it )
215  fprintf(f, "%s accept #%d", it == accept->begin() ? "" : ",", *it);
216  }

218  fprintf(f, "\n");
219 
220  int num_trans = 0;
221  for ( int sym = 0; sym < num_sym; ++sym )
222  {
223  DFA_State* s = xtions[sym];
224 
225  if ( ! s )
226  continue;
227 
228  // Look ahead for compression.
229  int i;
230  for ( i = sym + 1; i < num_sym; ++i )
231  if ( xtions[i] != s )
232  break;
233 
234  char xbuf[512];
235 
236  int r = m->Rep(sym);
237  if ( ! r )
238  r = '.';
239 
240  if ( i == sym + 1 )
241  sprintf(xbuf, "'%c'", r);
242  else
243  sprintf(xbuf, "'%c'-'%c'", r, m->Rep(i-1));
244 
245  if ( s == DFA_UNCOMPUTED_STATE_PTR )
246  fprintf(f, "%stransition on %s to <uncomputed>",
247  ++num_trans == 1 ? "\t" : "\n\t", xbuf);
248  else
249  fprintf(f, "%stransition on %s to state %d",
250  ++num_trans == 1 ? "\t" : "\n\t", xbuf,
251  s->StateNum());
252 
253  sym = i - 1;
254  }
255 
256  if ( num_trans > 0 )
257  fprintf(f, "\n");
258 
259  SetMark(this);
260 
261  for ( int sym = 0; sym < num_sym; ++sym )
262  {
263  DFA_State* s = xtions[sym];
264 
265  if ( s && s != DFA_UNCOMPUTED_STATE_PTR )
266  s->Dump(f, m);

311 DFA_State* DFA_State_Cache::Lookup(const NFA_state_list& nfas,
312  HashKey** hash)
313  {
314  // We assume that state ID's don't exceed 10 digits, plus
315  // we allow one more character for the delimiter.
316  u_char id_tag[nfas.length() * 11 + 1];
317  u_char* p = id_tag;
318 
319  for ( int i = 0; i < nfas.length(); ++i )
320  {
321  NFA_State* n = nfas[i];
322  if ( n->TransSym() != SYM_EPSILON || n->Accept() != NO_ACCEPT )
323  {
324  int id = n->ID();
325  do
326  {
327  *p++ = '0' + (char)(id % 10);
328  id /= 10;
329  }
330  while ( id > 0 );
331  *p++ = '&';
332  }
333  }
334 
335  *p++ = '\0';
336 
337  // We use the short MD5 instead of the full string for the
338  // HashKey because the data is copied into the key.
339  u_char digest[16];
340  MD5(id_tag, p - id_tag, digest);
341 
342  *hash = new HashKey(&digest, sizeof(digest));
343  CacheEntry* e = states.Lookup(*hash);
344  if ( ! e )
345  {
346  ++misses;
347  return 0;
348  }
349  ++hits;
350 
351  delete *hash;
352  *hash = 0;
353 
354  return e->state;
355  }
