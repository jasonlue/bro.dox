1 //
2 // DNP3 was initially used over serial links; it defined its own application
3 // layer, transport layer, and data link layer. This hierarchy cannot be
4 // mapped to the TCP/IP stack directly. As a result, all three DNP3 layers
5 // are packed together as a single application layer payload over the TCP
6 // layer. Each DNP3 packet in the application layer may look like this DNP3
7 // Packet:
8 //
9 // DNP3 Link Layer | DNP3 Transport Layer | DNP3 Application Layer
10 //
11 // (This hierarchy can be viewed in the Wireshark visually.)
12 //
13 // === Background on DNP3
14 //
15 // 1. Basic structure of DNP3 Protocol over serial links. This information
16 // can be found in detail in
17 //
18 // DNP3 Specification Volume 2, Part 1 Basic, Application Layer
19 // DNP3 Specification Volume 4, Data Link Layer
20 //
21 // Traditionally, the DNP3 Application Layer in serial links contains a
22 // "DNP3 Application Layer Fragment". The data that is parsed by the end
23 // device and then executed. As the "DNP3 Application Layer Fragment" can
24 // be long (>255 bytes), it may be trunkcated and carried in different
25 // DNP3 Application Layer of more than one DNP3 packets.
26 //
27 // So we may find a long DNP3 Application Layer Fragment to be transmitted in the following format
28 //
29 // DNP3 Packet #1 : DNP3 Link Layer | DNP3 Transport Layer | DNP3 Application Layer #1
30 // DNP3 Packet #2 : DNP3 Link Layer | DNP3 Transport Layer | DNP3 Application Layer #2
31 // ....
32 // DNP3 Packet #n : DNP3 Link Layer | DNP3 Transport Layer | DNP3 Application Layer #n
33 //
34 // So to get the whole DNP3 application layer fragment, we concatenate
35 // each DNP3 Application Layer Data into a logic DNP3 Application Layer
36 // Fragment:
37 //
38 // DNP3 Application Layer #1 + DNP3 Application Layer #2 + ... + DNP3 Application Layer #n
39 //
40 // 2. Packing DNP3 Network Packet into TCP/IP stack
41 //
42 // We will call the original DNP3 Link Layer, Transport Layer and Application
43 // Layer used in serial link as Pseudo Link Layer, Pseudo Transport Layer and
44 // Pseudo Application Layer.
45 //
46 // For a long DNP3 application layer fragment, we may find it tramistted
47 // over IP network in the following format:
48 //
49 // Network Packet #1 : TCP Header | DNP3 Pseudo Link Layer | DNP3 Pseudo Transport Layer | DNP3 Pseudo Application Layer #1
50 // Network Packet #2 : TCP Header | DNP3 Pseudo Link Layer | DNP3 Pseudo Transport Layer | DNP3 Pseudo Application Layer #2
51 // ....
52 // Network Packet #n : TCP Header | DNP3 Pseudo Link Layer | DNP3 Pseudo Transport Layer | DNP3 Pseudo Application Layer #n
53 //
54 // === Challenges of Writing DNP3 Analyzer on Binpac ===
55 //
56 // The detailed structure of the DNP3 Link Layer is:
57 //
58 // 0x05 0x64 Len Ctrl Dest_LSB Dest_MSB Src_LSB Src_MSB CRC_LSB CRC_MSB
59 //
60 // Each field is a byte; LSB: least significant byte; MSB: most significatn byte.
61 //
62 // "Len" indicates the length of the byte stream right after this field
63 // (excluding CRC fields) in the current DNP3 packet.
64 //
65 // Since "Len" is of size one byte, the largest length it can represent is
66 // 255 bytes. The larget DNP3 Application Layer size is "255 - 5 + size of
67 // all CRC fields". "minus 5" is coming from the 5 bytes after "Len" field in
68 // the DNP3 Link Layer, i.e. Ctrl Dest_LSB Dest_MSB Src_LSB Src_MSB Hence,
69 // the largest size of a DNP3 Packet (DNP3 Data Link Layer : DNP3 Transport
70 // Layer : DNP3 Application Layer) can only be 292 bytes.
71 //
72 // The "Len" field indicates the length of of a single chunk of DNP3 Psuedo
73 // Application Layer data instead of the whole DNP3 Application Layer
74 // Fragment. However, we can not know the whole length of the DNP3
75 // Application Layer Fragment (which Binpac would normally need) until all
76 // chunks of Pseudo Application Layer Data are received.
77 //
78 // We hence exploit the internal flow_buffer class used in Binpac to buffer
79 // the application layer data until all chunk are received, which does
80 // require a bit of internal knowledge of the generated code.
81 //
82 // The binpac analyzer parses the DNP3 Application Layer Fragment. However,
83 // we manually add the original Pseudo Link Layer data as an additional
84 // header before the DNP3 Application Fragment. This helps to know how many
85 // bytes are in the current chunk of DNP3 application layer data (not the
86 // whole Application Layer Fragment).
87 //
88 // Graphically, the procedure is:
89 //
90 // DNP3 Packet : DNP3 Pseudo Data Link Layer : DNP3 Pseudo Transport Layer : DNP3 Pseudo Application Layer
91 // || ||
92 // || (length field) || (original paylad byte stream)
93 // \/ \/
94 // DNP3 Additional Header : Reassembled DNP3 Pseudo Application Layer Data
95 // ||
96 // \/
97 // Binpac DNP3 Analyzer

132 bool DNP3_Base::ProcessData(int len, const u_char* data, bool orig)
133  {
134  Endpoint* endp = orig ? &orig_state : &resp_state;
135 
136  while ( len )
137  {
138  if ( endp->in_hdr )
139  {
140  // We're parsing the DNP3 header and link layer, get that in full.
141  int res = AddToBuffer(endp, PSEUDO_APP_LAYER_INDEX, &data, &len);
142 
143  if ( res == 0 )
144  return true;
145 
146  if ( res < 0 )
147  return false;
148 
149  // The first two bytes must always be 0x0564.
150  if( endp->buffer[0] != 0x05 || endp->buffer[1] != 0x64 )
151  {
152  analyzer->Weird("dnp3_header_lacks_magic");
153  return false;
154  }
155 
156  // Make sure header checksum is correct.
157  if ( ! CheckCRC(PSEUDO_LINK_LAYER_LEN, endp->buffer, endp->buffer + PSEUDO_LINK_LAYER_LEN, "header") )
158  {
159  analyzer->ProtocolViolation("broken_checksum");
160  return false;
161  }
162 
163  // If the checksum works out, we're pretty certainly DNP3.
164  analyzer->ProtocolConfirmation();
165 
166  // DNP3 packets without transport and application
167  // layers can happen, we ignore them.
168  if ( (endp->buffer[PSEUDO_LENGTH_INDEX] + 3) == (char)PSEUDO_LINK_LAYER_LEN )
169  {
170  ClearEndpointState(orig);
171  return true;
172  }
173 
174  // Double check the direction in case the first
175  // received packet is a response.
176  u_char ctrl = endp->buffer[PSEUDO_CONTROL_FIELD_INDEX];
177 
178  if ( orig != (bool)(ctrl & 0x80) )
179  analyzer->Weird("dnp3_unexpected_flow_direction");
180 
181  // Update state.
182  endp->pkt_length = endp->buffer[PSEUDO_LENGTH_INDEX];
183  endp->tpflags = endp->buffer[PSEUDO_TRANSPORT_INDEX];
184  endp->in_hdr = false; // Now parsing application layer.
185 
186  // For the first packet, we submit the header to
187  // BinPAC.
188  if ( ++endp->pkt_cnt == 1 )
189  interp->NewData(orig, endp->buffer, endp->buffer + PSEUDO_LINK_LAYER_LEN);
190  }
191 
192  if ( ! endp->in_hdr )
193  {
194  if ( endp->pkt_length <= 0 )
195  {
196  analyzer->Weird("dnp3_negative_or_zero_length_link_layer");
197  return false;
198  }
199 
200  // We're parsing the DNP3 application layer, get that
201  // in full now as well. We calculate the number of
202  // raw bytes the application layer consists of from
203  // the packet length by determining how much 16-byte
204  // chunks fit in there, and then add 2 bytes CRC for
205  // each.
206  int n = PSEUDO_APP_LAYER_INDEX + (endp->pkt_length - 5) + ((endp->pkt_length - 5) / 16) * 2
207  + 2 * ( ((endp->pkt_length - 5) % 16 == 0) ? 0 : 1) - 1 ;
208 
209  int res = AddToBuffer(endp, n, &data, &len);
210 
211  if ( res == 0 )
212  return true;
213 
214  if ( res < 0 )
215  return false;
216 
217  // Parse the the application layer data.
218  if ( ! ParseAppLayer(endp) )
219  return false;
220 
221  // Done with this packet, prepare for next.
222  endp->buffer_len = 0;
223  endp->in_hdr = true;
224  }
225  }
226 
227  return true;
228  }

230 int DNP3_Base::AddToBuffer(Endpoint* endp, int target_len, const u_char** data, int* len)
231  {
232  if ( ! target_len )
233  return 1;
234 
235  if ( *len < 0 )
236  {
237  reporter->AnalyzerError(analyzer, "dnp3 negative input length: %d", *len);
238  return -1;
239  }
240 
241  if ( target_len < endp->buffer_len )
242  {
243  reporter->AnalyzerError(analyzer, "dnp3 invalid target length: %d - %d",
244  target_len, endp->buffer_len);
245  return -1;
246  }
247 
248  int to_copy = min(*len, target_len - endp->buffer_len);
249 
250  if ( endp->buffer_len + to_copy > MAX_BUFFER_SIZE )
251  {
252  reporter->AnalyzerError(analyzer, "dnp3 buffer length exceeded: %d + %d",
253  endp->buffer_len, to_copy);
254  return -1;
255  }
256 
257  memcpy(endp->buffer + endp->buffer_len, *data, to_copy);
258  *data += to_copy;
259  *len -= to_copy;
260  endp->buffer_len += to_copy;
261 
262  if ( endp->buffer_len == target_len )
263  return 1;
264 
265  return 0;

268 bool DNP3_Base::ParseAppLayer(Endpoint* endp)
269  {
270  bool orig = (endp == &orig_state);
271  binpac::DNP3::DNP3_Flow* flow = orig ? interp->upflow() : interp->downflow();
272 
273  u_char* data = endp->buffer + PSEUDO_TRANSPORT_INDEX; // The transport layer byte counts as app-layer it seems.
274  int len = endp->pkt_length - 5;
275 
276  // DNP3 Packet : DNP3 Pseudo Link Layer | DNP3 Pseudo Transport Layer | DNP3 Pseudo Application Layer
277  // DNP3 Serial Transport Layer data is always 1 byte.
278  // Get FIN FIR seq field in transport header.
279  // FIR indicate whether the following DNP3 Serial Application Layer is first chunk of bytes or not.
280  // FIN indicate whether the following DNP3 Serial Application Layer is last chunk of bytes or not.
281 
282  int is_first = (endp->tpflags & 0x40) >> 6; // Initial chunk of data in this packet.
283  int is_last = (endp->tpflags & 0x80) >> 7; // Last chunk of data in this packet.
284 
285  int transport = PSEUDO_TRANSPORT_LEN;
286 
287  int i = 0;
288  while ( len > 0 )
289  {
290  int n = min(len, 16);
291 
292  // Make sure chunk has a correct checksum.
293  if ( ! CheckCRC(n, data, data + n, "app_chunk") )
294  return false;
295 
296  if ( data + n >= endp->buffer + endp->buffer_len )
297  {
298  reporter->AnalyzerError(analyzer,
299  "dnp3 app layer parsing overflow %d - %d",
300  endp->buffer_len, n);
301  return false;
302  }
303 
304  // Pass on to BinPAC.
305  flow->flow_buffer()->BufferData(data + transport, data + n);
306  transport = 0;
307 
308  data += n + 2;
309  len -= n;
310  }
311 
312  if ( is_first )
313  endp->encountered_first_chunk = true;
314 
315  if ( ! is_first && ! endp->encountered_first_chunk )
316  {
317  // We lost the first chunk.
318  analyzer->Weird("dnp3_first_application_layer_chunk_missing");
319  return false;
320  }
321 
322  if ( is_last )
323  {
324  flow->flow_buffer()->FinishBuffer();
325  flow->FlowEOF();
326  ClearEndpointState(orig);
327  }
328 
329  return true;
330  }

356 void DNP3_Base::PrecomputeCRCTable()
357  {
358  for( unsigned int i = 0; i < 256; i++)
359  {
360  unsigned int crc = i;
361 
362  for ( unsigned int j = 0; j < 8; ++j )
363  {
364  if ( crc & 0x0001 )
365  crc = (crc >> 1) ^ 0xA6BC; // Generating polynomial.
366  else
367  crc >>= 1;
368  }
369 
370  crc_table[i] = crc;
371  }

374 unsigned int DNP3_Base::CalcCRC(int len, const u_char* data)
375  {
376  unsigned int crc = 0x0000;
377 
378  for ( int i = 0; i < len; i++ )
379  {
380  unsigned int index = (crc ^ data[i]) & 0xFF;
381  crc = crc_table[index] ^ (crc >> 8);
382  }
383 
384  return ~crc & 0xFFFF;
