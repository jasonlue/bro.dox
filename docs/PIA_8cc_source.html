289  a->SetSignature(rule);
290 
291  // We have two cases here:
292  //
293  // (a) We have already got stream input.
294  // => Great, somebody's already reassembling and we can just
295  // replay our stream buffer to the new analyzer.
296  if ( stream_mode )
297  {
298  ReplayStreamBuffer(a);
299  return;
300  }
301 
302  // (b) We have only got packet input so far (or none at all).
303  // => We have to switch from packet-mode to stream-mode.
304  //
305  // Here's what we do:
306  //
307  // (1) We create new tcp::TCP_Reassemblers and feed them the buffered
308  // packets.
309  //
310  // (2) The reassembler will give us their results via the
311  // stream-interface and we buffer it as usual.
312  //
313  // (3) We replay the now-filled stream-buffer to the analyzer.
314  //
315  // (4) We hand the two reassemblers to the TCP Analyzer (our parent),
316  // turning reassembly now on for all subsequent data.
317 
318  DBG_LOG(DBG_ANALYZER, "PIA_TCP switching from packet-mode to stream-mode");

320

322  // if this is works in all cases...
323

327  // we have been inserted somewhere further down in the
328  // analyzer tree. In this case, we will never have seen
329  // any input at this point (because we don't get packets).
330  assert(!pkt_buffer.head);
331  assert(!stream_buffer.head);
332  return;
333  }
