128 Value* SQLite::EntryToVal(sqlite3_stmt *st, const threading::Field *field, int pos, int subpos)
129  {
130  if ( sqlite3_column_type(st, pos ) == SQLITE_NULL )
131  return new Value(field->type, field->subtype, false);
132 
133  Value* val = new Value(field->type, true);
134 
135  switch ( field->type ) {
136  case TYPE_ENUM:
137  case TYPE_STRING:
138  {
139  const char *text = (const char*) sqlite3_column_text(st, pos);
140  int length = sqlite3_column_bytes(st, pos);
141 
142  char *out = new char[length];

144

146  val->val.string_val.data = out;
147  break;
148  }
149 
150  case TYPE_BOOL:
151  {
152  if ( sqlite3_column_type(st, pos) != SQLITE_INTEGER )
153  {
154  Error("Invalid data type for boolean - expected Integer");
155  delete val;
156  return 0;
157  }
158

160 
161  if ( res == 0 || res == 1 )
162  val->val.int_val = res;
163  else
164  {
165  Error(Fmt("Invalid value for boolean: %d", res));
166  delete val;
167  return 0;
168  }
169  break;
170  }
171 
172  case TYPE_INT:
173  val->val.int_val = sqlite3_column_int64(st, pos);
174  break;
175 
176  case TYPE_DOUBLE:
177  case TYPE_TIME:
178  case TYPE_INTERVAL:
179  val->val.double_val = sqlite3_column_double(st, pos);
180  break;
181 
182  case TYPE_COUNT:
183  case TYPE_COUNTER:
184  val->val.uint_val = sqlite3_column_int64(st, pos);
185  break;
186 
187  case TYPE_PORT:
188  {
189  val->val.port_val.port = sqlite3_column_int(st, pos);
190  val->val.port_val.proto = TRANSPORT_UNKNOWN;
191  if ( subpos != -1 )
192  {
193  const char *text = (const char*) sqlite3_column_text(st, subpos);
194 
195  if ( text == 0 )
196  Error("Port protocol definition did not contain text");
197  else
198  {
199  string s(text, sqlite3_column_bytes(st, subpos));
200  val->val.port_val.proto = io->ParseProto(s);
201  }
202  }
203  break;
204  }
205 
206  case TYPE_SUBNET:
207  {
208  const char *text = (const char*) sqlite3_column_text(st, pos);
209  string s(text, sqlite3_column_bytes(st, pos));
210  int pos = s.find("/");
211  int width = atoi(s.substr(pos+1).c_str());
212  string addr = s.substr(0, pos);
213 
214  val->val.subnet_val.prefix = io->ParseAddr(addr);
215  val->val.subnet_val.length = width;
216  break;
217  }
218 
219  case TYPE_ADDR:

221  const char *text = (const char*) sqlite3_column_text(st, pos);
222  string s(text, sqlite3_column_bytes(st, pos));
223  val->val.addr_val = io->ParseAddr(s);

226

228  case TYPE_VECTOR:
229  {
