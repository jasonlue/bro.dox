102 void Packet::ProcessLayer2()
103  {
104  l2_valid = true;
105 
106  // Unfortunately some packets on the link might have MPLS labels
107  // while others don't. That means we need to ask the link-layer if
108  // labels are in place.
109  bool have_mpls = false;
110 
111  const u_char* pdata = data;
112  const u_char* end_of_data = data + cap_len;
113 
114  switch ( link_type ) {
115  case DLT_NULL:
116  {
117  int protocol = (pdata[3] << 24) + (pdata[2] << 16) + (pdata[1] << 8) + pdata[0];
118  pdata += GetLinkHeaderSize(link_type);
119 
120  // From the Wireshark Wiki: "AF_INET6, unfortunately, has
121  // different values in {NetBSD,OpenBSD,BSD/OS},
122  // {FreeBSD,DragonFlyBSD}, and {Darwin/Mac OS X}, so an IPv6
123  // packet might have a link-layer header with 24, 28, or 30
124  // as the AF_ value." As we may be reading traces captured on
125  // platforms other than what we're running on, we accept them
126  // all here.
127 
128  if ( protocol == AF_INET )
129  l3_proto = L3_IPV4;
130  else if ( protocol == 24 || protocol == 28 || protocol == 30 )
131  l3_proto = L3_IPV6;
132  else
133  {
134  Weird("non_ip_packet_in_null_transport");
135  return;
136  }
137 
138  break;
139  }
140 
141  case DLT_EN10MB:
142  {
143  // Skip past Cisco FabricPath to encapsulated ethernet frame.
144  if ( pdata[12] == 0x89 && pdata[13] == 0x03 )
145  {
146  auto constexpr cfplen = 16;
147 
148  if ( pdata + cfplen + GetLinkHeaderSize(link_type) >= end_of_data )
149  {
150  Weird("truncated_link_header_cfp");
151  return;
152  }
153 
154  pdata += cfplen;
155  }
156 
157  // Get protocol being carried from the ethernet frame.
158  int protocol = (pdata[12] << 8) + pdata[13];
159 
160  eth_type = protocol;
161  l2_dst = pdata;
162  l2_src = pdata + 6;
163 
164  pdata += GetLinkHeaderSize(link_type);
165 
166  bool saw_vlan = false;
167 
168  while ( protocol == 0x8100 || protocol == 0x9100 ||
169  protocol == 0x8864 )
170  {
171  switch ( protocol )
172  {
173  // VLAN carried over the ethernet frame.
174  // 802.1q / 802.1ad
175  case 0x8100:
176  case 0x9100:
177  {
178  if ( pdata + 4 >= end_of_data )
179  {
180  Weird("truncated_link_header");
181  return;
182  }
183 
184  auto& vlan_ref = saw_vlan ? inner_vlan : vlan;
185  vlan_ref = ((pdata[0] << 8) + pdata[1]) & 0xfff;
186  protocol = ((pdata[2] << 8) + pdata[3]);
187  pdata += 4; // Skip the vlan header
188  saw_vlan = true;
189  eth_type = protocol;
190  }
191  break;
192 
193  // PPPoE carried over the ethernet frame.
194  case 0x8864:
195  {
196  if ( pdata + 8 >= end_of_data )
197  {
198  Weird("truncated_link_header");
199  return;
200  }
201 
202  protocol = (pdata[6] << 8) + pdata[7];
203  pdata += 8; // Skip the PPPoE session and PPP header
204 
205  if ( protocol == 0x0021 )
206  l3_proto = L3_IPV4;
207  else if ( protocol == 0x0057 )
208  l3_proto = L3_IPV6;
209  else
210  {
211  // Neither IPv4 nor IPv6.
212  Weird("non_ip_packet_in_pppoe_encapsulation");
213  return;
214  }
215  }
216  break;
217  }
218  }
219 
220  // Check for MPLS in VLAN.
221  if ( protocol == 0x8847 )
222  have_mpls = true;
223 
224  // Normal path to determine Layer 3 protocol.
225  if ( ! have_mpls && l3_proto == L3_UNKNOWN )
226  {
227  if ( protocol == 0x800 )
228  l3_proto = L3_IPV4;
229  else if ( protocol == 0x86dd )
230  l3_proto = L3_IPV6;
231  else if ( protocol == 0x0806 || protocol == 0x8035 )
232  l3_proto = L3_ARP;
233  else
234  {
235  // Neither IPv4 nor IPv6.
236  Weird("non_ip_packet_in_ethernet");
237  return;
238  }
239  }
240 
241  break;
242  }
243 
244  case DLT_PPP_SERIAL:
245  {
246  // Get PPP protocol.
247  int protocol = (pdata[2] << 8) + pdata[3];
248  pdata += GetLinkHeaderSize(link_type);
249 
250  if ( protocol == 0x0281 )
251  {
252  // MPLS Unicast. Remove the pdata link layer and
253  // denote a header size of zero before the IP header.
254  have_mpls = true;
255  }
256  else if ( protocol == 0x0021 )
257  l3_proto = L3_IPV4;
258  else if ( protocol == 0x0057 )
259  l3_proto = L3_IPV6;
260  else
261  {
262  // Neither IPv4 nor IPv6.
263  Weird("non_ip_packet_in_ppp_encapsulation");
264  return;
265  }
266  break;
267  }
268 
269  case DLT_IEEE802_11_RADIO:
270  {
271  if ( pdata + 3 >= end_of_data )
272  {
273  Weird("truncated_radiotap_header");
274  return;
275  }
276 
277  // Skip over the RadioTap header
278  int rtheader_len = (pdata[3] << 8) + pdata[2];
279 
280  if ( pdata + rtheader_len >= end_of_data )
281  {
282  Weird("truncated_radiotap_header");
283  return;
284  }
285 
286  pdata += rtheader_len;
287  // fallthrough
288  }
289 
290  case DLT_IEEE802_11:
291  {
292  u_char len_80211 = 24; // minimal length of data frames
293 
294  if ( pdata + len_80211 >= end_of_data )
295  {
296  Weird("truncated_802_11_header");
297  return;
298  }
299 
300  u_char fc_80211 = pdata[0]; // Frame Control field
301 
302  // Skip non-data frame types (management & control).
303  if ( ! ((fc_80211 >> 2) & 0x02) )
304  return;
305 
306  // Skip subtypes without data.
307  if ( (fc_80211 >> 4) & 0x04 )
308  return;
309 
310  // 'To DS' and 'From DS' flags set indicate use of the 4th
311  // address field.
312  if ( (pdata[1] & 0x03) == 0x03 )
313  len_80211 += l2_addr_len;
314 
315  // Look for the QoS indicator bit.
316  if ( (fc_80211 >> 4) & 0x08 )
317  {
318  // Skip in case of A-MSDU subframes indicated by QoS
319  // control field.
320  if ( pdata[len_80211] & 0x80)
321  return;
322 
323  len_80211 += 2;
324  }
325 
326  if ( pdata + len_80211 >= end_of_data )
327  {
328  Weird("truncated_802_11_header");
329  return;
330  }
331 
332  // Determine link-layer addresses based
333  // on 'To DS' and 'From DS' flags
334  switch ( pdata[1] & 0x03 ) {
335  case 0x00:
336  l2_src = pdata + 10;
337  l2_dst = pdata + 4;
338  break;
339 
340  case 0x01:
341  l2_src = pdata + 10;
342  l2_dst = pdata + 16;
343  break;
344 
345  case 0x02:
346  l2_src = pdata + 16;
347  l2_dst = pdata + 4;
348  break;
349 
350  case 0x03:
351  l2_src = pdata + 24;
352  l2_dst = pdata + 16;
353  break;
354  }
355 
356  // skip 802.11 data header
357  pdata += len_80211;
358 
359  if ( pdata + 8 >= end_of_data )
360  {
361  Weird("truncated_802_11_header");
362  return;
363  }
364  // Check that the DSAP and SSAP are both SNAP and that the control
365  // field indicates that this is an unnumbered frame.
366  // The organization code (24bits) needs to also be zero to
367  // indicate that this is encapsulated ethernet.
368  if ( pdata[0] == 0xAA && pdata[1] == 0xAA && pdata[2] == 0x03 &&
369  pdata[3] == 0 && pdata[4] == 0 && pdata[5] == 0 )
370  {
371  pdata += 6;
372  }
373  else
374  {
375  // If this is a logical link control frame without the
376  // possibility of having a protocol we care about, we'll
377  // just skip it for now.
378  return;
379  }
380 
381  int protocol = (pdata[0] << 8) + pdata[1];
382  if ( protocol == 0x0800 )
383  l3_proto = L3_IPV4;
384  else if ( protocol == 0x86DD )
385  l3_proto = L3_IPV6;
386  else if ( protocol == 0x0806 || protocol == 0x8035 )
387  l3_proto = L3_ARP;
388  else
389  {
390  Weird("non_ip_packet_in_ieee802_11");
391  return;
392  }
393  pdata += 2;
394 
395  break;
396  }
397 
398  default:
399  {
400  // Assume we're pointing at IP. Just figure out which version.
401  pdata += GetLinkHeaderSize(link_type);
402  if ( pdata + sizeof(struct ip) >= end_of_data )
403  {
404  Weird("truncated_link_header");
405  return;
406  }
407 
408  const struct ip* ip = (const struct ip *)pdata;
409 
410  if ( ip->ip_v == 4 )
411  l3_proto = L3_IPV4;
412  else if ( ip->ip_v == 6 )
413  l3_proto = L3_IPV6;
414  else
415  {
416  // Neither IPv4 nor IPv6.
417  Weird("non_ip_packet");
418  return;
419  }
420 
421  break;
422  }
423  }
424 
425  if ( have_mpls )
426  {
427  // Skip the MPLS label stack.
428  bool end_of_stack = false;
429 
430  while ( ! end_of_stack )
431  {
432  if ( pdata + 4 >= end_of_data )
433  {
434  Weird("truncated_link_header");
435  return;
436  }
437 
438  end_of_stack = *(pdata + 2) & 0x01;
439  pdata += 4;
440  }
441 
442  // We assume that what remains is IP
443  if ( pdata + sizeof(struct ip) >= end_of_data )
444  {
445  Weird("no_ip_in_mpls_payload");
446  return;
447  }
448 
449  const struct ip* ip = (const struct ip *)pdata;
450 
451  if ( ip->ip_v == 4 )
452  l3_proto = L3_IPV4;
453  else if ( ip->ip_v == 6 )
454  l3_proto = L3_IPV6;
455  else
456  {
457  // Neither IPv4 nor IPv6.
458  Weird("no_ip_in_mpls_payload");
459  return;
460  }
461  }
462 
463  else if ( encap_hdr_size )
464  {
465  // Blanket encapsulation. We assume that what remains is IP.
466  if ( pdata + encap_hdr_size + sizeof(struct ip) >= end_of_data )
467  {
468  Weird("no_ip_left_after_encap");
469  return;
470  }
471 
472  pdata += encap_hdr_size;
473 
474  const struct ip* ip = (const struct ip *)pdata;
475 
476  if ( ip->ip_v == 4 )
477  l3_proto = L3_IPV4;
478  else if ( ip->ip_v == 6 )
479  l3_proto = L3_IPV6;
480  else
481  {
482  // Neither IPv4 nor IPv6.
483  Weird("no_ip_in_encap");
484  return;
485  }
486 
487  }
488

490  // (d) L3_UNKNOWN.

492  // Calculate how much header we've used up.
493  hdr_size = (pdata - data);

496 RecordVal* Packet::BuildPktHdrVal() const
497  {
498  RecordVal* pkt_hdr = new RecordVal(raw_pkt_hdr_type);
499  RecordVal* l2_hdr = new RecordVal(l2_hdr_type);
500 
501  int is_ethernet = (link_type == DLT_EN10MB) ? 1 : 0;
502 
503  int l3 = BifEnum::L3_UNKNOWN;
504 
505  if ( l3_proto == L3_IPV4 )
506  l3 = BifEnum::L3_IPV4;
507 
508  else if ( l3_proto == L3_IPV6 )
509  l3 = BifEnum::L3_IPV6;
510 
511  else if ( l3_proto == L3_ARP )
512  l3 = BifEnum::L3_ARP;
513 
514  // l2_hdr layout:
515  // encap: link_encap; ##< L2 link encapsulation
516  // len: count; ##< Total frame length on wire
517  // cap_len: count; ##< Captured length
518  // src: string &optional; ##< L2 source (if ethernet)
519  // dst: string &optional; ##< L2 destination (if ethernet)
520  // vlan: count &optional; ##< VLAN tag if any (and ethernet)
521  // inner_vlan: count &optional; ##< Inner VLAN tag if any (and ethernet)
522  // ethertype: count &optional; ##< If ethernet
523  // proto: layer3_proto; ##< L3 proto
524 
525  if ( is_ethernet )
526  {
527  // Ethernet header layout is:
528  // dst[6bytes] src[6bytes] ethertype[2bytes]...
529  l2_hdr->Assign(0, new EnumVal(BifEnum::LINK_ETHERNET, BifType::Enum::link_encap));
530  l2_hdr->Assign(3, FmtEUI48(data + 6)); // src
531  l2_hdr->Assign(4, FmtEUI48(data)); // dst
532 
533  if ( vlan )
534  l2_hdr->Assign(5, new Val(vlan, TYPE_COUNT));
535 
536  if ( inner_vlan )
537  l2_hdr->Assign(6, new Val(inner_vlan, TYPE_COUNT));
538 
539  l2_hdr->Assign(7, new Val(eth_type, TYPE_COUNT));
540 
541  if ( eth_type == ETHERTYPE_ARP || eth_type == ETHERTYPE_REVARP )
542  // We also identify ARP for L3 over ethernet
543  l3 = BifEnum::L3_ARP;
544  }
545  else
546  l2_hdr->Assign(0, new EnumVal(BifEnum::LINK_UNKNOWN, BifType::Enum::link_encap));
547 
548  l2_hdr->Assign(1, new Val(len, TYPE_COUNT));
549  l2_hdr->Assign(2, new Val(cap_len, TYPE_COUNT));
550 
551  l2_hdr->Assign(8, new EnumVal(l3, BifType::Enum::layer3_proto));
552 
553  pkt_hdr->Assign(0, l2_hdr);
554 
555  if ( l3_proto == L3_IPV4 )
556  {
557  IP_Hdr ip_hdr((const struct ip*)(data + hdr_size), false);
558  return ip_hdr.BuildPktHdrVal(pkt_hdr, 1);
559  }
560 
561  else if ( l3_proto == L3_IPV6 )
562  {
563  IP_Hdr ip6_hdr((const struct ip6_hdr*)(data + hdr_size), false, cap_len);
564  return ip6_hdr.BuildPktHdrVal(pkt_hdr, 1);
565  }
566 
567  else
568  return pkt_hdr;
569  }
