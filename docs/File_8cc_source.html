89 // Maximizes the number of open file descriptors and returns the number
90 // that we should use for the cache.
91 static int maximize_num_fds()
92  {
93  struct rlimit rl;
94  if ( getrlimit(RLIMIT_NOFILE, &rl) < 0 )
95  reporter->FatalError("maximize_num_fds(): getrlimit failed");
96 
97  if ( rl.rlim_max == RLIM_INFINITY )
98  {
99  // Don't try raising the current limit.
100  if ( rl.rlim_cur == RLIM_INFINITY )
101  // Let's not be too ambitious.
102  return MAX_FILE_CACHE_SIZE;
103  else
104  return rl.rlim_cur / 2;
105  }
106 
107  // See if we can raise the current to the maximum.
108  rl.rlim_cur = rl.rlim_max;
109 
110  if ( setrlimit(RLIMIT_NOFILE, &rl) < 0 )
111  reporter->FatalError("maximize_num_fds(): setrlimit failed");
112 
113  return rl.rlim_cur / 2;
114  }

857 BroFile* BroFile::Unserialize(UnserialInfo* info)
858  {
859  BroFile* file = (BroFile*) SerialObj::Unserialize(info, SER_BRO_FILE);
860 
861  if ( ! file )
862  return 0;
863 
864  if ( file->is_open )
865  return file;
866 
867  // If there is already an object for this file, return it.
868  if ( file->name )
869  {
870  for ( BroFile* f = head; f; f = f->next )
871  {
872  if ( f->name && streq(file->name, f->name) )
873  {
874  Unref(file);
875  Ref(f);
876  return f;
877  }
878  }
879  }
880 
881  // Otherwise, open, but don't clobber.
882  if ( ! file->Open(0, "a") )
883  {
884  info->s->Error(fmt("cannot open %s: %s",
885  file->name, strerror(errno)));
886  return 0;
887  }
888 
889  // Here comes a hack. This method will return a pointer to a newly
890  // instantiated file object. As soon as this pointer is Unref'ed, the
891  // file will be closed. That means that when we unserialize the same
892  // file next time, we will re-open it and thereby delete the first one,
893  // i.e., we will be keeping to delete what we've written just before.
894  //
895  // To avoid this loop, we do an extra Ref here, i.e., this file will
896  // *never* be closed anymore (as long the file cache does not overflow).
897  Ref(file);
898 
899  // We deliberately override log rotation attributes with our defaults.
900  file->rotate_interval = log_rotate_interval;
901  file->rotate_size = log_max_size;
902  file->InstallRotateTimer();
903  file->SetBuf(file->buffered);
904 
905  return file;
906  }
