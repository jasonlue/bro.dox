123 int TCP_Endpoint::ValidChecksum(const struct tcphdr* tp, int len) const
124  {
125  uint32 sum = checksum_base;
126  int tcp_len = tp->th_off * 4 + len;
127 
128  if ( len % 2 == 1 )
129  // Add in pad byte.
130  sum += htons(((const u_char*) tp)[tcp_len - 1] << 8);
131 
132  sum += htons((unsigned short) tcp_len); // fill out pseudo header
133  sum = ones_complement_checksum((void*) tp, tcp_len, sum);
134 
135  return sum == 0xffff;
136  }

166 uint64 TCP_Endpoint::Size() const
167  {
168  if ( prev_state == TCP_ENDPOINT_SYN_SENT && state == TCP_ENDPOINT_RESET &&
169  peer->state == TCP_ENDPOINT_INACTIVE && ! NoDataAcked() )
170  // This looks like a half-open connection was discovered and aborted.
171  // Sequence numbers could be misleading if used in context of data size
172  // and there was never a chance for this endpoint to send data anyway.
173  return 0;
174 
175  uint64 size;
176  uint64 last_seq_64 = ToFullSeqSpace(LastSeq(), SeqWraps());
177  uint64 ack_seq_64 = ToFullSeqSpace(AckSeq(), AckWraps());
178 
179  // Going straight to relative sequence numbers and comparing those might
180  // make more sense, but there's some cases (e.g. due to RSTs) where
181  // last_seq might not be initialized to a trustworthy value such that
182  // rel_seq > rel_ack, but last_seq_64 < start_seq, which is obviously wrong.
183  if ( last_seq_64 > ack_seq_64 )
184  size = last_seq_64 - StartSeqI64();
185  else
186  size = ack_seq_64 - StartSeqI64();
187 
188  // Don't include SYN octet in sequence space. For partial connections
189  // (no SYN seen), we're still careful to adjust start_seq as though
190  // there was an initial SYN octet, because if we don't then the
191  // packet reassembly code gets confused.
192  if ( size != 0 )
193  --size;
194 
195  if ( FIN_cnt > 0 && size != 0 )
196  --size; // don't include FIN octet.
197 
198  return size;
199  }
