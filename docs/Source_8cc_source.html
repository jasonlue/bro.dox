54  {
55  char errbuf[PCAP_ERRBUF_SIZE];
56  char tmp_errbuf[PCAP_ERRBUF_SIZE];
57 
58  // Determine interface if not specified.
59  if ( props.path.empty() )
60  props.path = pcap_lookupdev(tmp_errbuf);
61 
62  if ( props.path.empty() )
63  {
64  safe_snprintf(errbuf, sizeof(errbuf),
65  "pcap_lookupdev: %s", tmp_errbuf);
66  Error(errbuf);
67  return;
68  }
69 
70  // Determine network and netmask.
71  uint32 net;
72  if ( pcap_lookupnet(props.path.c_str(), &net, &props.netmask, tmp_errbuf) < 0 )
73  {
74  // ### The lookup can fail if no address is assigned to
75  // the interface; and libpcap doesn't have any useful notion
76  // of error codes, just error std::strings - how bogus - so we
77  // just kludge around the error :-(.
78  // sprintf(errbuf, "pcap_lookupnet %s", tmp_errbuf);
79  // return;
80  props.netmask = 0xffffff00;
81  }
82 
83 #ifdef PCAP_NETMASK_UNKNOWN
84  // Defined in libpcap >= 1.1.1
85  if ( props.netmask == PCAP_NETMASK_UNKNOWN )
86  props.netmask = PktSrc::NETMASK_UNKNOWN;
87 #endif
88 
89  pd = pcap_create(props.path.c_str(), errbuf);
90 
91  if ( ! pd )
92  {
93  PcapError("pcap_create");
94  return;
95  }
96 
97  if ( pcap_set_snaplen(pd, BifConst::Pcap::snaplen) )
98  {
99  PcapError("pcap_set_snaplen");
100  return;
101  }
102 
103  if ( pcap_set_promisc(pd, 1) )
104  {
105  PcapError("pcap_set_promisc");
106  return;
107  }
108 
109  // We use the smallest time-out possible to return almost immediately
110  // if no packets are available. (We can't use set_nonblocking() as
111  // it's broken on FreeBSD: even when select() indicates that we can
112  // read something, we may get nothing if the store buffer hasn't
113  // filled up yet.)
114  //
115  // TODO: The comment about FreeBSD is pretty old and may not apply
116  // anymore these days.
117  if ( pcap_set_timeout(pd, 1) )

120  return;
121  }
122 
123  if ( pcap_set_buffer_size(pd, BifConst::Pcap::bufsize * 1024 * 1024) )
124  {

126  return;
127  }
128

130  {
131  PcapError("pcap_activate");
132  return;
133  }
134 
135 #ifdef HAVE_LINUX
136  if ( pcap_setnonblock(pd, 1, tmp_errbuf) < 0 )
137  {
138  PcapError("pcap_setnonblock");
139  return;
140  }
141 #endif
142 
143 #ifdef HAVE_PCAP_INT_H
144  Info(fmt("pcap bufsize = %d\n", ((struct pcap *) pd)->bufsize));
145 #endif

187 bool PcapSource::ExtractNextPacket(Packet* pkt)
188  {
189  if ( ! pd )
190  return false;
191 
192  const u_char* data = pcap_next(pd, &current_hdr);
193 
194  if ( ! data )
195  {
196  // Source has gone dry. If it's a network interface, this just means
197  // it's timed out. If it's a file, though, then the file has been
198  // exhausted.
199  if ( ! props.is_live )
200  Close();
201 
202  return false;
203  }
204 
205  last_data = data;
206  pkt->Init(props.link_type, &current_hdr.ts, current_hdr.caplen, current_hdr.len, data);
207 
208  if ( current_hdr.len == 0 || current_hdr.caplen == 0 )
209  {
210  Weird("empty_pcap_header", pkt);
211  return false;
212  }
213 
214  last_hdr = current_hdr;
215  last_data = data;
216  ++stats.received;
217  stats.bytes_received += current_hdr.len;
218 
219  return true;
220  }
