27 int InterConnEndpoint::DataSent(double t, uint64 seq, int len, int caplen,
28  const u_char* data, const IP_Hdr* /* ip */,
29  const struct tcphdr* /* tp */)
30  {
31  if ( caplen < len )
32  len = caplen;
33 
34  if ( len <= 0 )
35  return 0;
36 
37  if ( endp->state == tcp::TCP_ENDPOINT_PARTIAL )
38  is_partial = 1;
39 
40  uint64 ack = endp->ToRelativeSeqSpace(endp->AckSeq(), endp->AckWraps());
41  uint64 top_seq = seq + len;
42 
43  if ( top_seq <= ack || top_seq <= max_top_seq )
44  // There is no new data in this packet
45  return 0;
46 
47  if ( seq < max_top_seq )
48  { // Only consider new data
49  int64 amount_seen = max_top_seq - seq;
50  seq += amount_seen;
51  data += amount_seen;
52  len -= amount_seen;
53  }
54 
55  if ( max_top_seq && seq > max_top_seq )
56  // We've got a pkt above a hole
57  num_pkts += EstimateGapPacketNum(seq - max_top_seq);
58 
59  ++num_pkts;
60  max_top_seq = top_seq;
61 
62  // Count the bytes.
63  num_bytes += len;
64 
65  int last_char = 0;
66  int offset = 0; // where we consider the latest line to have begun
67 
68  for ( int i = 0; i < len; ++i )
69  {
70  unsigned int c = data[i];
71 
72  if ( c == '\n' && last_char == '\r' )
73  {
74  // Compress CRLF to just one line termination.
75  last_char = c;
76  continue;
77  }
78 
79  if ( c == '\n' || c == '\r' )
80  {
81  ++num_lines;
82  if ( i - offset <= NORMAL_LINE_LENGTH )
83  ++num_normal_lines;
84  offset = i;
85  }
86 
87  else if ( c != 0 && c < 128 )
88  ++num_7bit_ascii;
89 
90  last_char = c;
91  }
92 
93  if ( IsPotentialKeystrokePacket(len) )
94  {
95  if ( keystroke_just_seen )
96  {
97  ++num_keystrokes_two_in_a_row;
98 
99  if ( IsNormalKeystrokeInterarrival(t - last_keystroke_time) )
100  ++num_normal_interarrivals;
101  }
102  else
103  keystroke_just_seen = 1;
104 
105  // Look for packets matching the SSH signature of
106  // being either 0 or 4 modulo 8.
107  switch ( len & 7 ) {
108  case 0:
109  if ( len >= 16 )
110  ++num_8k0_pkts;
111  break;
112 
113  case 4:
114  ++num_8k4_pkts;
115  break;
116  }
117 
118  last_keystroke_time = t;
119  }
120  else
121  keystroke_just_seen = 0;
122 
123  return 1;
124  }
