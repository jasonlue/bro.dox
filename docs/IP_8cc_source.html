449 void IPv6_Hdr_Chain::Init(const struct ip6_hdr* ip6, int total_len,
450  bool set_next, uint16 next)
451  {
452  length = 0;
453  uint8 current_type, next_type;
454  next_type = IPPROTO_IPV6;
455  const u_char* hdrs = (const u_char*) ip6;
456 
457  if ( total_len < (int)sizeof(struct ip6_hdr) )
458  {
459  reporter->InternalWarning("truncated IP header in IPv6_HdrChain::Init");
460  return;
461  }
462 
463  do
464  {
465  // We can't determine a given header's length if there's less than
466  // two bytes of data available (2nd byte of extension headers is length)
467  if ( total_len < 2 )
468  return;
469 
470  current_type = next_type;
471  IPv6_Hdr* p = new IPv6_Hdr(current_type, hdrs);
472 
473  next_type = p->NextHdr();
474  uint16 cur_len = p->Length();
475 
476  // If this header is truncated, don't add it to chain, don't go further.
477  if ( cur_len > total_len )
478  {
479  delete p;
480  return;
481  }
482 
483  if ( set_next && next_type == IPPROTO_FRAGMENT )
484  {
485  p->ChangeNext(next);
486  next_type = next;
487  }
488 
489  chain.push_back(p);
490 
491  // Check for routing headers and remember final destination address.
492  if ( current_type == IPPROTO_ROUTING )
493  ProcessRoutingHeader((const struct ip6_rthdr*) hdrs, cur_len);
494 
495 #ifdef ENABLE_MOBILE_IPV6
496  // Only Mobile IPv6 has a destination option we care about right now.
497  if ( current_type == IPPROTO_DSTOPTS )
498  ProcessDstOpts((const struct ip6_dest*) hdrs, cur_len);
499 #endif
500 
501  hdrs += cur_len;
502  length += cur_len;
503  total_len -= cur_len;
504 
505  } while ( current_type != IPPROTO_FRAGMENT &&
506  current_type != IPPROTO_ESP &&
507 #ifdef ENABLE_MOBILE_IPV6
508  current_type != IPPROTO_MOBILITY &&
509 #endif
510  isIPv6ExtHeader(next_type) );
511  }

513 void IPv6_Hdr_Chain::ProcessRoutingHeader(const struct ip6_rthdr* r, uint16 len)
514  {
515  if ( finalDst )
516  {
517  // RFC 2460 section 4.1 says Routing should occur at most once.
518  reporter->Weird(SrcAddr(), DstAddr(), "multiple_routing_headers");
519  return;
520  }
521 
522  // Last 16 bytes of header (for all known types) is the address we want.
523  const in6_addr* addr = (const in6_addr*)(((const u_char*)r) + len - 16);
524 
525  switch ( r->ip6r_type ) {
526  case 0: // Defined by RFC 2460, deprecated by RFC 5095
527  {
528  if ( r->ip6r_segleft > 0 && r->ip6r_len >= 2 )
529  {
530  if ( r->ip6r_len % 2 == 0 )
531  finalDst = new IPAddr(*addr);
532  else
533  reporter->Weird(SrcAddr(), DstAddr(), "odd_routing0_len");
534  }
535 
536  // Always raise a weird since this type is deprecated.
537  reporter->Weird(SrcAddr(), DstAddr(), "routing0_hdr");
538  }
539  break;
540 
541 #ifdef ENABLE_MOBILE_IPV6
542  case 2: // Defined by Mobile IPv6 RFC 6275.
543  {
544  if ( r->ip6r_segleft > 0 )
545  {
546  if ( r->ip6r_len == 2 )
547  finalDst = new IPAddr(*addr);
548  else
549  reporter->Weird(SrcAddr(), DstAddr(), "bad_routing2_len");
550  }
551  }
552  break;
553 #endif
554 
555  default:
556  reporter->Weird(fmt("unknown_routing_type_%d", r->ip6r_type));
557  break;
558  }
559  }
