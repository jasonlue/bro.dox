223 void* Dictionary::DoRemove(DictEntry* entry, hash_t h,
224  PList(DictEntry)* chain, int chain_offset)
225  {
226  void* entry_value = entry->value;
227 
228  chain->remove_nth(chain_offset);
229  if ( order )
230  order->remove(entry);
231 
232  // Adjust existing cookies.
233  loop_over_list(cookies, i)
234  {
235  IterCookie* c = cookies[i];
236 
237  // Is the affected bucket the current one?
238  if ( (unsigned int) c->bucket == h )
239  {
240  if ( c->offset > chain_offset )
241  --c->offset;
242 
243  // The only other important case here occurs when we
244  // are deleting the current entry which
245  // simultaniously happens to be the last one in this
246  // bucket. This means that we would have to move on
247  // to the next non-empty bucket. Fortunately,
248  // NextEntry() will do exactly the right thing in
249  // this case. :-)
250  }
251 
252  // This item may have been inserted during this iteration.
253  if ( (unsigned int) c->bucket > h )
254  c->inserted.remove(entry);

384  {

389

391  // If the new entry hashes to a bucket in the old table we
392  // haven't moved yet, we need to put it in the old table. If
393  // we didn't do it this way, we would sometimes have to
394  // search both tables which is probably more expensive.
395 
396  if ( ! tbl2 || h >= tbl_next_ind )
397  {

399  num_entries_ptr = &num_entries;
400  max_num_entries_ptr = &max_num_entries;
401  }
402  else
403  {
404  ttbl = tbl2;
405  h = new_entry->hash % num_buckets2;
406  num_entries_ptr = &num_entries2;
407  max_num_entries_ptr = &max_num_entries2;
408  }
409

411

413 
414  if ( chain )

417  {
418  DictEntry* entry = (*chain)[i];
419

421  entry->len == n &&
422  ! memcmp(entry->key, new_entry->key, n) )
423  {

425  entry->value = new_entry->value;
426  return old_value;
427  }
428  }
429  }
