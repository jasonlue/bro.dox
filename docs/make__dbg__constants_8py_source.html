1 # Build the DebugCmdConstants.h and DebugCmdInfoConstants.cc files from the
2 # DebugCmdInfoConstants.in file.
3 #
4 # We do this via a script rather than maintaining them directly because
5 # the struct is a little complicated, so has to be initialized from code,
6 # plus we want to make adding new constants somewhat less painful.
7 #
8 # The input filename should be supplied as an argument.
9 #
10 # DebugCmds are printed to DebugCmdConstants.h
11 # DebugCmdInfos are printed to DebugCmdInfoConstants.cc
12 #
13 # The input format is:
14 #
15 # cmd: [DebugCmd]
16 # names: [space delimited names of cmd]
17 # resume: ['true' or 'false': should execution resume after this command?]
18 # help: [some help text]
19 #
20 # Blank lines are skipped.
21 # Comments should start with // and should be on a line by themselves.
22

32  %(repeatable)s);

34  }
35 '''
