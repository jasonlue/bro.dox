48 IOSource* Manager::FindSoonest(double* ts)
49  {
50  // Remove sources which have gone dry. For simplicity, we only
51  // remove at most one each time.
52  for ( SourceList::iterator i = sources.begin();
53  i != sources.end(); ++i )
54  if ( ! (*i)->src->IsOpen() )
55  {
56  (*i)->src->Done();
57  delete *i;
58  sources.erase(i);
59  break;
60  }
61 
62  // Ideally, we would always call select on the fds to see which
63  // are ready, and return the soonest. Unfortunately, that'd mean
64  // one select-call per packet, which we can't afford in high-volume
65  // environments. Thus, we call select only every SELECT_FREQUENCY
66  // call (or if all sources report that they are dry).
67 
68  ++call_count;
69 
70  IOSource* soonest_src = 0;
71  double soonest_ts = 1e20;
72  double soonest_local_network_time = 1e20;
73  bool all_idle = true;
74 
75  // Find soonest source of those which tell us they have something to
76  // process.
77  for ( SourceList::iterator i = sources.begin(); i != sources.end(); ++i )
78  {
79  if ( ! (*i)->src->IsIdle() )
80  {
81  all_idle = false;
82  double local_network_time = 0;
83  double ts = (*i)->src->NextTimestamp(&local_network_time);
84  if ( ts >= 0 && ts < soonest_ts )
85  {
86  soonest_ts = ts;
87  soonest_src = (*i)->src;
88  soonest_local_network_time =
89  local_network_time ?
90  local_network_time : ts;
91  }
92  }
93  }
94 
95  // If we found one and aren't going to select this time,
96  // return it.
97  int maxx = 0;
98 
99  if ( soonest_src && (call_count % SELECT_FREQUENCY) != 0 )
100  goto finished;
101 
102  // Select on the join of all file descriptors.
103  fd_set fd_read, fd_write, fd_except;
104 
105  FD_ZERO(&fd_read);
106  FD_ZERO(&fd_write);
107  FD_ZERO(&fd_except);
108 
109  for ( SourceList::iterator i = sources.begin();
110  i != sources.end(); ++i )
111  {
112  Source* src = (*i);
113 
114  if ( ! src->src->IsIdle() )
115  // No need to select on sources which we know to
116  // be ready.
117  continue;
118 
119  src->Clear();
120  src->src->GetFds(&src->fd_read, &src->fd_write, &src->fd_except);
121  src->SetFds(&fd_read, &fd_write, &fd_except, &maxx);
122  }
123 
124  // We can't block indefinitely even when all sources are dry:
125  // we're doing some IOSource-independent stuff in the main loop,
126  // so we need to return from time to time. (Instead of no time-out
127  // at all, we use a very small one. This lets FreeBSD trigger a
128  // BPF buffer switch on the next read when the hold buffer is empty
129  // while the store buffer isn't filled yet.
130 
131  struct timeval timeout;
132 
133  if ( all_idle )
134  {
135  // Interesting: when all sources are dry, simply sleeping a
136  // bit *without* watching for any fd becoming ready may
137  // decrease CPU load. I guess that's because it allows
138  // the kernel's packet buffers to fill. - Robin
139  timeout.tv_sec = 0;
140  timeout.tv_usec = 20; // SELECT_TIMEOUT;
141  select(0, 0, 0, 0, &timeout);
142  }
143 
144  if ( ! maxx )
145  // No selectable fd at all.
146  goto finished;
147 
148  timeout.tv_sec = 0;
149  timeout.tv_usec = 0;
150 
151  if ( select(maxx + 1, &fd_read, &fd_write, &fd_except, &timeout) > 0 )
152  { // Find soonest.
153  for ( SourceList::iterator i = sources.begin();
154  i != sources.end(); ++i )
155  {
156  Source* src = (*i);
157 
158  if ( ! src->src->IsIdle() )
159  continue;
160 
161  if ( src->Ready(&fd_read, &fd_write, &fd_except) )
162  {
163  double local_network_time = 0;
164  double ts = src->src->NextTimestamp(&local_network_time);
165  if ( ts >= 0.0 && ts < soonest_ts )
166  {
167  soonest_ts = ts;
168  soonest_src = src->src;
169  soonest_local_network_time =
170  local_network_time ?
171  local_network_time : ts;
172  }
173  }
174  }
175  }
176 
177 finished:
178  *ts = soonest_local_network_time;
179  return soonest_src;
180  }
