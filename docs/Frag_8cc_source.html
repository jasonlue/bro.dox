74 void FragReassembler::AddFragment(double t, const IP_Hdr* ip, const u_char* pkt)
75  {
76  const struct ip* ip4 = ip->IP4_Hdr();
77 
78  if ( ip4 )
79  {
80  if ( ip4->ip_p != ((const struct ip*)proto_hdr)->ip_p ||
81  ip4->ip_hl != ((const struct ip*)proto_hdr)->ip_hl )
82  // || ip4->ip_tos != proto_hdr->ip_tos
83  // don't check TOS, there's at least one stack that actually
84  // uses different values, and it's hard to see an associated
85  // attack.
86  s->Weird("fragment_protocol_inconsistency", ip);
87  }
88  else
89  {
90  if ( ip->NextProto() != next_proto ||
91  ip->HdrLen() - 8 != proto_hdr_len )
92  s->Weird("fragment_protocol_inconsistency", ip);
93  // TODO: more detailed unfrag header consistency checks?
94  }
95 
96  if ( ip->DF() )
97  // Linux MTU discovery for UDP can do this, for example.
98  s->Weird("fragment_with_DF", ip);
99 
100  uint16 offset = ip->FragOffset();
101  uint32 len = ip->TotalLen();
102  uint16 hdr_len = ip->HdrLen();
103 
104  if ( len < hdr_len )
105  {
106  s->Weird("fragment_protocol_inconsistency", ip);
107  return;
108  }
109 
110  uint64 upper_seq = offset + len - hdr_len;
111 
112  if ( ! offset )
113  // Make sure to use the first fragment header's next field.
114  next_proto = ip->NextProto();
115 
116  if ( ! ip->MF() )
117  {
118  // Last fragment.
119  if ( frag_size == 0 )
120  frag_size = upper_seq;
121 
122  else if ( upper_seq != frag_size )
123  {
124  s->Weird("fragment_size_inconsistency", ip);
125 
126  if ( upper_seq > frag_size )
127  frag_size = upper_seq;
128  }
129  }
130 
131  else if ( len < MIN_ACCEPTABLE_FRAG_SIZE )
132  s->Weird("excessively_small_fragment", ip);
133 
134  if ( upper_seq > MAX_ACCEPTABLE_FRAG_SIZE )
135  s->Weird("excessively_large_fragment", ip);
136 
137  if ( frag_size && upper_seq > frag_size )
138  {
139  // This can happen if we receive a fragment that's *not*
140  // the last fragment, but still imputes a size that's
141  // larger than the size we derived from a previously-seen
142  // "last fragment".
143 
144  s->Weird("fragment_size_inconsistency", ip);
145  frag_size = upper_seq;
146  }
147 
148  // Do we need to check for consistent options? That's tricky
149  // for things like LSRR that get modified in route.
150 
151  // Remove header.
152  pkt += hdr_len;
153  len -= hdr_len;
154 
155  NewBlock(network_time, offset, len, pkt);
156  }

189 void FragReassembler::BlockInserted(DataBlock* /* start_block */)
190  {
191  if ( blocks->seq > 0 || ! frag_size )
192  // For sure don't have it all yet.
193  return;
194 
195  // We might have it all - look for contiguous all the way.
196  DataBlock* b;
197  for ( b = blocks; b->next; b = b->next )
198  if ( b->upper != b->next->seq )
199  break;
200 
201  if ( b->next )
202  {
203  // We have a hole.
204  if ( b->upper >= frag_size )
205  {
206  // We're stuck. The point where we stopped is
207  // contiguous up through the expected end of
208  // the fragment, but there's more stuff still
209  // beyond it, which is not contiguous. This
210  // can happen for benign reasons when we're
211  // intermingling parts of two fragmented packets.
212  Weird("fragment_size_inconsistency");
213 
214  // We decide to analyze the contiguous portion now.
215  // Extend the fragment up through the end of what
216  // we have.
217  frag_size = b->upper;
218  }
219  else
220  return;
221  }
222 
223  else if ( last_block->upper > frag_size )
224  {
225  Weird("fragment_size_inconsistency");
226  frag_size = last_block->upper;
227  }
228 
229  else if ( last_block->upper < frag_size )
230  // Missing the tail.
231  return;
232 
233  // We have it all. Compute the expected size of the fragment.
234  uint64 n = proto_hdr_len + frag_size;
235 
236  // It's possible that we have blocks associated with this fragment
237  // that exceed this size, if we saw MF fragments (which don't lead
238  // to us setting frag_size) that went beyond the size indicated by
239  // the final, non-MF fragment. This can happen for benign reasons
240  // due to intermingling of fragments from an older datagram with those
241  // for a more recent one.
242 
243  u_char* pkt = new u_char[n];
244  memcpy((void*) pkt, (const void*) proto_hdr, proto_hdr_len);
245 
246  u_char* pkt_start = pkt;
247 
248  pkt += proto_hdr_len;
249 
250  for ( b = blocks; b; b = b->next )
251  {
252  // If we're above a hole, stop. This can happen because
253  // the logic above regarding a hole that's above the
254  // expected fragment size.
255  if ( b->prev && b->prev->upper < b->seq )
256  break;
257 
258  if ( b->upper > n )
259  {
260  reporter->InternalWarning("bad fragment reassembly");
261  DeleteTimer();
262  Expire(network_time);
263  delete [] pkt_start;
264  return;
265  }
266 
267  memcpy((void*) &pkt[b->seq], (const void*) b->block,
268  b->upper - b->seq);
269  }
270 
271  delete reassembled_pkt;
272  reassembled_pkt = 0;
273 
274  unsigned int version = ((const struct ip*)pkt_start)->ip_v;
275 
276  if ( version == 4 )
277  {
278  struct ip* reassem4 = (struct ip*) pkt_start;
279  reassem4->ip_len = htons(frag_size + proto_hdr_len);
280  reassembled_pkt = new IP_Hdr(reassem4, true);
281  DeleteTimer();
282  }
283 
284  else if ( version == 6 )
285  {
286  struct ip6_hdr* reassem6 = (struct ip6_hdr*) pkt_start;
287  reassem6->ip6_plen = htons(frag_size + proto_hdr_len - 40);
288  const IPv6_Hdr_Chain* chain = new IPv6_Hdr_Chain(reassem6, next_proto, n);
289  reassembled_pkt = new IP_Hdr(reassem6, true, n, chain);
290  DeleteTimer();
291  }
292 
293  else
294  {
295  reporter->InternalWarning("bad IP version in fragment reassembly: %d",
296  version);
297  delete [] pkt_start;
298  }
299  }
