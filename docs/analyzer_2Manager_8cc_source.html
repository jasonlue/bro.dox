384  DBG_ANALYZER(conn, "activated ICMP analyzer");

386  }

390  return false;
391  }
392 
393  bool scheduled = ApplyScheduledAnalyzers(conn, false, root);
394

396  // other potential analyzers as well? For now we only take
397  // the scheduled ones.
398  if ( ! scheduled )
399  { // Let's see if it's a port we know.
400  if ( check_port && ! dpd_ignore_ports )
401  {

408  {

410

413

415  DBG_ANALYZER_ARGS(conn, "activated %s analyzer due to port %d",

417  }
418  }
419  }
420  }
421 
422  if ( tcp )
423  {
424  // We have to decide whether to reassamble the stream.
425  // We turn it on right away if we already have an app-layer

427  // asks us to do so. In all other cases, reassembly may
428  // be turned on later by the TCP PIA.
429 
430  bool reass = root->GetChildren().size() ||
431  dpd_reassemble_first_packets ||
432  tcp_content_deliver_all_orig ||
433  tcp_content_deliver_all_resp;
434

484  // Add ConnSize analyzer. Needs to see packets, not stream.
485  tcp->AddChildPacketAnalyzer(new conn_size::ConnSize_Analyzer(conn));
486  }
487

492  root->AddChildAnalyzer(new conn_size::ConnSize_Analyzer(conn));
493  }
