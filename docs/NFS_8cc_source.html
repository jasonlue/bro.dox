761  int&n, BifEnum::NFS3::status_t status)
762  {
763  RecordVal *rep = new RecordVal(BifType::Record::NFS3::readdir_reply_t);
764 
765  rep->Assign(0, new Val(isplus, TYPE_BOOL));
766 
767  if ( status == BifEnum::NFS3::NFS3ERR_OK )
768  {
769  unsigned pos;
770  VectorVal *entries = new VectorVal(BifType::Vector::NFS3::direntry_vec_t);
771 
772  rep->Assign(1, nfs3_post_op_attr(buf,n)); // dir_attr
773  rep->Assign(2, ExtractUint64(buf,n)); // cookieverf
774 
775  pos = 1;
776 
777  while ( extract_XDR_uint32(buf,n) )
778  {
779  RecordVal *entry = new RecordVal(BifType::Record::NFS3::direntry_t);
780  entry->Assign(0, ExtractUint64(buf,n)); // fileid
781  entry->Assign(1, nfs3_filename(buf,n)); // fname
782  entry->Assign(2, ExtractUint64(buf,n)); // cookie
783 
784  if ( isplus )
785  {
786  entry->Assign(3, nfs3_post_op_attr(buf,n));
787  entry->Assign(4, nfs3_post_op_fh(buf,n));
788  }
789 
790  entries->Assign(pos, entry);
791  pos++;
792  }
793 
794  rep->Assign(3, entries);
795  rep->Assign(4, ExtractBool(buf,n)); // eof
796  }
797  else
798  {
799  rep->Assign(1, nfs3_post_op_attr(buf,n));
