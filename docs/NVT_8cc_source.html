315 char* TelnetEnvironmentOption::ExtractEnv(u_char*& data, int& len, int& code)
316  {
317  code = data[0];
318 
319  if ( code != ENVIRON_VAR && code != ENVIRON_VAL &&
320  code != ENVIRON_USERVAR )
321  return 0;
322 
323  // Move past code.
324  --len;
325  ++data;
326 
327  // Find the end of this piece of the option.
328  u_char* data_end = data + len;
329  u_char* d;
330  for ( d = data; d < data_end; ++d )
331  {
332  if ( *d == ENVIRON_VAR || *d == ENVIRON_VAL || *d == ENVIRON_USERVAR )
333  break;
334 
335  if ( *d == ENVIRON_ESC )
336  {
337  ++d; // move past ESC
338  if ( d >= data_end )
339  return 0;
340  break;
341  }
342  }
343 
344  int size = d - data;
345  char* env = new char[size+1];
346 
347  // Now copy into env.
348  int d_ind = 0;
349  int i;
350  for ( i = 0; i < size; ++i )
351  {
352  if ( data[d_ind] == ENVIRON_ESC )
353  ++d_ind;
354 
355  env[i] = data[d_ind];
356  ++d_ind;
357  }
358 
359  env[i] = '\0';
360 
361  data = d;
362  len -= size;
363

482 void NVT_Analyzer::DoDeliver(int len, const u_char* data)
483  {
484  // This code is very similar to that for TCP_ContentLine. We
485  // don't virtualize out the differences because some of them
486  // would require per-character function calls, too expensive.
487  if ( pending_IAC )
488  {
489  ScanOption(seq, len, data);
490  return;
491  }
492 
493  // Add data up to IAC or end.
494  for ( ; len > 0; --len, ++data )
495  {
496  if ( offset >= buf_len )
497  InitBuffer(buf_len * 2);
498 
499  int c = data[0];
500 
501  if ( binary_mode && c != TELNET_IAC )
502  c &= 0x7f;
503 
504 #define EMIT_LINE \
505  { \
506  buf[offset] = '\0'; \
507  ForwardStream(offset, buf, IsOrig()); \
508  offset = 0; \
509  }
510 
511  switch ( c ) {
512  case '\r':
513  if ( CRLFAsEOL() & CR_as_EOL )
514  EMIT_LINE
515  else
516  buf[offset++] = c;
517  break;
518 
519  case '\n':
520  if ( last_char == '\r' )
521  {
522  if ( CRLFAsEOL() & CR_as_EOL )
523  // we already emited, skip
524  ;
525  else
526  {
527  --offset; // remove '\r'
528  EMIT_LINE
529  }
530  }
531 
532  else if ( CRLFAsEOL() & LF_as_EOL )
533  EMIT_LINE
534 
535  else
536  {
537  if ( Conn()->FlagEvent(SINGULAR_LF) )
538  Conn()->Weird("line_terminated_with_single_LF");
539  buf[offset++] = c;
540  }
541  break;
542 
543  case '\0':
544  if ( last_char == '\r' )
545  // Allow a NUL just after a \r - Solaris
546  // Telnet servers generate these, and they
547  // appear harmless.
548  ;
549 
550  else if ( flag_NULs )
551  CheckNUL();
552 
553  else
554  buf[offset++] = c;
555  break;
556 
557  case TELNET_IAC:
558  pending_IAC = 1;
559  IAC_pos = offset;
560  is_suboption = 0;
561  buf[offset++] = c;
562  ScanOption(seq, len - 1, data + 1);
563  return;
564 
565  default:
566  buf[offset++] = c;
567  break;
568  }
569 
570  if ( ! (CRLFAsEOL() & CR_as_EOL) &&
571  last_char == '\r' && c != '\n' && c != '\0' )
572  {
573  if ( Conn()->FlagEvent(SINGULAR_CR) )
574  Weird("line_terminated_with_single_CR");
575  }
576 
577  last_char = c;
578  }

581 void NVT_Analyzer::ScanOption(int seq, int len, const u_char* data)
582  {
583  if ( len <= 0 )
584  return;
585 
586  if ( IAC_pos == offset - 1 )
587  { // All we've seen so far is the IAC.
588  unsigned int code = data[0];
589 
590  if ( code == TELNET_IAC )
591  {
592  // An escaped 255, throw away the second
593  // instance and drop the IAC state.
594  pending_IAC = 0;
595  last_char = code;
596  }
597 
598  else if ( code == TELNET_OPT_SB )
599  {
600  is_suboption = 1;
601  last_was_IAC = 0;
602  buf[offset++] = code;
603  }
604 
605  else if ( IS_3_BYTE_OPTION(code) )
606  {
607  is_suboption = 0;
608  buf[offset++] = code;
609  }
610 
611  else
612  {
613  // We've got the whole 2-byte option.
614  SawOption(code);
615 
616  // Throw it and the IAC away.
617  --offset;
618  pending_IAC = 0;
619  }
620 
621  // Recurse to munch on the remainder.
622  DeliverStream(len - 1, data + 1, IsOrig());
623  return;
624  }
625 
626  if ( ! is_suboption )
627  {
628  // We now have the full 3-byte option.
629  SawOption(u_char(buf[offset-1]), data[0]);
630 
631  // Delete the option.
632  offset -= 2; // code + IAC
633  pending_IAC = 0;
634 
635  DeliverStream(len - 1, data + 1, IsOrig());
636  return;
637  }
638 
639  // A suboption. Spin looking for end.
640  for ( ; len > 0; --len, ++data )
641  {
642  unsigned int code = data[0];
643 
644  if ( last_was_IAC )
645  {
646  last_was_IAC = 0;
647 
648  if ( code == TELNET_IAC )
649  {
650  // This is an escaped IAC, eat
651  // the second copy.
652  continue;
653  }
654 
655  if ( code != TELNET_OPT_SE )
656  // BSD Telnet treats this case as terminating
657  // the suboption, so that's what we do here
658  // too. Below we make sure to munch on the
659  // new IAC.
660  BadOptionTermination(code);
661 
662  int opt_start = IAC_pos + 2;
663  int opt_stop = offset - 1;
664  int opt_len = opt_stop - opt_start;
665  SawSubOption((const char*)&buf[opt_start], opt_len);
666 
667  // Delete suboption.
668  offset = IAC_pos;
669  pending_IAC = is_suboption = 0;
670 
671  if ( code == TELNET_OPT_SE )
672  DeliverStream(len - 1, data + 1, IsOrig());
673  else
674  {
675  // Munch on the new (broken) option.
676  pending_IAC = 1;
677  IAC_pos = offset;
678  buf[offset++] = TELNET_IAC;
679  DeliverStream(len, data, IsOrig());
680  }
681  return;
682  }
683 
684  else
685  {
686  buf[offset++] = code;
687  last_was_IAC = (code == TELNET_IAC);
688  }
689  }
690  }
