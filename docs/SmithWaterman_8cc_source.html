198 // A node in Smith-Waterman's dynamic programming matrix. Each node
199 // contains the byte it represents in the case of a match, the score
200 // at this point, and a pointer to the previous cell. Previous means
201 // one up and left in case of a match, or a jump somewhere above and
202 // left in case of a gap.
203 //
204 struct SWNode {
205  // ID field for the cell, for debugging purposes.
206  int id;
207 
208  u_char swn_byte;
209  bool swn_byte_assigned;
210  bool swn_visited;
211 
212  // The score in this cell. The cell with the globally best score
213  // marks the end of the alignment.
214  int swn_score;
215 
216  // Pointer to previous match, walking back yields subsequence.
217  SWNode* swn_prev;
218 };

275 static void sw_collect_single(BroSubstring::Vec* result, SWNodeMatrix& matrix,
276  SWNode* node, SWParams& params)
277  {
278  string substring("");
279  int row = 0, col = 0;
280 
281  while ( node )
282  {
283 // printf("NODE: %i\n", node->id);
284  node->swn_visited = true;
285 
286  // Once we hit a gap, terminate the string and prepend
287  // it to our result vector, IF it has at least the length
288  // requested through the params._min_toklen parameter.
289  //
290  if ( node->swn_byte_assigned )
291  {
292  matrix.GetNodeIndices(node, row, col);
293  substring += node->swn_byte;
294 // printf("SUBSTRING: %s\n", substring.c_str());
295  }
296  else
297  {
298 // printf("GAP\n");
299  if ( substring.size() >= params._min_toklen )
300  {
301  reverse(substring.begin(), substring.end());
302  BroSubstring* bst = new BroSubstring(substring);
303  bst->AddAlignment(matrix.GetRowsString(), row-1);
304  bst->AddAlignment(matrix.GetColsString(), col-1);
305  result->push_back(bst);
306  }
307 
308  substring = "";
309  }
310 
311  node = node->swn_prev;
312  }
313 
314  // Anything left over now is the first string of an alignment and is
315  // manually added and marked as the beginning of a new alignment.
316  //
317  if ( substring.size() > 0 )
318  {
319  reverse(substring.begin(), substring.end());
320  BroSubstring* bst = new BroSubstring(substring);
321  bst->AddAlignment(matrix.GetRowsString(), row-1);
322  bst->AddAlignment(matrix.GetColsString(), col-1);
323  result->push_back(bst);
324  }
325 
326  if ( result->size() > 0 )
327  result->back()->MarkNewAlignment(true);
328  }

330 // Returns repeated common-subsequence alignments.
331 // @result: vector holding results on return.
332 // @matrix: SW matrix.
333 // @params: SW parameters.
334 //
335 // The approach taken is to essentially follow back from all starting points of
336 // common subsequences while tracking which nodes were visited earlier and which
337 // substrings are redundant (i.e., fully covered by a larger common substring).
338 //
339 static void sw_collect_multiple(BroSubstring::Vec* result,
340  SWNodeMatrix& matrix, SWParams& params)
341  {
342  vector<BroSubstring::Vec*> als;
343 
344  for ( int i = matrix.GetHeight() - 1; i > 0; --i )
345  {
346  for ( int j = matrix.GetWidth() - 1; j > 0; --j )
347  {
348  SWNode* node = matrix(i, j);
349 
350  if ( ! (node->swn_byte_assigned && ! node->swn_visited) )
351  continue;
352 
353  BroSubstring::Vec* new_al = new BroSubstring::Vec();
354  sw_collect_single(new_al, matrix, node, params);
355 
356  for ( vector<BroSubstring::Vec*>::iterator it = als.begin();
357  it != als.end(); ++it )
358  {
359  BroSubstring::Vec* old_al = *it;
360 
361  if ( old_al == 0 )
362  continue;
363 
364  for ( BroSubstring::VecIt it2 = old_al->begin();
365  it2 != old_al->end(); ++it2 )
366  {
367  for ( BroSubstring::VecIt it3 = new_al->begin();
368  it3 != new_al->end(); ++it3 )
369  {
370  if ( (*it2)->DoesCover(*it3) )
371  {
372  delete_each(new_al);
373  delete new_al;
374  new_al = 0;
375  goto end_loop;
376  }
377 
378  if ( (*it3)->DoesCover(*it2) )
379  {
380  delete_each(old_al);
381  delete old_al;
382  *it = 0;
383  goto end_loop;
384  }
385  }
386  }
387  }
388 
389 end_loop:
390  if ( new_al )
391  als.push_back(new_al);
392  }
393  }
394 
395  for ( vector<BroSubstring::Vec*>::iterator it = als.begin();
396  it != als.end(); ++it )
397  {
398  BroSubstring::Vec* al = *it;
399 
400  if ( al == 0 )
401  continue;
402 
403  for ( BroSubstring::VecIt it2 = al->begin();
404  it2 != al->end(); ++it2 )
405  result->push_back(*it2);
406 
407  delete al;
408  }

411 // The main Smith-Waterman algorithm.
412 //
413 BroSubstring::Vec* smith_waterman(const BroString* s1, const BroString* s2,
414  SWParams& params)
415  {
416  BroSubstring::Vec* result = new BroSubstring::Vec();
417 
418  if ( ! s1 || s1->Len() < int(params._min_toklen) ||
419  ! s2 || s2->Len() < int(params._min_toklen) )
420  return result;
421 
422  // Length of both strings, plus one because SW needs
423  // an extra row and column.
424  //
425  int i, len1 = s1->Len() + 1;
426  int j, len2 = s2->Len() + 1;
427 
428  int row = 0, col = 0;
429 
430  byte_vec string1 = s1->Bytes();
431  byte_vec string2 = s2->Bytes();
432 
433  SWNodeMatrix matrix(s1, s2); // dynamic programming matrix.
434  SWNode* node_max = 0; // pointer to the best score's node
435  SWNode* node_br_max = 0; // pointer to lowest-right matching node
436 
437  // The highest score in the matrix, globally. We initialize to 1
438  // because we are only interested in real scores (initializing to
439  // -infty would mean 0 is larger, and would complicate the link
440  // structure in the matrix).
441  //
442  int matrix_max = 1;
443  int br_max_r = 0;
444  int br_max_b = 0;
445 
446 
447  // Matrix initialization ----------------------------------------------
448 
449  // Assign IDs to each cell -- this is only for debugging purposes
450  // and can go later.
451 
452  int counter = 1;
453 
454  for ( i = 1; i < len1; ++i )
455  for ( j = 1; j < len2; ++j )
456  matrix(i, j)->id = counter++;
457 
458  // Subsequence calculation --------------------------------------------
459 
460  for ( i = 1; i < len1; ++i )
461  {
462  for ( j = 1; j < len2; ++j )
463  {
464  // Current node, top/left neighbours.
465  //
466  SWNode* current = matrix(i, j);
467  SWNode* node_tl = matrix(i-1, j-1);
468  SWNode* node_l = matrix(i, j-1);
469  SWNode* node_t = matrix(i-1, j);
470 
471  // Scores of neighbouring nodes.
472  //
473  int score_t = node_t->swn_score;
474  int score_l = node_l->swn_score;
475  int score_tl = node_tl->swn_score;
476 
477  // If strings at current indices match, assign new
478  // score to current node. Minus-one adjustments
479  // are necessary since matrix has one extra
480  // row + column.
481  //
482  if ( string1[i-1] == string2[j-1] )
483  {
484  // We have a match: improve previous score.
485  //
486  score_tl += 1;
487 
488  // If we're continuing a chain of matches, rate
489  // higher. This favours longer consecutive
490  // substrings.
491  //
492  if ( node_tl->swn_byte_assigned )
493  score_tl += 99;
494 
495  // Store the byte we've matched in the node for
496  // easier access.
497  //
498  current->swn_byte = string1[i-1];
499  current->swn_byte_assigned = true;
500  }
501 
502  // Pick the score among the neighbours that is now highest.
503  // This is the core of Smith-Waterman.
504  //
505  if ( current->swn_byte_assigned )
506  current->swn_score = score_tl;
507  else
508  current->swn_score = max(max(score_t, score_l), score_tl);
509 
510  // Establish predecessor chain according to neighbor
511  // with best score.
512  //
513  if ( current->swn_score == score_tl &&
514  current->swn_byte_assigned )
515  {
516  // If we had matched bytes (*and* it's the
517  // best neighbor), marke the node accordingly
518  //
519  if ( i >= br_max_b && j >= br_max_r )
520  {
521  node_br_max = current;
522  br_max_b = i;
523  br_max_r = j;
524  }
525 
526  current->swn_prev = node_tl;
527  }
528  else if ( current->swn_score == score_t )
529  current->swn_prev = node_t;
530  else
531  current->swn_prev = node_l;
532 
533  // Check if we have a new global maximum -- we
534  // specifically track the node that is the global
535  // maximum so we now from where to backtrack at
536  // the end of the matrix iteration.
537  //
538  if ( current->swn_score > matrix_max )
539  {
540  node_max = current;
541  matrix_max = current->swn_score;
542  }
543 
544 #if 0
545  printf("%4i/%.5i%c/%.5i[%c%c] ",
546  current->swn_score,
547  current->id,
548  current->swn_byte_assigned ? '*' : ' ',
549  current->swn_prev ? current->swn_prev->id : 0,
550  string1[i-1], string2[j-1]);
551 #endif
552  //printf("%.5i ", current->swn_score);
553  }
554 
555 #if 0
556  printf("\n");
557 #endif
558  }
559 
560  // Result generation.
561 
562  // How we do this depends on the mode we operate in. In SW_SINGLE, we
563  // follow the path from the best node until there is no predecessor
564  // (that is, when we hit a node in row 0), and stop. In SW_MULTIPLE,
565  // we collect all non-redundant common subsequences.
566 
567  if ( params._sw_variant == SW_MULTIPLE )
568  sw_collect_multiple(result, matrix, params);
569  else
570  sw_collect_single(result, matrix, node_max, params);
571 
572  if ( len1 > len2 )
573  sort(result->begin(), result->end(), BroSubstringCmp(0));
574  else
575  sort(result->begin(), result->end(), BroSubstringCmp(1));
576 
577  return result;
578  }
