136 void ContentLine_Analyzer::DoDeliver(int len, const u_char* data)
137  {
138  seq_delivered_in_lines = seq;
139 
140  while ( len > 0 && ! SkipDeliveries() )
141  {
142  if ( (CR_LF_as_EOL & CR_as_EOL) &&
143  last_char == '\r' && *data == '\n' )
144  {
145  // CR is already considered as EOL.
146  // Compress CRLF to just one line termination.
147  //
148  // Note, we test this prior to checking for
149  // "plain delivery" because (1) we might have
150  // made the decision to switch to plain delivery
151  // based on a line terminated with '\r' for
152  // which a '\n' then arrived, and (2) we are
153  // careful when executing plain delivery to
154  // clear last_char once we do so.
155  last_char = *data;
156  --len; ++data; ++seq;
157  ++seq_delivered_in_lines;
158  }
