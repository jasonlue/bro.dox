86  {
87  // Note that it doesn't matter whether or not the character class is

89

104 
105  int j = i + 1;
106 
107  for ( int k = fwd[sym]; k && k < size; k = fwd[k] )
108  { // look for the symbol in the character class
109  for ( ; j < csyms->length(); ++j )
110  {
111  if ( (*csyms)[j] > k )
112  // Since the character class is sorted,
113  // we can stop.
114  break;
115

117  {
118  // We found an old companion of sym
119  // in the ccl. Link it into the new
120  // equivalence class and flag it as
121  // having been processed.
122  bck[k] = new_ec;
123  fwd[new_ec] = k;
124  new_ec = k;
125 
126  // Set flag so we don't reprocess.
127  ccl_flags[j] = 1;
128 
129  // Get next equivalence class member.
130  break;
131  }
132  }
133 
134  if ( j < csyms->length() && (*csyms)[j] == k )
135  // We broke out of the above loop by finding
136  // an old companion - go to the next symbol.
137  continue;
138

140  // equivalence class.
141  bck[k] = old_ec;
142  if ( old_ec != ec_nil )
143  fwd[old_ec] = k;
144 
145  old_ec = k;
146  }
147

149  {
150  bck[sym] = ec_nil;
151  fwd[old_ec] = ec_nil;
152  }
153 
154  fwd[new_ec] = ec_nil;
155

157  for ( ++i; i < csyms->length() && ccl_flags[i]; ++i )
158  // Reset "doesn't need processing" flag.
