190 char* BroString::Render(int format, int* len) const
191  {
192  // Maxmimum character expansion is as \xHH, so a factor of 4.
193  char* s = new char[n*4 + 1]; // +1 is for final '\0'
194  char* sp = s;
195  int tmp_len;
196 
197  for ( int i = 0; i < n; ++i )
198  {
199  if ( b[i] == '\\' && (format & ESC_ESC) )
200  {
201  *sp++ = '\\'; *sp++ = '\\';
202  }
203 
204  else if ( (b[i] == '\'' || b[i] == '"') && (format & ESC_QUOT) )
205  {
206  *sp++ = '\\'; *sp++ = b[i];
207  }
208 
209  else if ( (b[i] < ' ' || b[i] > 126) && (format & ESC_HEX) )
210  {
211  char hex_fmt[16];
212 
213  *sp++ = '\\'; *sp++ = 'x';
214  sprintf(hex_fmt, "%02x", b[i]);
215  *sp++ = hex_fmt[0]; *sp++ = hex_fmt[1];
216  }
217 
218  else if ( (b[i] < ' ' || b[i] > 126) && (format & ESC_DOT) )
219  {
220  *sp++ = '.';
221  }
222 
223  else
224  {
225  *sp++ = b[i];
226  }
227  }
228 
229  *sp++ = '\0'; // NUL-terminate.
230  tmp_len = sp - s;
231 
232  if ( (format & ESC_SER) )
233  {
234  char* result = new char[tmp_len + 16];
235  snprintf(result, tmp_len + 16, "%u ", tmp_len - 1);
236  tmp_len += strlen(result);
237  memcpy(result + strlen(result), s, sp - s);
238  delete [] s;
239  s = result;
240  }
241 
242  if ( len )
243  *len = tmp_len;
244 
245  return s;
246  }
